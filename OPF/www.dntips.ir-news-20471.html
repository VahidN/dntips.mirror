Щ…Ш·Ш§Щ„ШЁ
ЩҶЩҲЫҢШіЩҶШҜЪҜШ§ЩҶ
ЪҜШұЩҲЩҮвҖҢЩҮШ§ЫҢ Щ…Ш·Ш§Щ„ШЁ
ЩҶЩӮШҙЩҮвҖҢЩҮШ§ЫҢ ШұШ§ЩҮ
ШҜЩҲШұЩҮвҖҢЩҮШ§
Ш§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§
ЩҫШұШіШҙвҖҢЩҮШ§
→ вҖ«Ш§ШіШӘШұШ§ШӘЪҳЫҢвҖҢЩҮШ§ЫҢ Ъ©Ш§ШұШўЩ…ШҜ ШЁШұШ§ЫҢ Ш®ЩҲШҜШўЩ…ЩҲШІЫҢ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ ЩҫШі Ш§ШІ ШіШ§Ш№Ш§ШӘ Ъ©Ш§ШұЫҢ
вҖ«ШұШ§ЩҮЩҶЩ…Ш§ЫҢ Щ…ЩҮЩҶШҜШіЫҢ ЩҫШұШ§Щ…ЩҫШӘ ШЁШұШ§ЫҢ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҶЩҲЫҢШіШ§ЩҶ ←
Ш№ЩҶЩҲШ§ЩҶ:

вҖ«Щ…ШҜЫҢШұЫҢШӘ Ш®Ш·Ш§ ШҜШұ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҶЩҲЫҢШіЫҢ ЩҶШ§ЩҮЩ…ЪҜШ§Щ… #C: ШҜШұЪ© Task.FromException


ЩҶЩҲЫҢШіЩҶШҜЩҮ: ЩҲШӯЫҢШҜ ЩҶШөЫҢШұЫҢ
ШӘШ§ШұЫҢШ®: ЫұЫҙЫ°Ыҙ/Ы°Ыі/Ы°Ы· Ы°Ыё:ЫіЫ№
ШўШҜШұШі: www.dntips.ir
ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҶЩҲЫҢШіЫҢ ЩҶШ§ЩҮЩ…ЪҜШ§Щ… (Asynchronous Programming) ШЁЩҮ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҶЩҲЫҢШіШ§ЩҶ Ш§Ш¬Ш§ШІЩҮ Щ…ЫҢвҖҢШҜЩҮШҜ ШӘШ§ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҮШ§ЫҢ ЩҫШ§ШіШ®ЪҜЩҲШӘШұ ЩҲ Ъ©Ш§ШұШўЩ…ШҜШӘШұЫҢ ШЁШіШ§ШІЩҶШҜ. ШЁШ§ Ш§ЫҢЩҶ ШӯШ§Щ„ШҢ Щ…ШҜЫҢШұЫҢШӘ Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ (Exception Handling) ШҜШұ Щ…ШӯЫҢШ·вҖҢЩҮШ§ЫҢ ЩҶШ§ЩҮЩ…ЪҜШ§Щ… Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ ЩҫЫҢЪҶЫҢШҜЪҜЫҢвҖҢЩҮШ§ЫҢЫҢ ШЁЩҮ ЩҮЩ…ШұШ§ЩҮ ШҜШ§ШҙШӘЩҮ ШЁШ§ШҙШҜ. ШҜШұ ШӯШ§Щ„ЫҢ Ъ©ЩҮ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ ШЁЩҮ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ШЁЩ„ЩҲЪ©вҖҢЩҮШ§ЫҢ try-catch ШЁШұШ§ЫҢ Щ…ШҜЫҢШұЫҢШӘ Ш®Ш·Ш§ЩҮШ§ Ш№Ш§ШҜШӘ ШҜШ§ШұЩҶШҜШҢ Щ…ШӘШҜ Task.FromException ШұЩҲЫҢЪ©ШұШҜЫҢ ЩӮШҜШұШӘЩ…ЩҶШҜ ШЁШұШ§ЫҢ Щ…ШҜЫҢШұЫҢШӘ Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ ШҜШұ Ш§Щ„ЪҜЩҲЫҢ ЩҶШ§ЩҮЩ…ЪҜШ§Щ… Щ…ШЁШӘЩҶЫҢ ШЁШұ ШӘШіЪ© (Task-based Asynchronous Pattern - TAP) Ш§ШұШ§ШҰЩҮ Щ…ЫҢвҖҢШҜЩҮШҜ. Ш§ЫҢЩҶ Щ…ЩӮШ§Щ„ЩҮ ШЁЩҮ ШЁШұШұШіЫҢ Ш№Щ…ЫҢЩӮ Task.FromExceptionШҢ Ъ©Ш§ШұШЁШұШҜЩҮШ§ЫҢ ШўЩҶШҢ Ш§Щ„ЪҜЩҲЩҮШ§ЫҢ ЩҫЫҢШҙШұЩҒШӘЩҮШҢ ШЁЩҮШӘШұЫҢЩҶ ШҙЫҢЩҲЩҮвҖҢЩҮШ§ ЩҲ ЫҢЪ© Щ…Ш«Ш§Щ„ Ъ©Ш§ШұШЁШұШҜЫҢ ШҜШұ ШҜЩҶЫҢШ§ЫҢ ЩҲШ§ЩӮШ№ЫҢ Щ…ЫҢвҖҢЩҫШұШҜШ§ШІШҜ.

Щ…ЩӮШҜЩ…ЩҮвҖҢШ§ЫҢ ШЁШұTask.FromException
Task.FromException ЫҢЪ© Щ…ШӘШҜ Ш§ШіШӘШ§ШӘЫҢЪ© Ш§ШіШӘ Ъ©ЩҮ ЫҢЪ© ШҙЫҢШЎ Task ШұШ§ Ш§ЫҢШ¬Ш§ШҜ Щ…ЫҢвҖҢЪ©ЩҶШҜ Ъ©ЩҮ ШҜШұ ЩҲШ¶Ш№ЫҢШӘ Ш®Ш·Ш§ (Faulted State) ШЁЩҮ ЩҫШ§ЫҢШ§ЩҶ ШұШіЫҢШҜЩҮ Ш§ШіШӘ ЩҲ ШҙШ§Щ…Щ„ Ш§ШіШӘШ«ЩҶШ§ЫҢ Щ…ШҙШ®ШөвҖҢШҙШҜЩҮ Ш§ШіШӘ. Ш§ЫҢЩҶ Щ…ШӘШҜ ШЁШ®ШҙЫҢ Ш§ШІ Ъ©ШӘШ§ШЁШ®Ш§ЩҶЩҮ Щ…ЩҲШ§ШІЫҢ ШӘШіЪ© (Task Parallel Library - TPL) Ш§ШіШӘ ЩҲ ШұШ§ЩҮЫҢ ШөШұЫҢШӯ ШЁШұШ§ЫҢ Ш§ЫҢШ¬Ш§ШҜ ЫҢЪ© ШӘШіЪ© Ш®Ш·Ш§Ш®ЩҲШұШҜЩҮ ЩҒШұШ§ЩҮЩ… Щ…ЫҢвҖҢЪ©ЩҶШҜ. ШҜЩҲ ЩҶШіШ®ЩҮ Ш§ШІ Ш§ЫҢЩҶ Щ…ШӘШҜ ЩҲШ¬ЩҲШҜ ШҜШ§ШұШҜ:
  • public static Task FromException(Exception exception): ШЁШұШ§ЫҢ ШӘШіЪ©вҖҢЩҮШ§ЫҢЫҢ Ъ©ЩҮ ЩҶШӘЫҢШ¬ЩҮвҖҢШ§ЫҢ ШЁШ§ШІ ЩҶЩ…ЫҢвҖҢЪҜШұШҜШ§ЩҶЩҶШҜ.
  • public static Task<TResult> FromException<TResult>(Exception exception): ШЁШұШ§ЫҢ ШӘШіЪ©вҖҢЩҮШ§ЫҢЫҢ Ъ©ЩҮ ЫҢЪ© ЩҶШӘЫҢШ¬ЩҮ Ш§ШІ ЩҶЩҲШ№ TResult ШұШ§ ШЁШ§ШІ Щ…ЫҢвҖҢЪҜШұШҜШ§ЩҶЩҶШҜ.

Ъ©Ш§ШұШЁШұШҜ ЩҫШ§ЫҢЩҮ
ШҜШұ ШіЩҶШ§ШұЫҢЩҲЩҮШ§ЫҢ ЩҫШ§ЫҢЩҮШҢ Task.FromException ШЁШұШ§ЫҢ ШӘШЁШҜЫҢЩ„ ЫҢЪ© Ш§ШіШӘШ«ЩҶШ§ЫҢ ЪҜШұЩҒШӘЩҮ ШҙШҜЩҮ ШЁЩҮ ЫҢЪ© ШӘШіЪ© Ш®Ш·Ш§Ш®ЩҲШұШҜЩҮ Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ. ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ Щ…Ш«Ш§Щ„ШҢ ШҜШұ ЫҢЪ© Щ…ШӘШҜ asyncШҢ Ш§ЪҜШұ Ш№Щ…Щ„ЫҢШ§ШӘЫҢ ЩҶШ§Щ…ЩҲЩҒЩӮ ШЁШ§ШҙШҜШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ ШЁЩҮ Ш¬Ш§ЫҢ ЩҫШұШӘШ§ШЁ Щ…ШіШӘЩӮЫҢЩ… Ш§ШіШӘШ«ЩҶШ§ШҢ ЫҢЪ© ШӘШіЪ© Ш®Ш·Ш§Ш®ЩҲШұШҜЩҮ ШұШ§ ШЁШ§ШІЪҜШұШҜШ§ЩҶШҜ:
public async Task ProcessDataAsync()
{
    try
    {
        // Ш№Щ…Щ„ЫҢШ§ШӘ ЩҶШ§ЩҮЩ…ЪҜШ§Щ…
        await DoWorkAsync();
    }
    catch (Exception ex)
    {
        return Task.FromException(ex);
    }
}
ЩҮЩ…ЪҶЩҶЫҢЩҶШҢ ШЁШұШ§ЫҢ ШӘШіЪ©вҖҢЩҮШ§ЫҢ Ш№Щ…ЩҲЩ…ЫҢ (Generic Tasks) Ъ©ЩҮ ШҜШ§ШұШ§ЫҢ ЫҢЪ© ЩҶЩҲШ№ ШЁШ§ШІЪҜШҙШӘЫҢ (Return Type) Ш®Ш§Шө ЩҮШіШӘЩҶШҜШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ Ш§ШІ ЩҶШіШ®ЩҮ Ш№Щ…ЩҲЩ…ЫҢ Task.FromException Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ШұШҜ:
public async Task<string> GetDataAsync()
{
    try
    {
        return await FetchDataAsync();
    }
    catch (Exception ex)
    {
        return Task.FromException<string>(ex);
    }
}
Ш§ЫҢЩҶ ШұЩҲЫҢЪ©ШұШҜ ШӘШ¶Щ…ЫҢЩҶ Щ…ЫҢвҖҢЪ©ЩҶШҜ Ъ©ЩҮ Ш¬ШұЫҢШ§ЩҶ Ъ©ЩҶШӘШұЩ„ ЩҶШ§ЩҮЩ…ЪҜШ§Щ… ШӯЩҒШё ШҙЩҲШҜ ЩҲ Щ…ШөШұЩҒвҖҢЪ©ЩҶЩҶШҜЩҮ ШӘШіЪ© ШЁШӘЩҲШ§ЩҶШҜ Ш§ШіШӘШ«ЩҶШ§ ШұШ§ ШЁЩҮ ШөЩҲШұШӘ Ш§ШіШӘШ§ЩҶШҜШ§ШұШҜ Щ…ШҜЫҢШұЫҢШӘ Ъ©ЩҶШҜ.

Щ…ЩҲШ§ШұШҜ Ш§ШіШӘЩҒШ§ШҜЩҮ ШұШ§ЫҢШ¬Task.FromException
Task.FromException ШҜШұ ШіЩҶШ§ШұЫҢЩҲЩҮШ§ЫҢ Щ…Ш®ШӘЩ„ЩҒЫҢ Ъ©ЩҮ ЩҶЫҢШ§ШІ ШЁЩҮ Щ…ШҜЫҢШұЫҢШӘ ШөШұЫҢШӯ Ш®Ш·Ш§ ШҜШұ ШӘШіЪ©вҖҢЩҮШ§ЫҢ ЩҶШ§ЩҮЩ…ЪҜШ§Щ… Ш§ШіШӘШҢ ШЁШіЫҢШ§Шұ Щ…ЩҒЫҢШҜ Ш§ШіШӘ.

ШіЩҶШ§ШұЫҢЩҲЩҮШ§ЫҢ Ш§Ш№ШӘШЁШ§ШұШіЩҶШ¬ЫҢ (Validation Scenarios)
ШҜШұ Щ…ЩҲШ§ШұШҜЫҢ Ъ©ЩҮ ЩҫШ§ШұШ§Щ…ШӘШұЩҮШ§ЫҢ ЩҲШұЩҲШҜЫҢ ЩӮШЁЩ„ Ш§ШІ Ш§Ш¬ШұШ§ЫҢ Щ…ЩҶШ·ЩӮ Ш§ШөЩ„ЫҢ ЩҶЫҢШ§ШІ ШЁЩҮ Ш§Ш№ШӘШЁШ§ШұШіЩҶШ¬ЫҢ ШҜШ§ШұЩҶШҜШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ Ш§ШІ Task.FromException ШЁШұШ§ЫҢ ШЁШ§ШІЪҜШұШҜШ§ЩҶШҜЩҶ ШіШұЫҢШ№ ЫҢЪ© ШӘШіЪ© Ш®Ш·Ш§Ш®ЩҲШұШҜЩҮ ШЁШ§ Ш§ШіШӘШ«ЩҶШ§ЫҢ Ш§Ш№ШӘШЁШ§ШұШіЩҶШ¬ЫҢ Щ…ЩҶШ§ШіШЁ Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ШұШҜ. Ш§ЫҢЩҶ Ъ©Ш§Шұ Ш§ШІ Ш§ШҜШ§Щ…ЩҮ Ш§Ш¬ШұШ§ЫҢ Ъ©ШҜ ШЁШ§ ШҜШ§ШҜЩҮвҖҢЩҮШ§ЫҢ ЩҶШ§Щ…Ш№ШӘШЁШұ Ш¬Щ„ЩҲЪҜЫҢШұЫҢ Щ…ЫҢвҖҢЪ©ЩҶШҜ:
public async Task<ValidationResult> ValidateDataAsync(DataModel data)
{
    if (data == null)
    {
        return await Task.FromException<ValidationResult>(
            new ArgumentNullException(nameof(data)));
    }
    
    if (!IsValid(data))
    {
        return await Task.FromException<ValidationResult>(
            new ValidationException("Invalid data format"));
    }
    
    return await PerformValidationAsync(data);
}

Ш§ЩҶШӘШҙШ§Шұ Ш®Ш·Ш§ (Error Propagation)
ШІЩ…Ш§ЩҶЫҢ Ъ©ЩҮ ЩҶЫҢШ§ШІ Ш§ШіШӘ Ш®Ш·Ш§ЩҮШ§ ШЁЩҮ ШіЩ…ШӘ ШЁШ§Щ„Ш§ ШҜШұ ШІЩҶШ¬ЫҢШұЩҮ ЩҒШұШ§Ш®ЩҲШ§ЩҶЫҢ (Call Stack) Щ…ЩҶШӘШҙШұ ШҙЩҲЩҶШҜШҢ Task.FromException ЫҢЪ© ШұЩҲШҙ ШӘЩ…ЫҢШІ ШұШ§ ЩҒШұШ§ЩҮЩ… Щ…ЫҢвҖҢЪ©ЩҶШҜ. Ш§ЫҢЩҶ ШұЩҲЫҢЪ©ШұШҜ ШЁЩҮ ЩҲЫҢЪҳЩҮ ШҜШұ Щ„Ш§ЫҢЩҮвҖҢЩҮШ§ЫҢ Щ…Ш®ШӘЩ„ЩҒ ЫҢЪ© ШЁШұЩҶШ§Щ…ЩҮ Щ…ЩҒЫҢШҜ Ш§ШіШӘ Ъ©ЩҮ ЩҮШұ Щ„Ш§ЫҢЩҮ Щ…ШіШҰЩҲЩ„ Щ…ШҜЫҢШұЫҢШӘ Ш®Ш·Ш§ЩҮШ§ЫҢ Ш®Ш§Шө Ш®ЩҲШҜ Ш§ШіШӘ Ш§Щ…Ш§ ШҜШұ ЩҶЩҮШ§ЫҢШӘ Ш®Ш·Ш§ ШұШ§ ШЁШұШ§ЫҢ Щ…ШҜЫҢШұЫҢШӘ ЩҶЩҮШ§ЫҢЫҢ ШЁЩҮ Щ„Ш§ЫҢЩҮвҖҢЩҮШ§ЫҢ ШЁШ§Щ„Ш§ШӘШұ Щ…ЩҶШӘЩӮЩ„ Щ…ЫҢвҖҢЪ©ЩҶШҜ:
public async Task<Result> ProcessWithErrorHandlingAsync()
{
    try
    {
        var data = await LoadDataAsync();
        return await ProcessDataAsync(data);
    }
    catch (DataLoadException ex)
    {
        // _logger.LogError(ex, "Failed to load data"); // ЩҒШұШ¶ ШЁШұ ЩҲШ¬ЩҲШҜ Щ„Ш§ЪҜШұ
        return await Task.FromException<Result>(ex);
    }
    catch (ProcessingException ex)
    {
        // _logger.LogError(ex, "Failed to process data"); // ЩҒШұШ¶ ШЁШұ ЩҲШ¬ЩҲШҜ Щ„Ш§ЪҜШұ
        return await Task.FromException<Result>(ex);
    }
}

Ш§ЩҶЩҲШ§Ш№ Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ЫҢ ШіЩҒШ§ШұШҙЫҢ (Custom Exception Types)
ЩҮЩҶЪҜШ§Щ… Ъ©Ш§Шұ ШЁШ§ Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ЫҢ ШіЩҒШ§ШұШҙЫҢ Ъ©ЩҮ Щ…ЩҶШ·ЩӮ ШӘШ¬Ш§ШұЫҢ Ш®Ш§ШөЫҢ ШұШ§ ЩҶШҙШ§ЩҶ Щ…ЫҢвҖҢШҜЩҮЩҶШҜШҢ Task.FromException ШЁЩҮ ШҙЩ…Ш§ Ш§Щ…Ъ©Ш§ЩҶ Щ…ЫҢвҖҢШҜЩҮШҜ ШӘШ§ Ш§ЫҢЩҶ Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ ШұШ§ ШЁЩҮ Ш·ЩҲШұ ШөШұЫҢШӯ ШҜШұ ЫҢЪ© ШӘШіЪ© Ш®Ш·Ш§Ш®ЩҲШұШҜЩҮ Ъ©ЩҫШіЩҲЩ„ЩҮ Ъ©ЩҶЫҢШҜ:
public class BusinessException : Exception
{
    public BusinessException(string message) : base(message) { }
}

public async Task<BusinessResult> ProcessBusinessLogicAsync()
{
    if (!IsBusinessRuleValid())
    {
        return await Task.FromException<BusinessResult>(
            new BusinessException("Business rule violation"));
    }
    
    return await ExecuteBusinessLogicAsync();
}

Ш§Щ„ЪҜЩҲЩҮШ§ЫҢ ЩҫЫҢШҙШұЩҒШӘЩҮ ШЁШ§Task.FromException
Task.FromException Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ ШҜШұ ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ Ш§Щ„ЪҜЩҲЩҮШ§ЫҢ ЩҫЫҢЪҶЫҢШҜЩҮвҖҢШӘШұ Щ…ШҜЫҢШұЫҢШӘ Ш®Ш·Ш§ ШҜШұ ШіЫҢШіШӘЩ…вҖҢЩҮШ§ЫҢ ШӘЩҲШІЫҢШ№вҖҢШҙШҜЩҮ ЩҲ Щ…ЩӮШ§ЩҲЩ… ЩҶЫҢШІ ШЁЩҮ Ъ©Ш§Шұ ШұЩҲШҜ.

ШӘШ¬Щ…ЫҢШ№ Ш§ШіШӘШ«ЩҶШ§ (Exception Aggregation)
ШҜШұ ШіЩҶШ§ШұЫҢЩҲЩҮШ§ЫҢЫҢ Ъ©ЩҮ ЪҶЩҶШҜЫҢЩҶ Ш№Щ…Щ„ЫҢШ§ШӘ Щ…ШіШӘЩӮЩ„ Щ…Щ…Ъ©ЩҶ Ш§ШіШӘ ШҙЪ©ШіШӘ ШЁШ®ЩҲШұЩҶШҜ ЩҲ ШҙЩ…Ш§ ЩҶЫҢШ§ШІ ШЁЩҮ Ш¬Щ…Ш№вҖҢШўЩҲШұЫҢ ШӘЩ…Ш§Щ… Ш®Ш·Ш§ЩҮШ§ ШҜШ§ШұЫҢШҜШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЫҢШҜ Ш§ШІ AggregateException Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ЩҶЫҢШҜ:
public async Task<Result> ProcessMultipleItemsAsync(IEnumerable<Item> items)
{
    var exceptions = new List<Exception>();
    
    foreach (var item in items)
    {
        try
        {
            await ProcessItemAsync(item);
        }
        catch (Exception ex)
        {
            exceptions.Add(ex);
        }
    }
    
    if (exceptions.Any())
    {
        return await Task.FromException<Result>(
            new AggregateException(exceptions));
    }
    
    return new Result();
}
ШҜШұ Ш§ЫҢЩҶ Ш§Щ„ЪҜЩҲШҢ ШӯШӘЫҢ Ш§ЪҜШұ ШЁШұШ®ЫҢ Ш§ШІ Ш№Щ…Щ„ЫҢШ§ШӘ ШҙЪ©ШіШӘ ШЁШ®ЩҲШұЩҶШҜШҢ ШЁШұЩҶШ§Щ…ЩҮ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ ШЁЩҮ ЩҫШұШҜШ§ШІШҙ Ш§ШҜШ§Щ…ЩҮ ШҜЩҮШҜ ЩҲ ШҜШұ ЩҶЩҮШ§ЫҢШӘ ШӘЩ…Ш§Щ… Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ЫҢ ШұШ® ШҜШ§ШҜЩҮ ШұШ§ ЪҜШІШ§ШұШҙ Ъ©ЩҶШҜ.

Ш§Щ„ЪҜЩҲЫҢ ШӘЩ„Ш§Шҙ Щ…Ш¬ШҜШҜ (Retry Pattern)
ЩҮЩҶЪҜШ§Щ… ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ ЫҢЪ© Ш§Щ„ЪҜЩҲЫҢ ШӘЩ„Ш§Шҙ Щ…Ш¬ШҜШҜ ШЁШұШ§ЫҢ Ш№Щ…Щ„ЫҢШ§ШӘвҖҢЩҮШ§ЫҢ Щ…ШіШӘШ№ШҜ Ш®Ш·Ш§ШҢ Task.FromException Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ ШЁШұШ§ЫҢ ШЁШ§ШІЪҜШұШҜШ§ЩҶШҜЩҶ ЫҢЪ© Ш§ШіШӘШ«ЩҶШ§ЫҢ Ш®Ш§Шө ЩҫШі Ш§ШІ Ш§ШӘЩ…Ш§Щ… ШӘЩ…Ш§Щ… ШӘЩ„Ш§ШҙвҖҢЩҮШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ ШҙЩҲШҜ:
public async Task<Result> ExecuteWithRetryAsync(
    Func<Task<Result>> operation,
    int maxRetries)
{
    for (int i = 0; i < maxRetries; i++)
    {
        try
        {
            return await operation();
        }
        catch (Exception ex)
        {
            if (i == maxRetries - 1)
            {
                return await Task.FromException<Result>(
                    new RetryException($"Failed after {maxRetries} attempts", ex));
            }
            
            await Task.Delay(TimeSpan.FromSeconds(Math.Pow(2, i))); // ШӘШ§Ш®ЫҢШұ ЩҶЩ…Ш§ЫҢЫҢ
        }
    }
    
    // Ш§ЫҢЩҶ Ш®Ш· ЩҶШЁШ§ЫҢШҜ ЩӮШ§ШЁЩ„ ШҜШіШӘШұШіЫҢ ШЁШ§ШҙШҜ Щ…ЪҜШұ Ш§ЫҢЩҶЪ©ЩҮ Щ…ШҙЪ©Щ„ЫҢ ШҜШұ Щ…ЩҶШ·ЩӮ ЩҲШ¬ЩҲШҜ ШҜШ§ШҙШӘЩҮ ШЁШ§ШҙШҜ
    return await Task.FromException<Result>(
        new InvalidOperationException("Should not reach here"));
}
Ш§ЫҢЩҶ Ш§Щ„ЪҜЩҲ ШЁЩҮ ШіЫҢШіШӘЩ…вҖҢЩҮШ§ Ш§Ш¬Ш§ШІЩҮ Щ…ЫҢвҖҢШҜЩҮШҜ ШӘШ§ Ш§ШІ Ш®Ш·Ш§ЩҮШ§ЫҢ Щ…ЩҲЩӮШӘЫҢ ШЁШ§ШІЫҢШ§ШЁЫҢ ШҙЩҲЩҶШҜ ЩҲ ШҜШұ ШөЩҲШұШӘ ШҙЪ©ШіШӘвҖҢЩҮШ§ЫҢ ЩҫШ§ЫҢШҜШ§ШұШҢ Ш®Ш·Ш§ ШұШ§ ШЁЩҮ ШҙЪ©Щ„ Ъ©ЩҶШӘШұЩ„вҖҢШҙШҜЩҮвҖҢШ§ЫҢ ЪҜШІШ§ШұШҙ Ъ©ЩҶЩҶШҜ.

Ш§Щ„ЪҜЩҲЫҢ Щ…ШҜШ§ШұШҙЪ©ЩҶ (Circuit Breaker Pattern)
Ш§Щ„ЪҜЩҲЫҢ Щ…ШҜШ§ШұШҙЪ©ЩҶ ШЁШұШ§ЫҢ Ш¬Щ„ЩҲЪҜЫҢШұЫҢ Ш§ШІ ЩҒШұШ§Ш®ЩҲШ§ЩҶЫҢ Щ…Ъ©ШұШұ ШіШұЩҲЫҢШівҖҢЩҮШ§ЫҢ Щ…Ш№ЫҢЩҲШЁ Ш·ШұШ§ШӯЫҢ ШҙШҜЩҮ Ш§ШіШӘ. Task.FromException Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ ШЁШ®ШҙЫҢ Ш§ШІ Ш§ЫҢЩҶ Ш§Щ„ЪҜЩҲ ШЁШұШ§ЫҢ ШЁШ§ШІЪҜШұШҜШ§ЩҶШҜЩҶ ЩҒЩҲШұЫҢ ЫҢЪ© ШӘШіЪ© Ш®Ш·Ш§Ш®ЩҲШұШҜЩҮ ШІЩ…Ш§ЩҶЫҢ Ъ©ЩҮ Щ…ШҜШ§Шұ ШЁШ§ШІ (Open) Ш§ШіШӘШҢ Ш§ШіШӘЩҒШ§ШҜЩҮ ШҙЩҲШҜ:
public class CircuitBreaker
{
    // private readonly ILogger<CircuitBreaker> _logger; // ЩҒШұШ¶ ШЁШұ ЩҲШ¬ЩҲШҜ Щ„Ш§ЪҜШұ
    private bool _isOpen;
    
    public async Task<Result> ExecuteAsync(Func<Task<Result>> operation)
    {
        if (_isOpen)
        {
            return await Task.FromException<Result>(
                new CircuitBreakerException("Circuit is open"));
        }
        
        try
        {
            return await operation();
        }
        catch (Exception ex)
        {
            _isOpen = true;
            // _logger.LogError(ex, "Circuit breaker opened"); // ЩҒШұШ¶ ШЁШұ ЩҲШ¬ЩҲШҜ Щ„Ш§ЪҜШұ
            return await Task.FromException<Result>(ex);
        }
    }
}
Ш§ЫҢЩҶ Ш§Щ„ЪҜЩҲ ШЁЩҮ ШіЫҢШіШӘЩ… Ъ©Щ…Ъ© Щ…ЫҢвҖҢЪ©ЩҶШҜ ШӘШ§ Ш§ШІ ЩҒШұЩҲЩҫШ§ШҙЫҢ (Cascade Failure) Ш¬Щ„ЩҲЪҜЫҢШұЫҢ Ъ©ЩҶШҜ ЩҲ ШЁЩҮ ШіШұЩҲЫҢШі Щ…Ш№ЫҢЩҲШЁ ЩҒШұШөШӘ ШЁШ§ШІЫҢШ§ШЁЫҢ ШҜЩҮШҜ.

ШЁЩҮШӘШұЫҢЩҶ ШҙЫҢЩҲЩҮвҖҢЩҮШ§ (Best Practices)
ШЁШұШ§ЫҢ Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ШӨШ«Шұ Ш§ШІ Task.FromException ЩҲ ШӯЩҒШё Ъ©ШҜ ЩӮШ§ШЁЩ„ ЩҶЪҜЩҮШҜШ§ШұЫҢШҢ ШұШ№Ш§ЫҢШӘ ШЁЩҮШӘШұЫҢЩҶ ШҙЫҢЩҲЩҮвҖҢЩҮШ§ Ш¶ШұЩҲШұЫҢ Ш§ШіШӘ.

Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ЫҢ Ш®Ш§Шө (Specific Exceptions)
ЩҮЩ…ЫҢШҙЩҮ Ш§ШІ Ш®Ш§ШөвҖҢШӘШұЫҢЩҶ ЩҶЩҲШ№ Ш§ШіШӘШ«ЩҶШ§ЫҢ Щ…Щ…Ъ©ЩҶ Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ЩҶЫҢШҜ. Ш§ЫҢЩҶ Ъ©Ш§Шұ ШЁЩҮ Щ…ШөШұЩҒвҖҢЪ©ЩҶЩҶШҜЪҜШ§ЩҶ ШӘШіЪ© Ъ©Щ…Ъ© Щ…ЫҢвҖҢЪ©ЩҶШҜ ШӘШ§ ЩҶЩҲШ№ Щ…ШҙЪ©Щ„ ШұШ§ ШЁЩҮ Ш·ЩҲШұ ШҜЩӮЫҢЩӮвҖҢШӘШұЫҢ ШӘШҙШ®ЫҢШө ШҜШ§ШҜЩҮ ЩҲ Щ…ШҜЫҢШұЫҢШӘ Ъ©ЩҶЩҶШҜ. ЩҫШұШӘШ§ШЁ ЫҢЪ© Exception Ш№Щ…ЩҲЩ…ЫҢ Ъ©Щ…ШӘШұ Щ…ЩҒЫҢШҜ Ш§ШіШӘ ШІЫҢШұШ§ Ш§Ш·Щ„Ш§Ш№Ш§ШӘ Ъ©Ш§ЩҒЫҢ ШҜШұ Щ…ЩҲШұШҜ Щ…Ш§ЩҮЫҢШӘ Ш®Ш·Ш§ Ш§ШұШ§ШҰЩҮ ЩҶЩ…ЫҢвҖҢШҜЩҮШҜ:
// Ш®ЩҲШЁ
return Task.FromException(new ValidationException("Invalid input"));

// ШЁШҜ
return Task.FromException(new Exception("Invalid input"));

ШҜШұШ¬ Ш§Ш·Щ„Ш§Ш№Ш§ШӘ Щ…ШӘЩҶЫҢ (Contextual Information)
ЩҮЩ…ШұШ§ЩҮ ШЁШ§ Ш§ШіШӘШ«ЩҶШ§ШҢ ЩҫЫҢШ§Щ…вҖҢЩҮШ§ЫҢ Ш®Ш·Ш§ЫҢ Щ…Ш№ЩҶЫҢвҖҢШҜШ§Шұ ЩҲ Ш§Ш·Щ„Ш§Ш№Ш§ШӘ Щ…ШӘЩҶЫҢ Щ…ШұШӘШЁШ· ШұШ§ Ш§ШұШ§ШҰЩҮ ШҜЩҮЫҢШҜ. Ш§ЫҢЩҶ Ш§Ш·Щ„Ш§Ш№Ш§ШӘ ШЁЩҮ ШұЩҒШ№ Ш§ШҙЪ©Ш§Щ„ (Debugging) ЩҲ ШҜШұЪ© ШіШұЫҢШ№вҖҢШӘШұ Ш№Щ„ШӘ Ш®Ш·Ш§ Ъ©Щ…Ъ© Щ…ЫҢвҖҢЪ©ЩҶШҜ:
public async Task<Result> ProcessWithContextAsync(InputData data)
{
    try
    {
        return await ProcessAsync(data);
    }
    catch (Exception ex)
    {
        return await Task.FromException<Result>(
            new ProcessingException(
                $"Failed to process data with ID: {data.Id}",
                ex));
    }
}

Ш«ШЁШӘ Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ (Logging Exceptions)
ЩҮЩ…ЫҢШҙЩҮ Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ ШұШ§ ЩӮШЁЩ„ Ш§ШІ Ш§ЫҢШ¬Ш§ШҜ ШӘШіЪ©вҖҢЩҮШ§ЫҢ Ш®Ш·Ш§Ш®ЩҲШұШҜЩҮ Ш«ШЁШӘ (Log) Ъ©ЩҶЫҢШҜ. Ш«ШЁШӘ Ш®Ш·Ш§ЩҮШ§ ШЁШұШ§ЫҢ ЩҫШ§ЫҢШҙ (Monitoring) ЩҲ Ш№ЫҢШЁвҖҢЫҢШ§ШЁЫҢ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҮШ§ ШҜШұ Щ…ШӯЫҢШ·вҖҢЩҮШ§ЫҢ ШӘЩҲЩ„ЫҢШҜ (Production Environments) ШЁШіЫҢШ§Шұ Щ…ЩҮЩ… Ш§ШіШӘ. Ш§ЫҢЩҶ Ъ©Ш§Шұ ШЁЩҮ ШҙЩ…Ш§ Ш§Щ…Ъ©Ш§ЩҶ Щ…ЫҢвҖҢШҜЩҮШҜ ШӘШ§ Ш§Щ„ЪҜЩҲЩҮШ§ЫҢ Ш®Ш·Ш§ ШұШ§ ШҙЩҶШ§ШіШ§ЫҢЫҢ Ъ©ШұШҜЩҮ ЩҲ ШЁЩҮ Щ…ШҙЪ©Щ„Ш§ШӘ ШЁЩҮ ШіШұШ№ШӘ ЩҫШ§ШіШ® ШҜЩҮЫҢШҜ:
public async Task<Result> ProcessWithLoggingAsync()
{
    try
    {
        return await ProcessAsync();
    }
    catch (Exception ex)
    {
        // _logger.LogError(ex, "Operation failed"); // ЩҒШұШ¶ ШЁШұ ЩҲШ¬ЩҲШҜ Щ„Ш§ЪҜШұ
        return await Task.FromException<Result>(ex);
    }
}

Щ…Ш«Ш§Щ„ Ъ©Ш§ШұШЁШұШҜЫҢ ШҜШұ ШҜЩҶЫҢШ§ЫҢ ЩҲШ§ЩӮШ№ЫҢ
ШЁШұШ§ЫҢ ШҜШұЪ© ШЁЩҮШӘШұ ЩҶШӯЩҲЩҮ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Task.FromException ШҜШұ ЫҢЪ© ШіЩҶШ§ШұЫҢЩҲЫҢ ЩҲШ§ЩӮШ№ЫҢШҢ ЫҢЪ© ШіШұЩҲЫҢШі ЩҫШұШҜШ§ШІШҙ ШҜШ§ШҜЩҮ ШұШ§ ШҜШұ ЩҶШёШұ ШЁЪҜЫҢШұЫҢШҜ:
public class DataProcessingService
{
    // private readonly ILogger<DataProcessingService> _logger; // ЩҒШұШ¶ ШЁШұ ЩҲШ¬ЩҲШҜ Щ„Ш§ЪҜШұ
    private readonly IDataValidator _validator;
    private readonly IDataProcessor _processor;
    
    public DataProcessingService(IDataValidator validator, IDataProcessor processor) // ШіШ§ШІЩҶШҜЩҮ
    {
        _validator = validator;
        _processor = processor;
    }

    public async Task<ProcessedData> ProcessDataAsync(RawData data)
    {
        try
        {
            // Ш§Ш№ШӘШЁШ§ШұШіЩҶШ¬ЫҢ ЩҲШұЩҲШҜЫҢ
            if (!await _validator.ValidateAsync(data))
            {
                // _logger.LogWarning("Invalid data received: {@Data}", data); // ЩҒШұШ¶ ШЁШұ ЩҲШ¬ЩҲШҜ Щ„Ш§ЪҜШұ
                return await Task.FromException<ProcessedData>(
                    new ValidationException("Invalid data format"));
            }
            
            // ЩҫШұШҜШ§ШІШҙ ШҜШ§ШҜЩҮ
            var result = await _processor.ProcessAsync(data);
            
            // Ш§Ш№ШӘШЁШ§ШұШіЩҶШ¬ЫҢ Ш®ШұЩҲШ¬ЫҢ
            if (!IsValid(result)) // ЩҒШұШ¶ ШЁШұ ЩҲШ¬ЩҲШҜ Щ…ШӘШҜ IsValid
            {
                // _logger.LogError("Invalid result generated: {@Result}", result); // ЩҒШұШ¶ ШЁШұ ЩҲШ¬ЩҲШҜ Щ„Ш§ЪҜШұ
                return await Task.FromException<ProcessedData>(
                    new ProcessingException("Invalid result generated"));
            }
            
            return result;
        }
        catch (Exception ex)
        {
            // _logger.LogError(ex, "Failed to process data: {@Data}", data); // ЩҒШұШ¶ ШЁШұ ЩҲШ¬ЩҲШҜ Щ„Ш§ЪҜШұ
            return await Task.FromException<ProcessedData>(ex);
        }
    }
}
ШҜШұ Ш§ЫҢЩҶ Щ…Ш«Ш§Щ„ШҢ ШіШұЩҲЫҢШі DataProcessingService Ш§ШІ Task.FromException ШҜШұ ЪҶЩҶШҜЫҢЩҶ ЩҶЩӮШ·ЩҮ Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶШҜ:
  • Ш§ЪҜШұ ШҜШ§ШҜЩҮвҖҢЩҮШ§ЫҢ ЩҲШұЩҲШҜЫҢ ЩҶШ§Щ…Ш№ШӘШЁШұ ШЁШ§ШҙЩҶШҜШҢ ЫҢЪ© ValidationException ШұШ§ ШҜШұ ЫҢЪ© ШӘШіЪ© Ш®Ш·Ш§Ш®ЩҲШұШҜЩҮ ШЁШ§ШІЩ…ЫҢвҖҢЪҜШұШҜШ§ЩҶШҜ.
  • Ш§ЪҜШұ ЩҶШӘЫҢШ¬ЩҮ ЩҫШұШҜШ§ШІШҙ ЩҶШ§Щ…Ш№ШӘШЁШұ ШЁШ§ШҙШҜШҢ ЫҢЪ© ProcessingException ШұШ§ ШЁШ§ШІЩ…ЫҢвҖҢЪҜШұШҜШ§ЩҶШҜ.
  • ШЁШұШ§ЫҢ ЩҮШұ Ш§ШіШӘШ«ЩҶШ§ЫҢ ШҜЫҢЪҜШұЫҢ Ъ©ЩҮ ШҜШұ Ш·ЩҲЩ„ ЩҫШұШҜШ§ШІШҙ ШұШ® Щ…ЫҢвҖҢШҜЩҮШҜШҢ Ш§ШіШӘШ«ЩҶШ§ ШұШ§ Ш«ШЁШӘ Ъ©ШұШҜЩҮ ЩҲ ШўЩҶ ШұШ§ ШҜШұ ЫҢЪ© ШӘШіЪ© Ш®Ш·Ш§Ш®ЩҲШұШҜЩҮ Ъ©ЩҫШіЩҲЩ„ЩҮ Щ…ЫҢвҖҢЪ©ЩҶШҜ. Ш§ЫҢЩҶ ШұЩҲЫҢЪ©ШұШҜ ШӘШ¶Щ…ЫҢЩҶ Щ…ЫҢвҖҢЪ©ЩҶШҜ Ъ©ЩҮ ШӘЩ…Ш§Щ…ЫҢ Ш®Ш·Ш§ЩҮШ§ ШЁЩҮ Ш·ЩҲШұ ШіШ§ШІЪҜШ§Шұ ЩҲ ЩӮШ§ШЁЩ„ ЩҫЫҢШҙвҖҢШЁЫҢЩҶЫҢ ШҜШұ Ш·ЩҲЩ„ ЩҒШұШ§ЫҢЩҶШҜ ЩҶШ§ЩҮЩ…ЪҜШ§Щ… Щ…ШҜЫҢШұЫҢШӘ Щ…ЫҢвҖҢШҙЩҲЩҶШҜ.

Ш§ЫҢЩҶ Щ…ШӘШҜ ШұШ§ЩҮЫҢ ШӘЩ…ЫҢШІ ЩҲ Ъ©Ш§ШұШўЩ…ШҜ ШЁШұШ§ЫҢ Ш§ЩҶШӘШҙШ§Шұ Ш®Ш·Ш§ЩҮШ§ ЩҲ ШӯЩҒШё ШІЩ…ЫҢЩҶЩҮ ЩҶШ§ЩҮЩ…ЪҜШ§Щ… ЩҒШұШ§ЩҮЩ… Щ…ЫҢвҖҢЪ©ЩҶШҜ. ШЁШ§ ШҜШұЪ© ШІЩ…Ш§ЩҶЫҢ Ъ©ЩҮ ШЁШ§ЫҢШҜ Ш§ШІ ШўЩҶ Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ШұШҜ ЩҲ ШЁШ§ ШұШ№Ш§ЫҢШӘ ШЁЩҮШӘШұЫҢЩҶ ШҙЫҢЩҲЩҮвҖҢЩҮШ§ Щ…Ш§ЩҶЩҶШҜ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ЫҢ Ш®Ш§ШөШҢ ЪҜЩҶШ¬Ш§ЩҶШҜЩҶ Ш§Ш·Щ„Ш§Ш№Ш§ШӘ Щ…ШӘЩҶЫҢ ЩҲ Ш«ШЁШӘ Ш®Ш·Ш§ЩҮШ§ШҢ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЩҶШҜ Ъ©ШҜЫҢ Щ…ЩӮШ§ЩҲЩ…вҖҢШӘШұ ЩҲ ЩӮШ§ШЁЩ„ ЩҶЪҜЩҮШҜШ§ШұЫҢ Ш§ЫҢШ¬Ш§ШҜ Ъ©ЩҶЩҶШҜ. ШҜШұ ШӯШ§Щ„ЫҢ Ъ©ЩҮ Task.FromException Щ…Щ…Ъ©ЩҶ Ш§ШіШӘ ШҜШұ ЩҮШұ Ш№Щ…Щ„ЫҢШ§ШӘ ЩҶШ§ЩҮЩ…ЪҜШ§Щ… Щ…ЩҲШұШҜ ЩҶЫҢШ§ШІ ЩҶШЁШ§ШҙШҜШҢ ШӘШіЩ„Ш· ШЁШұ ШўЩҶ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ ШЁЩҮ Ш·ЩҲШұ ЩӮШ§ШЁЩ„ ШӘЩҲШ¬ЩҮЫҢ Ъ©ЫҢЩҒЫҢШӘ ЩҲ Ш§ШіШӘШӯЪ©Ш§Щ… ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҮШ§ЫҢ .NET ШұШ§ ШЁЩҮШЁЩҲШҜ ШЁШ®ШҙШҜ.
