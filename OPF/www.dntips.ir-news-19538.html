Fewer Pull Requests with Dependabot groups
version: 2
updates:
- package-ecosystem: nuget
  directory: /
  schedule:
    interval: daily
  open-pull-requests-limit: 10
  groups:
    tests:
      patterns: ["*"]
      update-types: ["minor", "patch"]
