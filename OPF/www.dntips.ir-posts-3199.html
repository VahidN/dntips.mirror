ЫҢЪ© ЩҫШұЩҲЪҳЩҮвҖҢЫҢ ASP.NET Core ШұШ§ ШЁШ§ ЩӮШұШ§Шұ ШҜШ§ШҜЩҶ ЩҶШіШ®ЩҮвҖҢЫҢ NET Core. ШЁШұ ШұЩҲЫҢ 3.1 ЩҲ ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ЩӮШ§Щ„ШЁ Model View Controller Ш§ЫҢШ¬Ш§ШҜ Ъ©ЩҶЫҢШҜ. ШҜШұ Ш§ЫҢЩҶШ¬Ш§ ЩҶШ§Щ… ЩҫШұЩҲЪҳЩҮ ШұШ§ AspNetCoreDependencyInjection ЪҜШ°Ш§ШҙШӘЩҮвҖҢШ§Щ…. ШӯШ§Щ„Ш§ ШҜШұ  ЩҫЩҲШҙЩҮвҖҢЫҢ ModelsШҢ ЩҒШ§ЫҢЩ„ЫҢ ШұШ§ ШЁШ§ ЩҶШ§Щ… HomeViewModel.cs ШЁШ§ Щ…ШӯШӘЩҲЫҢШ§ШӘЫҢ ШЁЩҮ ШөЩҲШұШӘ ШІЫҢШұ Ш§Ш¶Ш§ЩҒЩҮ Ъ©ЩҶЫҢШҜ:
public class HomeViewModel
{
     public string Id { get; set; }
     public string Message { get; set; }
     public DateTime DateTime { get; set; }
}

ЩҲ ЩҮЩ…ЪҶЩҶЫҢЩҶ ЩҒШ§ЫҢЩ„ GuidHelper.cs ШұШ§ ЩҶЫҢШІ Ш§Ш¶Ш§ЩҒЩҮ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ…: 
namespace AspNetCoreDependencyInjection.Helpers
{
    public class GuidProvider
    {
        private readonly Guid _serviceGuid;

        public GuidProvider()
        {
            _serviceGuid = Guid.NewGuid();
        }

        public Guid GetNewGuid() => Guid.NewGuid();

        public string GetGuidAsFormatedString(string prefix = "") => getFormatedGuid(_serviceGuid, prefix);

        private string getFormatedGuid(Guid guid, string prefix = "")
        {
            var guidString = guid.GetHashCode().ToString("x");
            if (string.IsNullOrEmpty(prefix) == false)
                guidString = new StringBuilder($"{prefix}-").Append(guidString).ToString();
            return guidString;
        }
    }
}

ШӯШ§Щ„Ш§ ШҜШұЩҲЩҶ Ъ©ЩҶШӘШұЩ„ HomeControllerШҢ Ш§ЫҢЩҶ ШӘШәЫҢЫҢШұШ§ШӘ ШұШ§ Ш§ЩҶШ¬Ш§Щ… Щ…ЫҢвҖҢШҜЩҮЫҢЩ…: 

private readonly ILogger<HomeController> _logger;
private readonly MessageServiceAA _messageService;
private readonly GuidProvider _ guidProvider;

public HomeController(ILogger<HomeController> logger)
{
            _logger = logger;
            _messageService = new MessageServiceAA();
            _guidProvider = new GuidProvider();
}

public IActionResult Index()
{
            var model = new HomeViewModel()
            {
                Id = _ guidProvider.GetGuidAsFormatedString(),
                Message = _messageService.Message(),
                DateTime = DateTime.Now,
            };
            return View(model);
}

ЩҮЩ…Ш§ЩҶШ·ЩҲШұ Ъ©ЩҮ Щ…ЫҢвҖҢШЁЫҢЩҶЫҢШҜШҢ ШҜШұ Ъ©ШҜ ШЁШ§Щ„Ш§ШҢ Ъ©ЩҶШӘШұЩ„Шұ HomeControllerШҢ ШЁЩҮ ШҜЩҲ ШҙЫҢШЎ Ш§ШІ Ъ©Щ„Ш§ШівҖҢЩҮШ§ ЩҲ ЫҢШ§ ШіШұЩҲЫҢШівҖҢЩҮШ§ЫҢ GuidProvider ЩҲ MessageServiceAA ШЁЩҮ ШөЩҲШұШӘ Щ…ШіШӘЩӮЫҢЩ… ЩҲШ§ШЁШіШӘЩҮ ШҙШҜЩҮвҖҢШ§ШіШӘ ЩҲ ШЁШ§ ЩҮШұ ШӘШәЫҢЫҢШұЫҢ ШҜШұ ЩҮШұ Ъ©ШҜШ§Щ… Ш§ШІ Ш§ЫҢЩҶ ШіШұЩҲЫҢШіЩҮШ§ШҢ ШЁШ§ЫҢШҜ ШҜЩҲШЁШ§ШұЩҮ Ъ©Ш§Щ…ЩҫШ§ЫҢЩ„ ШҙЩҲШҜ. Ш№Щ„Ш§ЩҲЩҮ ШЁШұ Ш§ЫҢЩҶ Ш§ЪҜШұ ШЁШ®ЩҲШ§ЩҮЫҢЩ… ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢвҖҢЩҮШ§ЫҢ Щ…Ш®ШӘЩ„ЩҒЫҢ ШұШ§ ШЁШұШ§ЫҢ ЩҮШұ Ъ©ШҜШ§Щ… Ш§ШІ Ш§ЫҢЩҶ Щ…ЩҲШ§ШұШҜШҢ Ш§ШұШ§ШҰЩҮ ШҜЩҮЫҢЩ…ШҢ ШЁЩҮ Щ…ШҙЪ©Щ„ ШЁШұ Щ…ЫҢвҖҢШ®ЩҲШұЫҢЩ…. Ш®ШЁ ШЁЫҢШ§ЫҢШҜ ШӘШәЫҢЫҢШұШ§ШӘЫҢ ШұШ§ ШҜШұ Ъ©ШҜ ШЁШ§Щ„Ш§ ШЁШҜЩҮЫҢЩ… ШӘШ§ Щ…ШҙЪ©Щ„Ш§ШӘ Ш°Ъ©Шұ ШҙШҜЩҮ ШұШ§ ШӯЩ„ Ъ©ЩҶЫҢЩ….

ШЁШұШ§ЫҢ Ш§ЫҢЩҶ Щ…ЩҶШёЩҲШұ ЩҫЩҲШҙЩҮвҖҢШ§ЫҢ ШұШ§ ШЁЩҮ ЩҶШ§Щ… Services Щ…ЫҢвҖҢШіШ§ШІЫҢЩ… ЩҲ Ш§ЫҢЩҶШӘШұЩҒЫҢШіЫҢ ШұШ§ ШЁЩҮ ЩҶШ§Щ… IMessageBrokerA Ш§ЫҢШ¬Ш§ШҜ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ… ЩҲ ШіЩҫШі Ъ©Ш§ШұЫҢ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ… Ъ©ЩҮ MessageServiceAA Ш§ШІ Ш§ЫҢЩҶ Ш§ЫҢЩҶШӘШұЩҒЫҢШі Ш§ШұШ« ШЁШұЫҢ Ъ©ЩҶШҜ: 

namespace AspNetCoreDependencyInjection.Services
{
    public interface IMessageServiceA
    {
        string Message();
    }
}

ЩҲ ШӯШ§Щ„Ш§ Щ…ЫҢвҖҢШ®ЩҲШ§ЩҮЫҢЩ… ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ШӘШІШұЫҢЩӮ ЩҲШ§ШЁШіШӘЪҜЫҢШҢ ЩҲШ§ШЁШіШӘЪҜЫҢ Ъ©ЩҶШӘШұЩ„Шұ HomeController ШұШ§ Ш§ШІ Ъ©Щ„Ш§Ші MessageBrokerAA Щ„ШәЩҲ Ъ©ШұШҜЩҮ ЩҲ ШўЩҶ ШұШ§ ШЁЩҮ Ш§ЫҢЩҶШӘШұЩҒЫҢШі IMessageBrokerA (Ш§ЩҶШӘШІШ§Ш№) ЩҲШ§ШЁШіШӘЩҮ Ъ©ЩҶЫҢЩ…. ШҜШұ Ш§ЫҢЩҶШ¬Ш§ Щ…Ш§ Ш§ШІ ШӘЪ©ЩҶЫҢЪ© ШӘШІШұЫҢЩӮ ШҜШұЩҲЩҶ ШіШ§ШІЩҶШҜЩҮ ЫҢШ§ Constructor Injection Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ….


ШӘШІШұЫҢЩӮ ШҜШұЩҲЩҶ ШіШ§ШІЩҶШҜЩҮ

ШҜШұ Ш§ЫҢЩҶ ШӘЪ©ЩҶЫҢЪ©ШҢ Щ…Ш§ Щ„ЫҢШіШӘЫҢ Ш§ШІ ЩҲШ§ШЁШіШӘЪҜЫҢвҖҢЩҮШ§ЫҢ Щ…ЩҲШұШҜ ЩҶЫҢШ§ШІ ШұШ§ ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ ЩҫШ§ШұШ§Щ…ШӘШұЩҮШ§ЫҢ ЩҲШұЩҲШҜЫҢ ШіШ§ШІЩҶШҜЩҮвҖҢЫҢ Ъ©Щ„Ш§ШіШҢ ШӘШ№ШұЫҢЩҒ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ…: 
private readonly ILogger<HomeController> _logger;
private readonly IMessageServiceA _messageService;
private readonly GuidProvider _guidHelper;
public HomeController(ILogger<HomeController> logger , IMessageServiceA messageService)
{
        _logger = logger;
        _messageService = messageService;
        _messageService = new MessageServiceAA();
        _guidHelper = new GuidProvider();
}
ЩҲ ШӯШ§Щ„Ш§ Ш§ЪҜШұ ШЁШұЩҶШ§Щ…ЩҮ ШұШ§ Ш§Ш¬ШұШ§ Ъ©ЩҶЫҢЩ…ШҢ ШЁШ§ Ш®Ш·Ш§ЫҢЫҢ ШұЩҲШЁЩҮ ШұЩҲ Щ…ЫҢвҖҢШҙЩҲЫҢЩ… Ъ©ЩҮ ШҜШұ ШўЩҶ Щ…ЫҢвҖҢЪҜЩҲЫҢШҜ Ш§Щ…Ъ©Ш§ЩҶ ЩҲШ§Ъ©ШҙЫҢ (Resolve) ШіШұЩҲЫҢШівҖҢЩҮШ§ЫҢ Щ…ЩҲШұШҜ ЩҶШёШұ ЩҲШ¬ЩҲШҜ ЩҶШҜШ§ШұШҜ. Ш§ЫҢЩҶ Ш®Ш·Ш§ ШЁЩҮ ШҜЩ„ЫҢЩ„ Ш«ШЁШӘ ЩҶШҙШҜЩҶ Ш§ЫҢЩҶШӘШұЩҒЫҢШі IMessageServiceA ЩҲ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ ШўЩҶШҢ ШҜШұЩҲЩҶ Microsoft Dependency Injection Container Ш§ШіШӘ   DI ContainerвҖҢЩҮШ§ Щ…Ш№Щ…ЩҲЩ„Ш§ ШЁШ§ЫҢШҜ ШҜШұ ШІЩ…Ш§ЩҶ ШҙШұЩҲШ№ ШЁШұЩҶШ§Щ…ЩҮШҢ ЩҫЫҢЪ©ШұШЁЩҶШҜЫҢ ЩҲ Щ…ЩӮШҜШ§Шұ ШҜЩҮЫҢ ШҙЩҲШҜЩҶШҜШҢ ШӘШ§ ШҜШұ Ш§ШҜШ§Щ…ЩҮвҖҢЫҢ ЪҶШұШ®ЩҮвҖҢЫҢ ШӯЫҢШ§ШӘ ШЁШұЩҶШ§Щ…ЩҮШҢ ШЁШӘЩҲШ§ЩҶЩҶШҜ ШіШұЩҲЫҢШівҖҢЩҮШ§ ЩҲ Ш§ШҙЫҢШ§ШЎ Щ…ЩҲШұШҜ ЩҶЫҢШ§ШІ ШұШ§ ШЁЩҮ Ъ©Щ„Ш§ШівҖҢЩҮШ§ЫҢЫҢ Ъ©ЩҮ ЩҶЫҢШ§ШІ ШҜШ§ШұЩҶШҜШҢ ЩҲШ§Ъ©ШҙЫҢ ЩҲ ШӘШІШұЫҢЩӮ Ъ©ЩҶЩҶШҜ. Ш§ЩҲЩ„ЫҢЩҶ Щ…ШұШӯЩ„ЩҮ Ш§ШІ Ъ©Ш§Шұ ШЁШ§ DI ContainerвҖҢЩҮШ§ШҢ Ш«ШЁШӘ Ъ©ШұШҜЩҶ ШіШұЩҲЫҢШівҖҢЩҮШ§ ШҜШұЩҲЩҶ ШўЩҶЩҮШ§ШіШӘ.
ШҜШұ ASP.NET Core Ш§ШІ IServiceCollection ШЁШұШ§ЫҢ Ш«ШЁШӘ Ъ©ШұШҜЩҶ ШіШұЩҲЫҢШівҖҢЩҮШ§ЫҢ ШЁШұЩҶШ§Щ…ЩҮвҖҢЫҢ Ш®ЩҲШҜЩ…Ш§ЩҶ Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ…. ШӘЩ…Ш§Щ…ЫҢ ШіШұЩҲЫҢШівҖҢЩҮШ§ЫҢЫҢ ШұШ§ Ъ©ЩҮ Ш§ЩҶШӘШёШ§Шұ ШҜШ§ШұЫҢЩ… ШӘЩҲШіШ· DI Container ШЁЩҮ Ъ©Щ„Ш§ШівҖҢЩҮШ§ЫҢЫҢ ШӘШІШұЫҢЩӮ ШҙЩҲЩҶШҜШҢ ШЁШ§ЫҢШҜ ШҜШұЩҲЩҶ IServiceCollection Ш«ШЁШӘ ЪҜШұШҜЩҶШҜ. ШӘЩ…Ш§Щ… ШіШұЩҲЫҢШівҖҢЩҮШ§ЫҢЫҢ Ъ©ЩҮ ШЁЩҮ ЩҲШіЫҢЩ„ЩҮвҖҢЫҢ IServiceCollection Ш«ШЁШӘ ШҙШҜЩҮвҖҢШ§ЩҶШҜШҢ ЩҫШі Ш§ШІ ШіШ§Ш®ШӘЩҮ ШҙШҜЩҶШҢ ШӘЩҲШіШ· Ш§ЫҢЩҶШӘШұЩҒЫҢШі IServiceProvider ЩӮШ§ШЁЩ„ ЩҲШ§Ъ©ШҙЫҢ ЩҮШіШӘЩҶШҜ.

ШЁЩҶШ§ШЁШұШ§ЫҢЩҶ ШҜЩҲ Ш§ЫҢЩҶШӘШұЩҒЫҢШі ШӯЫҢШ§ШӘЫҢ ШЁШұШ§ЫҢ Ъ©Ш§Шұ Ъ©ШұШҜ ШөШӯЫҢШӯ Microsoft Dependency Injection Container ШҜШұЩҲЩҶ ASP.NET Core ЩҲШ¬ЩҲШҜ ШҜШ§ШұЩҶШҜ:
  • IServiceCollection : ШЁШұШ§ЫҢ Ш«ШЁШӘ ШіШұЩҲЫҢШівҖҢЩҮШ§
  • IServiceProvider : ШЁШұШ§ЫҢ ЩҲШ§Ъ©ШҙЫҢ ШіШұЩҲЫҢШівҖҢЩҮШ§

ШҜШұ ASP.NET Core Щ…Ш№Щ…ЩҲЩ„вҖҢШӘШұЫҢЩҶ Щ…Ъ©Ш§ЩҶ ШЁШұШ§ЫҢ Ш«ШЁШӘ Ъ©ШұШҜЩҶ ШіШұЩҲЫҢШівҖҢЩҮШ§ ШҜШұЩҲЩҶ ContainerШҢ ШЁЩҮ ШөЩҲШұШӘ ЩҫЫҢШҙ ЩҒШұШ¶ ШҜШұЩҲЩҶ Ъ©Щ„Ш§Ші Startup ЩҲ ШҜШұЩҲЩҶ Щ…ШӘШҜ ConfigureServices Ш§ЩҶШ¬Ш§Щ… Щ…ЫҢвҖҢЪҜЫҢШұШҜ.
ШЁЩҮ ШөЩҲШұШӘ ЩҫЫҢШҙ ЩҒШұШ¶ Ъ©Щ„Ш§Ші Startup ШҜЩҲ Щ…ШӘШҜ ШҜШ§ШұШҜ:
  • ConfigureServices : ШЁШұШ§ЫҢ ЩҫЫҢЪ©ШұШЁЩҶШҜЫҢ ЩҲ Ш«ШЁШӘ ШіШұЩҲЫҢШівҖҢЩҮШ§ЫҢ ШҜШұЩҲЩҶЫҢ DI Container Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ.
  • Configure : ШЁШұШ§ЫҢ ШӘЩҶШёЫҢЩ…Ш§ШӘ pipeline Щ…ЫҢШ§ЩҶ Ш§ЩҒШІШ§ШұЩҮШ§ ( Middlewares ) ШЁЪ©Ш§Шұ Щ…ЫҢвҖҢШұЩҲШҜ.

ШӘШ№ШҜШ§ШҜ ШІЫҢШ§ШҜЫҢ Extension method ШЁШұШ§ЫҢ IServiceCollection ЩҲШ¬ЩҲШҜ ШҜШ§ШұЩҶШҜ Ъ©ЩҮ ШЁШұШ§ЫҢ ЩҫШҙШӘЫҢШЁШ§ЩҶЫҢ Ш§ШІ Ш«ШЁШӘ Ъ©ШұШҜЩҶ ШіШұЩҲЫҢШівҖҢЩҮШ§ЫҢ Щ…Ш®ШӘЩ„ЩҒ ШҜШұ ШіЩҶШ§ШұЫҢЩҲЩҮШ§ЫҢ ЪҜЩҲЩҶШ§ЪҜЩҲЩҶ ШЁЩҮ Ъ©Ш§Шұ Щ…ЫҢвҖҢШұЩҲЩҶШҜ. ШҜШұ Ш§ЫҢЩҶШ¬Ш§ Щ…Ш§ Ш§ШІ ЩҶШіШ®ЩҮвҖҢЫҢ 3.1 ЪҶШ§ШұЪҶЩҲШЁ ASP.NET Core Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ…. ШЁШұШ§ЫҢ ЩҮЩ…ЫҢЩҶ ЩҮЩ… ШЁШұШ§ЫҢ Ш«ШЁШӘ ШіШұЩҲЫҢШівҖҢЩҮШ§ЫҢ ЩҫЫҢШҙ ЩҒШұШ¶ ЩҒШұЫҢЩ…ЩҲШұЪ© MVC Ш§ШІ Щ…ШӘШҜ ШӘЩҲШіШ№ЩҮвҖҢЫҢ services.AddControllersWithViews()    Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ….  Щ…ШӘШҜ ШӘЩҲШіШ№ЩҮвҖҢЫҢ AddControllersWithViews() ШіШұЩҲЫҢШівҖҢЩҮШ§ЫҢЫҢ ШұШ§ Ъ©ЩҮ Щ…Ш№Щ…ЩҲЩ„Ш§ ШҜШұ ЩҒШұЫҢЩ… ЩҲШұЪ© MVC Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢШҙЩҲЩҶШҜШҢ ШҜШұЩҲЩҶ IServiceCollection Ш«ШЁШӘ Щ…ЫҢвҖҢЪ©ЩҶШҜ. ШҜШұ ЩҶШіШ®ЩҮвҖҢЩҮШ§ЫҢ ЩӮШЁЩ„ЫҢ ЪҶШ§ШұЪҶЩҲШЁ ASP.NET CoreШҢ  Щ…Ш§ЩҶЩҶШҜ ЩҶШіШ®ЩҮвҖҢЩҮШ§ЫҢ 2.1 ЩҲ 2.2 ШЁШұШ§ЫҢ Ш§ЫҢЩҶ Ъ©Ш§Шұ Ш§ШІ Щ…ШӘШҜ ШӘЩҲШіШ№ЩҮвҖҢЫҢ AddMvc() Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢШҙШҜ.

ШҜШұ Microsoft Dependency Injection Container ШҢ Щ…Ш№Щ…ЩҲЩ„Ш§  ШӘШұШӘЫҢШЁ Ш«ШЁШӘ ШіШұЩҲЫҢШівҖҢЩҮШ§ Щ…ЩҮЩ… ЩҶЫҢШіШӘ.

Ш®ШЁШҢ Ш§ЩҲЩ„ЫҢЩҶ ШіШұЩҲЫҢШі Ш§Ш®ШӘШөШ§ШөЫҢ ШЁШұЩҶШ§Щ…ЩҮвҖҢЫҢ Ш®ЩҲШҜЩ…Ш§ЩҶ ШұШ§ ШЁШ§ ЪҶШұШ®ЩҮвҖҢЫҢ ШӯЫҢШ§ШӘ Transient ЩҲ ШІЫҢШұ ШіШұЩҲЫҢШі ЩҫЫҢШҙЫҢЩҶШҢ ШЁЩҮ ШҙЪ©Щ„ ШІЫҢШұ Ш«ШЁШӘ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ… :

ЩҮЩ…Ш§ЩҶШ·ЩҲШұ Ъ©ЩҮ Щ…ЫҢвҖҢШЁЫҢЩҶЫҢШҜШҢ ШҜШұ Ш§ЫҢЩҶШ¬Ш§ Щ…Ш§ Ш§ШІ Щ…ШӘШҜ AddTransient() Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ШұШҜЩҮвҖҢШ§ЫҢЩ…. ШҜШұ Ш§ЫҢЩҶШ¬Ш§ TService ШҢ Ш§ЫҢЩҶШӘШұЩҒЫҢШі ШіШұЩҲЫҢШі Щ…Ш§ШіШӘ. Ш§ЫҢЩҶ ЩҶЩҲШ№ШҢ ЩҮЩ…Ш§ЩҶ ЩҶЩҲШ№ЫҢ Ш§ШіШӘ Ъ©ЩҮ Ъ©Щ„Ш§ШівҖҢЩҮШ§ЫҢ Щ…Ш§ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЩҶШҜ ШЁЩҮ ШўЩҶ ЩҲШ§ШЁШіШӘЩҮ ШЁШ§ШҙЩҶШҜ.
