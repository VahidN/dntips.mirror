вҖ«Щ…ЩӮШ§ЫҢШіЩҮ value type ЩҲ reference type


ЩҶЩҲЫҢШіЩҶШҜЩҮ: Ш§Щ…ЫҢШұ ЩҮШ§ШҙЩ… ШІШ§ШҜЩҮ
ШӘШ§ШұЫҢШ®: ЫұЫіЫ№ЫІ/ЫұЫ°/ЫІЫі ЫұЫ°:ЫөЫө
ШўШҜШұШі: www.dntips.ir
ШҜШұ ШіЫҢ ШҙШ§ШұЩҫ ШҜЩҲ ЩҶЩҲШ№ class ЩҲ struct ЩҲШ¬ЩҲШҜ ШҜШ§ШұШҜ Ъ©ЩҮ ШӘЩӮШұЫҢШЁШ§ЩӢ Щ…ШҙШ§ШЁЩҮ ЫҢЪ©ШҜЫҢЪҜШұЩҶШҜ ШҜШұ ШӯШ§Щ„ЫҢЪ©ЩҮ ЫҢЪ©ЫҢ Ш§ШІ ШўЩҶЩҮШ§-value type ЩҲ ШҜЫҢЪҜШұЫҢ reference-type Ш§ШіШӘ.

struct ЪҶЫҢШіШӘШҹ
structЩҮШ§ Щ…ШҙШ§ШЁЩҮ classЩҮШ§ ЩҮШіШӘЩҶШҜ ШЁШ§ Ш§ЫҢЩҶ ШӘЩҒШ§ЩҲШӘ Ъ©ЩҮ structЩҮШ§ finalizer ЩҶШҜШ§ШұЩҶШҜ ЩҲ Ш§ШІ Ш§ШұШ« ШЁШұЫҢ ЩҫШҙШӘЫҢШЁШ§ЩҶЫҢ ЩҶЩ…ЫҢвҖҢЪ©ЩҶЩҶШҜ. structЩҮШ§ Ъ©Ш§Щ…Щ„Ш§ Щ…ШҙШ§ШЁЩҮ classЩҮШ§ ШӘШ№ШұЫҢЩҒ Щ…ЫҢвҖҢШҙЩҲЩҶШҜ ЩҲ ШҜШұ ШӘШ№ШұЫҢЩҒ ШўЩҶЩҮШ§ Ш§ШІ Ъ©Щ„Щ…ЩҮ Ъ©Щ„ЫҢШҜЫҢ struct Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ. ШўЩҶЩҮШ§ ШҙШ§Щ…Щ„ ЩҒЫҢЩ„ШҜЩҮШ§ШҢ Щ…ШӘШҜЩҮШ§ШҢ Ш®ШөЩҲШөЫҢШӘвҖҢЩҮШ§ ЩҶЫҢШІ Щ…ЫҢвҖҢШҙЩҲЩҶШҜ. ШҜШұ ШІЫҢШұ ЩҶШӯЩҲЩҮ ШӘШ№ШұЫҢЩҒ ШўЩҶ ШұШ§ Щ…ШҙШ§ЩҮШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶЫҢШҜ: 

struct Point
{
   private int x, y;             // private fields
 
   public Point (int x, int y)   // constructor
   {
         this.x = x;
         this.y = y;
   }

   public int X                  // property
   {
         get {return x;}
         set {x = value;}
   }

   public int Y
   {
         get {return y;}
         set {y = value;}
   }
}

value type ЩҲ reference type
ШӘЩҒШ§ЩҲШӘ ШҜЫҢЪҜШұЫҢ Ъ©ЩҮ ШЁЫҢЩҶ class ЩҲ structШҢ Ш§ШІ Ш§ЩҮЩ…ЫҢШӘ ЩҲЫҢЪҳЩҮвҖҢШ§ЫҢ ШЁШұШ®ЩҲШұШҜШ§Шұ Ш§ШіШӘ  ШўЩҶ Ш§ШіШӘ Ъ©ЩҮ classЩҮШ§ reference-type ЩҲ structЩҮШ§ value-type ЩҮШіШӘЩҶШҜ ЩҲ ШҜШұ ШІЩ…Ш§ЩҶ Ш§Ш¬ШұШ§ ШЁШ§ ШўЩҶЩҮШ§ Щ…ШӘЩҒШ§ЩҲШӘ ШұЩҒШӘШ§Шұ Щ…ЫҢвҖҢШҙЩҲШҜ ЩҲ ШҜШұ Ш§ШҜШ§Щ…ЩҮ ШЁЩҮ ШӘШҙШұЫҢШӯ ШўЩҶ Щ…ЫҢвҖҢЩҫШұШҜШ§ШІЫҢЩ….
ЩҲЩӮШӘЫҢ ЫҢЪ© ЩҲЩҮЩ„ЩҮ Ш§ШІ value-type Ш§ЫҢШ¬Ш§ШҜ ШҙЩҲШҜШҢ ЫҢЪ© ЩҒШ¶Ш§ЫҢ Ш®Ш§Щ„ЫҢ Ш§ШІ ШӯШ§ЩҒШёЩҮвҖҢЫҢ Ш§ШөЩ„ЫҢ (RAM) ШЁШұШ§ЫҢ Ш°Ш®ЫҢШұЩҮ ШіШ§ШІЫҢ Щ…ЩӮШҜШ§Шұ ШўЩҶ ШӘШ®ШөЫҢШө ШҜШ§ШҜЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ. ЩҶЩҲШ№вҖҢЩҮШ§ЫҢ Ш§ШөЩ„ЫҢ Щ…Ш§ЩҶЩҶШҜ int, float, bool ЩҲ char Ш§ШІ ЩҶЩҲШ№ value type ЩҮШіШӘЩҶШҜ. ШҜШұ Ш¶Щ…ЩҶ ШіШұШ№ШӘ ШҜШіШӘШұШіЫҢ ШЁЩҮ ШўЩҶЩҮШ§ ШЁШіЫҢШ§Шұ ШЁШ§Щ„Ш§ШіШӘ.
ЩҲЩ„ЫҢ ЩҲЩӮШӘЫҢ ЫҢЪ© ЩҲЩҮЩ„ЩҮ Ш§ШІ reference-type Ш§ЫҢШ¬Ш§ШҜ ШҙЩҲШҜШҢ ЫҢЪ© ЩҒШ¶Ш§ ШЁШұШ§ЫҢ object ЩҲ ЩҒШ¶Ш§ЫҢЫҢ ШҜЫҢЪҜШұ ШЁШұШ§ЫҢ Ш§ШҙШ§ШұЩҮвҖҢЪҜШұ ШЁЩҮ ШўЩҶ ШҙЫҢШЎ ШҜШұ ШӯШ§ЩҒШёЩҮ Ш§ШөЩ„ЫҢ Ш°Ш®ЫҢШұЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ. ШҜШұ ЩҲШ§ЩӮШ№ ШҜЩҲ ЩҒШ¶Ш§ Ш§ШІ ШӯШ§ЩҒШёЩҮ ШЁШұШ§ЫҢ Ш°Ш®ЫҢШұЩҮ ШіШ§ШІЫҢ ШўЩҶЩҮШ§ Ш§ШҙШәШ§Щ„ Щ…ЫҢвҖҢШҙЩҲШҜ. ШЁШұШ§ЫҢ ШҜШұЪ© ШЁЩҮШӘШұ ШЁЩҮ Щ…Ш«Ш§Щ„ ШІЫҢШұ ШӘЩҲШ¬ЩҮ Ъ©ЩҶЫҢШҜ:
Point p1 = new Point();         // Point is a *struct*
Form f1 = new Form();           // Form is a *class*
ЩҶЪ©ШӘЩҮ: Point Ш§ШІ ЩҶЩҲШ№ struct ЩҲ Form Ш§ШІ ЩҶЩҲШ№ reference Ш§ШіШӘ. ШҜШұ Щ…ЩҲШұШҜ Ш§ЩҲЩ„ШҢ ЫҢЪ© ЩҒШ¶Ш§ Ш§ШІ ШӯШ§ЩҒШёЩҮ ШЁШұШ§ЫҢ p1 ШӘШ®ШөЫҢШө ШҜШ§ШҜЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ ЩҲ ШҜШұ Щ…ЩҲШұШҜ ШҜЩҲЩ…ШҢ ШҜЩҲ ЩҒШ¶Ш§ Ш§ШІ ШӯШ§ЩҒШёЩҮ Ш§ШөЩ„ЫҢ ЫҢЪ©ЫҢ ШЁШұШ§ЫҢ Ш°Ш®ЫҢШұЩҮ Ъ©ШұШҜЩҶ Ш§ШҙШ§ШұЩҮвҖҢЪҜШұ f1 ШЁШұШ§ЫҢ Ш§ШҙШ§ШұЩҮ ШЁЩҮ Form object ЩҲ ШҜЫҢЪҜШұЫҢ ШЁШұШ§ЫҢ Ш°Ш®ЫҢШұЩҮ Ъ©ШұШҜЩҶ Form object ШӘШ®ШөЫҢШө ШҜШ§ШҜЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ.
Form f1;                        // Allocate the reference
f1 = new Form();                // Allocate the object
ШЁЩҮ ЩӮШ·Ш№ЩҮ Ъ©ШҜ ШІЫҢШұ ШҜЩӮШӘ Ъ©ЩҶЫҢШҜ:
Point p2 = p1;
Form f2 = f1;
ЩҮЩ…Ш§ЩҶШ·ЩҲШұ Ъ©ЩҮ ЩӮШЁЩ„Ш§ЩӢ ЪҜЩҒШӘЩҮ ШҙШҜ p2ШҢ ЫҢЪ© ЩҶЩҲШ№ struct Ш§ШіШӘ ШЁЩҶШ§ШЁШұШ§ЫҢЩҶ ШҜШұ Щ…ЩҲШұШҜ Ш§ЩҲЩ„ Щ…ЩӮШҜШ§Шұ p2 ЫҢЪ© Ъ©ЩҫЫҢ Ш§ШІ Щ…ЩӮШҜШ§Шұ p1 Ш®ЩҲШ§ЩҮШҜ ШЁЩҲШҜ ЩҲЩ„ЫҢ ШҜШұ Щ…ЩҲШұШҜ ШҜЩҲЩ…ШҢ ШўШҜШұШі f1 ШұШ§ ШҜШұЩҲЩҶ f2 Ъ©ЩҫЫҢ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ… ШҜШұ ЩҲШ§ЩӮШ№ f1 ЩҲ f2 ШЁЩҮ ЫҢЪ© ШҙЫҢШЎ Ш§ШҙШ§ШұЩҮ Ш®ЩҲШ§ЩҮЩҶШҜ Ъ©ШұШҜ. (ЫҢЪ© ШҙЫҢШЎ ШЁШ§ 2 Ш§ШҙШ§ШұЩҮ ЪҜШұ)
ШҜШұ ШіЫҢ ШҙШ§ШұЩҫШҢ ЩҫШ§ШұШ§Щ…ШӘШұЩҮШ§ (ШЁШөЩҲШұШӘ ЩҫЫҢШҙ ЩҒШұШ¶) ШЁШөЩҲШұШӘ ЫҢЪ© Ъ©ЩҫЫҢ Ш§ШІ ШўЩҶЩҮШ§ ШЁЩҮ Щ…ШӘШҜЩҮШ§ Ш§ШұШіШ§Щ„ Щ…ЫҢвҖҢШҙЩҲЩҶШҜШҢ ЫҢШ№ЩҶЫҢ Ш§ЪҜШұ ЩҫШ§ШұШ§Щ…ШӘШұ Ш§ШІ ЩҶЩҲШ№ value-type ШЁШ§ШҙШҜ ЫҢЪ© Ъ©ЩҫЫҢ Ш§ШІ ШўЩҶ ЩҲЩҮЩ„ЩҮ ЩҲ Ш§ЪҜШұ ЩҫШ§ШұШ§Щ…ШӘШұ reference-type ЫҢЪ© Ъ©ЩҫЫҢ Ш§ШІ ШўШҜШұШі Ш§ШұШіШ§Щ„ Ш®ЩҲШ§ЩҮШҜ ШҙШҜ. ШЁШұШ§ЫҢ ШӘЩҲШ¶ЫҢШӯ ШЁЩҮШӘШұ ШЁЩҮ Щ…Ш«Ш§Щ„ ШІЫҢШұ ШӘЩҲШ¬ЩҮ Ъ©ЩҶЫҢШҜ:
Point myPoint = new Point (0, 0);      // a new value-type variable
Form myForm = new Form();              // a new reference-type variable
Test (myPoint, myForm);                // Test is a method defined below
 
void Test (Point p, Form f)
{
      p.X = 100;                       // No effect on MyPoint since p is a copy
      f.Text = "Hello, World!";        // This will change myFormвҖҷs caption since
                                       // myForm and f point to the same object
      f = null;                        // No effect on myForm
}
Ш§ЩҶШӘШіШ§ШЁ null ШЁЩҮ f ШҜШұЩҲЩҶ Щ…ШӘШҜ Test ЩҮЫҢЪҶЫҢ Ш§Ш«ШұЫҢ ШЁШұ ШұЩҲЫҢ ШўШҜШұШі myForm ЩҶШҜШ§ШұШҜ ЪҶЩҲЩҶ fШҢ ЫҢЪ© Ъ©ЩҫЫҢ Ш§ШІ ШўШҜШұШі myForm Ш§ШіШӘ.
ШӯШ§Щ„ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЫҢЩ… ШұЩҲШҙ ЩҫЫҢШҙ ЩҒШұШ¶ ШұШ§ ШЁШ§ Ш§ЩҒШІЩҲШҜЩҶ Ъ©Щ„Щ…ЩҮ Ъ©Щ„ЫҢШҜ ref ШӘШәЫҢЫҢШұ ШҜЩҮЫҢЩ….  ЩҲЩӮШӘЫҢ Ш§ШІ ref Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ЩҶЫҢЩ… Щ…ШӘШҜ ШЁШ§ ЩҫШ§ШұШ§Щ…ШӘШұЩҮШ§ЫҢ ЩҒШұШ§Ш®ЩҲШ§ЩҶЫҢ Ъ©ЩҶЩҶШҜЩҮ (caller's arguments) ШЁШөЩҲШұШӘ Щ…ШіШӘЩӮЫҢЩ… ШҜШұ ШӘШ№Ш§Щ…Щ„ Ш§ШіШӘ ШҜШұ Ъ©ШҜ ШІЫҢШұ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЫҢЩ… ШӘШөЩҲШұ Ъ©ЩҶЫҢЩ… Ъ©ЩҮ ЩҫШ§ШұШ§Щ…ШӘШұЩҮШ§ЫҢ p ЩҲ f Щ…ШӘШҜ Test ЩҮЩ…Ш§ЩҶ Щ…ШӘШәЫҢШұЩҮШ§ЫҢ myPoint ЩҲ myForm Ш§ШіШӘ.
Point myPoint = new Point (0, 0);      // a new value-type variable
Form myForm = new Form();              // a new reference-type variable
Test (ref myPoint, ref myForm);        // pass myPoint and myForm by reference
 
void Test (ref Point p, ref Form f)
{
      p.X = 100;                       // This will change myPointвҖҷs position
      f.Text = вҖңHello, World!вҖқ;        // This will change MyFormвҖҷs caption
      f = null;                        // This will nuke the myForm variable!
}
ШҜШұ Ъ©ШҜ ШЁШ§Щ„Ш§ Ш§ЩҶШӘШіШ§ШЁ null ШЁЩҮ f ШЁШ§Ш№Ш« ШӘЩҮЫҢ ШҙШҜЩҶ myForm Щ…ЫҢвҖҢШҙЩҲШҜ ШЁШҜЩ„ЫҢЩ„ Ш§ЫҢЩҶЪ©ЩҮ Щ…ШӘШҜ Щ…ШіШӘЩӮЫҢЩ…Ш§ЩӢ ШЁЩҮ ШўЩҶ ШҜШіШӘШұШіЫҢ ШҜШ§ШҙШӘЩҮ Ш§ШіШӘ.

ШӘШ®ШөЫҢШө ШӯШ§ЩҒШёЩҮ
CLR Ш§ШҙЫҢШ§ШЎ ШұШ§ ШҜШұ ШҜЩҲ ЩӮШіЩ…ШӘ Ш°Ш®ЫҢШұЩҮ Щ…ЫҢвҖҢЪ©ЩҶШҜ:
  1. stack ЫҢШ§ ЩҫШҙШӘЩҮ
  2. heap
ШіШ§Ш®ШӘШ§Шұ stack ЫҢШ§ ЩҫШҙШӘЩҮ first-in last-out Ш§ШіШӘ Ъ©ЩҮ ШҜШіШӘШұШіЫҢ ШЁЩҮ ШўЩҶ ШіШұЫҢШ№ Ш§ШіШӘ. ШІЩ…Ш§ЩҶЫҢ Ъ©ЩҮ Щ…ШӘШҜЫҢ ЩҒШұШ§Ш®ЩҲШ§ЩҶЫҢ Щ…ЫҢвҖҢШҙЩҲШҜШҢ CLR ЩҫШҙШӘЩҮ ШұШ§ ЩҶШҙШ§ЩҶЩҮ ЪҜШ°Ш§ШұЫҢ Щ…ЫҢвҖҢЪ©ЩҶШҜ. ШіЩҫШі Щ…ШӘШҜ data ШұШ§ ШЁЩҮ ЩҫШҙШӘЩҮ Ш¬ЩҮШӘ Ш§Ш¬ШұШ§ push Щ…ЫҢвҖҢЪ©ЩҶШҜ ЩҲ ШІЩ…Ш§ЩҶЫҢ Ъ©ЩҮ Ш§Ш¬ШұШ§ЫҢШҙ ШЁЩҮ Ш§ШӘЩ…Ш§Щ… ШұШіЫҢШҜШҢ CLR ЩҫШҙШӘЩҮ ШұШ§ ШӘШ§ Щ…ШӯЩ„ ЩҶШҙШ§ЩҶЩҮ ЪҜШ°Ш§ШұЫҢ ШҙШҜЩҮ Щ…ШұШӯЩ„ЩҮ ЩӮШЁЩ„ШҢ ЩҫШ§Ъ© Щ…ЫҢвҖҢЪ©ЩҶШҜ (pop).
ЩҲЩ„ЫҢ ШіШ§Ш®ШӘШ§Шұ heap ШЁШөЩҲШұШӘ ШӘШөШ§ШҜЩҒЫҢ Ш§ШіШӘ. ЫҢШ№ЩҶЫҢ Ш§ШҙЫҢШ§ШЎ ШҜШұ Щ…ШӯЩ„вҖҢЩҮШ§ЫҢ ШӘШөШ§ШҜЩҒЫҢ ЩӮШұШ§Шұ ШҜШ§ШҜЩҮ Щ…ЫҢвҖҢШҙЩҲЩҶШҜ ШЁЩҮЩ…ЫҢЩҶ ШҜЩ„ЫҢЩ„ ШўЩҶЩҮШ§ ШҜШ§ШұШ§ЫҢ 2 ШіШұШЁШ§Шұ memory manager ЩҲ garbage-collector ЩҮШіШӘЩҶШҜ.
ШЁШұШ§ЫҢ ШўШҙЩҶШ§ЫҢЫҢ ШЁШ§ ЩҶШӯЩҲЩҮ Ш§ШіШӘЩҒШ§ШҜЩҮ ЩҫШҙШӘЩҮ ЩҲ heap ШЁЩҮ Ъ©ШҜ ШІЫҢШұ ШӘЩҲШ¬ЩҮ Ъ©ЩҶЫҢШҜ:
void CreateNewTextBox()
{
      TextBox myTextBox = new TextBox();             // TextBox is a class
}
ШҜШұ Ш§ЫҢЩҶ Щ…ШӘШҜШҢ Щ…Ш§ ЫҢЪ© Щ…ШӘШәЫҢШұ Щ…ШӯЩ„ЫҢ Ш§ЫҢШ¬Ш§ШҜ Ъ©ШұШҜЩҮ Ш§ЫҢЩ… Ъ©ЩҮ ШЁЩҮ ЫҢЪ© ШҙЫҢШЎ Ш§ШҙШ§ШұЩҮ Щ…ЫҢвҖҢЪ©ЩҶШҜ.

ЩҫШҙШӘЩҮ ЩҮЩ…ЫҢШҙЩҮ ШЁШұШ§ЫҢ Ш°Ш®ЫҢШұЩҮ ШіШ§ШІЫҢ Щ…ЩҲШ§ШұШҜ ШІЫҢШұ Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ:
  • ЩӮШіЩ…ШӘ reference Щ…ШӘШәЫҢШұЩҮШ§ЫҢ Щ…ШӯЩ„ЫҢ ЩҲ ЩҫШ§ШұШ§Щ…ШӘШұЩҮШ§ЫҢ Ш§ШІ ЩҶЩҲШ№ reference-typed (Щ…Ш§ЩҶЩҶШҜ myTextBox)
  • Щ…ШӘШәЫҢШұЩҮШ§ЫҢ Щ…ШӯЩ„ЫҢ ЩҲ ЩҫШ§ШұШ§Щ…ШӘШұЩҮШ§ЫҢ Щ…ШӘШҜ Ш§ШІ ЩҶЩҲШ№ value-typed (Щ…Ш§ЩҶЩҶШҜ integer, bool, char, DateTime ЩҲ ...)
ЩҮЩ…ЪҶЩҶЫҢЩҶ Ш§ШІ heap ШЁШұШ§ЫҢ Ш°Ш®ЫҢШұЩҮ ШіШ§ШІЫҢ Щ…ЩҲШ§ШұШҜ ШІЫҢШұ Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ:
  • Щ…ШӯШӘЩҲЫҢШ§ШӘ ШҙЫҢШЎ Ш§ШІ ЩҶЩҲШ№ reference-typed
  • ЩҮШұ ЪҶЫҢШІЫҢ Ъ©ЩҮ ЩӮШұШ§Шұ Ш§ШіШӘ ШҜШұ ШҙЫҢШЎ Ш§ШІ ЩҶЩҲШ№ reference-typed Ш°Ш®ЫҢШұЩҮ ШҙЩҲШҜ.

ШўШІШ§ШҜШіШ§ШІЫҢ ШӯШ§ЩҒШёЩҮ ШҜШұ heap

ШҜШұ Ъ©ШҜ ШЁШ§Щ„Ш§ ЩҲЩӮШӘЫҢ Ш§Ш¬ШұШ§ЫҢ Щ…ШӘШҜ CreateNewTextBox ШЁЩҮ Ш§ШӘЩ…Ш§Щ… ШЁШұШіШҜ Щ…ШӘШәЫҢШұ myTextBox Ш§ШІ ШҜЫҢШҜ (Scope) Ш®Ш§ШұШ¬ Щ…ЫҢвҖҢШҙЩҲШҜ. ШЁЩҶШ§ШЁШұШ§ЫҢЩҶ Ш§ШІ ЩҫШҙШӘЩҮ ЩҶЫҢШІ Ш®Ш§ШұШ¬ Щ…ЫҢвҖҢШҙЩҲШҜ ЩҲЩ„ЫҢ ШЁШ§ Ш®Ш§ШұШ¬ ШҙШҜЩҶ myTextBox Ш§ШІ ЩҫШҙШӘЩҮ ЪҶЩҮ Ш§ШӘЩҒШ§ЩӮЫҢ ШЁШұШ§ЫҢ TextBox object ШұШ® Ш®ЩҲШ§ЩҮШҜ ШҜШ§ШҜШҹ! ЩҫШ§ШіШ® ШҜШұ garbage-collector ЩҶЩҮЩҒШӘЩҮ Ш§ШіШӘ. garbage-collector ШЁШөЩҲШұШӘ Ш®ЩҲШҜЪ©Ш§Шұ Ш№Щ…Щ„ЫҢШ§ШӘ ЩҫШ§Ъ©ШіШ§ШІЫҢ heap ШұШ§ Ш§ЩҶШ¬Ш§Щ… Щ…ЫҢвҖҢШҜЩҮШҜ ЩҲ Ш§ШҙЫҢШ§ШҰЫҢ Ъ©ЩҮ Ш§ШҙШ§ШұЩҮ ЪҜШұ Щ…Ш№ШӘШЁШұ ЩҶШҜШ§ШұЩҶШҜ ШұШ§ ШӯШ°ЩҒ Щ…ЫҢвҖҢЩҶЩ…Ш§ЫҢШҜ. ШҜШұ ШӯШ§Щ„ШӘ Ъ©Щ„ЫҢ Ш§ЪҜШұ ШҙЫҢШЎ Ш§ШІ ШӯШ§ЩҒШёЩҮ Ш®Ш§ШұШ¬ ШҙШҜ ШЁШ§ЫҢШҜ Щ…ЩҶШ§ШЁШ№ ШіШ§ЫҢШұ ЩӮШіЩ…ШӘвҖҢЩҮШ§ЫҢ Ш§ШҙШәШ§Щ„ ШҙШҜЩҮ ШӘЩҲШіШ· ШўЩҶ ЩҮЩ… ШўШІШ§ШҜ ШҙЩҲШҜШҢ Ъ©ЩҮ Ш§ЫҢЩҶ ШўШІШ§ШҜ ШіШ§ШІЫҢ ШЁШ№ЩҮШҜЩҮ garbage-collector Ш§ШіШӘ.

ШӯШ§Щ„ ШўШІШ§ШҜ ШіШ§ШІЫҢ ШЁШұШ§ЫҢ Ъ©Щ„Ш§ШіЩҮШ§ЫҢЫҢ Ъ©ЩҮ Ш§ЫҢЩҶШӘШұЩҒЫҢШі IDisposable ШұШ§ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ Щ…ЫҢвҖҢЪ©ЩҶЩҶШҜ ШЁЩҮ ШҜЩҲ ШөЩҲШұШӘ Ш§ЩҶШ¬Ш§Щ… Щ…ЫҢвҖҢШҙЩҲШҜ:

  1. ШҜШіШӘЫҢ: ШЁШ§ ЩҒШұШ§Ш®ЩҲШ§ЩҶЫҢ Щ…ШӘШҜ Dispose Щ…ЫҢШіШұ Ш§ШіШӘ.
  2. Ш®ЩҲШҜЪ©Ш§Шұ: Ш§ЩҒШІЩҲШҜЩҶ ШҙЫҢШЎ ШЁЩҮ Net Container. Щ…Ш§ЩҶЩҶШҜ Form, Panel, TabPage ЫҢШ§ UserControl. Ш§ЫҢЩҶ ЩҶЪҜЩҮШҜШ§ШұЩҶШҜЩҮШ§ Ш§ЫҢЩҶ Ш§Ш·Щ…ЫҢЩҶШ§ЩҶ ШұШ§ ШЁЩҮ Щ…Ш§ Щ…ЫҢвҖҢШҜЩҮЩҶШҜ ШҜШұ ШөЩҲШұШӘЫҢЪ©ЩҮ ШўЩҶЩҮШ§ Ш§ШІ ШӯШ§ЩҒШёЩҮ Ш®Ш§ШұШ¬ ШҙШҜЩҶШҜ Ъ©Щ„ЫҢЩҮ Ш№Ш¶ЩҲЩҮШ§ЫҢ ШўЩҶ ЩҮЩ… Ш§ШІ ШӯШ§ЩҒШёЩҮ Ш®Ш§ШұШ¬ ШҙЩҲЩҶШҜ.

ШЁШұШ§ЫҢ ШўШІШ§ШҜШіШ§ШІЫҢ ШҜШіШӘЫҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЫҢЩ… Щ…Ш§ЩҶЩҶШҜ Ъ©ШҜЩҮШ§ЫҢ ШІЫҢШұ Ш№Щ…Щ„ Ъ©ЩҶЫҢЩ…:

using (Stream s = File.Create ("myfile.txt"))

{
   ...
}
ЫҢШ§
Stream s = File.Create ("myfile.txt");

try
{
   ...
}

finally
{
   if (s != null) s.Dispose();
}


Щ…Ш«Ш§Щ„ЫҢ Ш§ШІ Windows Forms
ЩҒШұШ¶ Ъ©ЩҶЫҢШҜ ЩӮШөШҜ ШҜШ§ШұЫҢЩ… ЩҒЩҲЩҶШӘ ЩҲ Ш§ЩҶШҜШ§ШІЩҮ ЫҢЪ© ЩҲЫҢЩҶШҜЩҲШІ ЩҒШұЩ… ШұШ§ ШӘШәЫҢЫҢШұ ШҜЩҮЫҢЩ….

Size s = new Size (100, 100);          // struct = value type
Font f = new Font (вҖңArialвҖқ,10);        // class = reference type

Form myForm = new Form();

myForm.Size = s;
myForm.Font = f;
ШӘЩҲШ¬ЩҮ Ъ©ЩҶЫҢШҜ Ъ©ЩҮ Щ…Ш§ ШҜШұ Ъ©ШҜ ШЁШ§Щ„Ш§ Ш§ШІ Ш§Ш№Ш¶Ш§ЫҢ myForm Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ШұШҜЫҢЩ… ЩҶЩҮ Ш§ШІ Ъ©Щ„Ш§ШіЩҮШ§ЫҢ Font ЩҲ Size Ъ©ЩҮ Ш§ЫҢЩҶ ШҜЩҲ ЪҜШ§ЩҶЪҜЫҢ ЩӮШ§ШЁЩ„ ЩӮШЁЩҲЩ„ Ш§ШіШӘ. ШӯШ§Щ„ ШЁЩҮ ШӘШөЩҲЫҢШұ ШІЫҢШұ Ъ©ЩҮ ШЁЩҮ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ Ъ©ШҜ ШЁШ§Щ„Ш§ Ш§ШҙШ§ШұЩҮ ШҜШ§ШұШҜ ШӘЩҲШ¬ЩҮ Ъ©ЩҶЫҢШҜ.

ЩҮЩ…Ш§ЩҶШ·ЩҲШұ Ъ©ЩҮ Щ…ШҙШ§ЩҮШҜ Щ…ЫҢвҖҢЪ©ЩҶЫҢШҜ Щ…ШӯШӘЩҲЫҢШ§ШӘ s ЩҲ ШўШҜШұШі f ШұШ§ ШҜШұ Form object Ш°Ш®ЫҢШұЩҮ Ъ©ШұШҜЩҮ Ш§ЫҢЩ… Ъ©ЩҮ ЩҶШҙШ§ЩҶ Щ…ЫҢвҖҢШҜЩҮШҜ ШӘШәЫҢЫҢШұ ШҜШұ s ШЁШұШұЩҲЫҢ ЩҒШұЩ… ШӘШәЫҢЫҢШұ Ш§ЫҢШ¬Ш§ШҜ ЩҶЩ…ЫҢвҖҢЪ©ЩҶШҜ ЩҲЩ„ЫҢ ШӘШәЫҢЫҢШұ ШҜШұ f ШЁШ§Ш№Ш« Ш§ЫҢШ¬Ш§ШҜ ШӘШәЫҢЫҢШұ ЩҒШұЩ… Щ…ЫҢвҖҢШҙЩҲШҜ. Form object ШҜЩҲ Ш§ШҙШ§ШұЩҮ ЪҜШұ ШЁЩҮ Font object ШҜШ§ШұШҜ.

In-Line Allocation (ШӘШ®ШөЫҢШө ШҜШұШ¬Ш§)
ШҜШұ ЩӮШЁЩ„ ЪҜЩҒШӘЩҮ ШҙШҜ ШЁШұШ§ЫҢ Ш°Ш®ЫҢШұЩҮ Щ…ШӘШәЫҢШұЩҮШ§ЫҢ Щ…ШӯЩ„ЫҢ Ш§ШІ ЩҶЩҲШ№ value-typed Ш§ШІ ЩҫШҙШӘЩҮ Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ ШўЫҢШ§ ШҙЫҢШЎ Size Ш¬ШҜЫҢШҜ ЩҮЩ… ШҜШұ ЩҫШҙШӘЩҮ Ш°Ш®ЫҢШұЩҮ Щ…ЫҢвҖҢШҙЩҲШҜШҹ Ш®ЫҢШұШҢ ШЁШҜЩ„ЫҢЩ„ Ш§ЫҢЩҶЪ©ЩҮ ШўЩҶ Щ…ШӘШәЫҢШұ Щ…ШӯЩ„ЫҢ ЩҶЫҢШіШӘ ЩҲ ШҜШұ ШҙЫҢШЎ ШҜЫҢЪҜШұ Ш°Ш®ЫҢШұЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ (ШҜШұ Щ…Ш«Ш§Щ„ ШЁШ§Щ„Ш§ ШҜШұ ЫҢЪ© ЩҒШұЩ… Ш°Ш®ЫҢШұЩҮ ШҙШҜЩҮ Ш§ШіШӘ) Ъ©ЩҮ ШўЩҶ ШҙЫҢШЎ ЩҮЩ… ШҜШұ heap Ш°Ш®ЫҢШұЩҮ ШҙШҜЩҮ Ш§ШіШӘ ЩҫШі ШҙЫҢШЎ Ш¬ШҜЫҢШҜ Size ЩҮЩ… ШҜШұ heap Ш°Ш®ЫҢШұ Щ…ЫҢвҖҢШҙЩҲШҜ Ъ©ЩҮ ШЁЩҮ Ш§ЫҢЩҶ ЩҶЩҲШ№ Ш°Ш®ЫҢШұЩҮ ШіШ§ШІЫҢ In-Line ЪҜЩҒШӘЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ.

ШӘЩ„ЩҮ (Trap)
ЩҒШұШ¶ Ъ©ЩҶЫҢШҜ Ъ©Щ„Ш§Ші Form ШЁШҙЪ©Щ„ ШІЫҢШұ ШӘШ№ШұЫҢЩҒ ШҙШҜЩҮ Ш§ШіШӘ:
class Form
{
      // Private field members
      Size size;
      Font font;

      // Public property definitions
      public Size Size
      {
            get    { return size; }
            set    { size = value; fire resizing events }
      }

      public Font Font
      {
            get    { return font;  }
            set    { font = value; }
      }
}
ШӯШ§Щ„ Щ…Ш§ ЩӮШөШҜ ШҜШ§ШұЫҢЩ… Ш§ШұШӘЩҒШ§Ш№ ШўЩҶ ШұШ§ ШҜЩҲ ШЁШұШ§ШЁШұ Ъ©ЩҶЫҢЩ…ШҢ ШЁЩҶШ§ШЁШұШ§ЫҢЩҶ Ш§ШІ Ъ©ШҜ ШІЫҢШұ Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ…:
myForm.ClientSize.Height = myForm.ClientSize.Height * 2;
ЩҲЩ„ЫҢ ШЁШ§ Ш®Ш·Ш§ЫҢ Ъ©Ш§Щ…ЩҫШ§ЫҢЩ„Шұ ШІЫҢШұ ШұЩҲШЁШұЩҲ Щ…ЫҢвҖҢШҙЩҲЫҢЩ…:
Cannot modify the return value of 'System.Windows.Forms.Form.ClientSize' because it is not a variable
Ш№Щ„ШӘ ЪҶЫҢШіШӘШҹ ШЁШҜЩ„ЫҢЩ„ Ш§ЫҢЩҶЪ©ЩҮ myForm.ClientSize ШҙЫҢШЎ Size Ъ©ЩҮ Ш§ШІ ЩҶЩҲШ№ Struct Ш§ШіШӘ ШұШ§ ШЁШұ Щ…ЫҢвҖҢЪҜШұШҜШ§ЩҶШҜ ЩҲ Ш§ЫҢЩҶ Struct Ш§ШІ ЩҶЩҲШ№ value-typed Ш§ШіШӘ ЩҲ Ш§ЫҢЩҶ ШҙЫҢШЎ ЫҢЪ© Ъ©ЩҫЫҢ Ш§ШІ Ш§ЩҶШҜШ§ШІЩҮ ЩҒШұЩ… Ш§ШіШӘ ЩҲ Щ…Ш§ ЩҮЩ…ШІЩ…Ш§ЩҶ ЩӮШөШҜ ШҜЩҲ ШЁШұШ§ШЁШұ ЩҶЩ…ЩҲШҜЩҶ ШўЩҶ Ъ©ЩҫЫҢ ШұШ§ ШҜШ§ШұЫҢЩ… Ъ©ЩҮ Ъ©Ш§Щ…ЩҫШ§ЫҢЩ„Шұ Ш®Ш·Ш§ЫҢ ШЁШ§Щ„Ш§ ШұШ§ ЩҶЩ…Ш§ЫҢШҙ Щ…ЫҢвҖҢШҜЩҮШҜ.

ШЁШұШ§ЫҢ ШӘЩҲШ¶ЫҢШӯ ШЁЫҢШҙШӘШұ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЫҢШҜ ШЁЩҮ Ш§ЫҢЩҶ ШіЩҲШ§Щ„ Щ…ШұШ§Ш¬Ш№ЩҮ Ъ©ЩҶЫҢШҜ ЩҲ ШҜШұ ШӘЪ©Щ…ЫҢЩ„ ШўЩҶ Ш§ЫҢЩҶ Щ„ЫҢЩҶЪ© ШұШ§ ЩҮЩ… ШЁШұШұШіЫҢ Ъ©ЩҶЫҢШҜ.

ЩҫШі ШЁЩҶШ§ШЁШұШ§ЫҢЩҶ Ъ©ШҜ ШЁШ§Щ„Ш§ ШұШ§ ШЁЩҮ Ъ©ШҜ ШІЫҢШұ Ш§ШөЩ„Ш§Шӯ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ…:
myForm.ClientSize = new Size (myForm.ClientSize.Width, myForm.ClientSize.Height * 2);

    Ш§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§ref struct ШҜШұ ШҜШ§ШӘвҖҢЩҶШӘ 9Ш§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§ШЁШұШұШіЫҢ ЩҮШІЫҢЩҶЩҮвҖҢЩҮШ§ЫҢ ЩҫЩҶЩҮШ§ЩҶ ШЁШұШұШіЫҢ Щ…ЩӮШҜШ§Шұ ШӘЩҮЫҢ (Null Check) ШҜШұ Ъ©ШҜвҖҢЩҮШ§ЫҢ Ш¬ЩҶШұЫҢЪ© ШіЫҢвҖҢШҙШ§ШұЩҫШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Ref Structs ШҜШұ #C ШЁШұШ§ЫҢ ШЁЩҮШЁЩҲШҜ Ш№Щ…Щ„Ъ©ШұШҜШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§Щ…ШҜЫҢШұЫҢШӘ ШЁЩҮЫҢЩҶЩҮ ШӯЩ„ЩӮЩҮвҖҢЩҮШ§ЫҢ ШӘЪ©ШұШ§Шұ ШҜШұ NET.: ШұШ§ЩҮЩҶЩ…Ш§ЫҢ Ш¬Ш§Щ…Ш№ Ъ©Ш§Шұ ШЁШ§ Ш§ЩҶЩҲШ§Ш№ Щ…ЩӮШҜШ§ШұЫҢЩ…Ш·Ш§Щ„ШЁШЁШұШұШіЫҢ Щ…ЩҒШ§ЩҮЫҢЩ… Щ…ШӘШәЫҢШұЩҮШ§ЫҢ Value Type ЩҲ Reference Type ШҜШұ ШіЫҢ ШҙШ§ШұЩҫЩ…Ш·Ш§Щ„ШЁGarbage Collector ШҜШұ #C - ЩӮШіЩ…ШӘ ШҜЩҲЩ…Щ…Ш·Ш§Щ„ШЁValue Types Ш§ШұШ¬Ш§Ш№ЫҢ ШҜШұ C# 7.2Щ…Ш·Ш§Щ„ШЁШўШҙЩҶШ§ЫҢЫҢ ШЁШ§ CLR: ЩӮШіЩ…ШӘ ШЁЫҢШіШӘ ЩҲ ЪҶЩҮШ§ШұЩ…Щ…Ш·Ш§Щ„ШЁШўЩ…ЩҲШІШҙ ШІШЁШ§ЩҶ Rust - ЩӮШіЩ…ШӘ 8 - Rust-Based CS MasterclassЩ…Ш·Ш§Щ„ШЁ ШҜЩҲШұЩҮвҖҢЩҮШ§ШҙЫҢ ЪҜШұШ§ЫҢЫҢ ШҜШұ #F
