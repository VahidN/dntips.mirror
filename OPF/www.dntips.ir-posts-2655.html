ШЁШұШ§ЫҢ Щ…Ш·Ш§Щ„Ш№ЩҮвҖҢЫҢ Ш§ШҜШ§Щ…ЩҮвҖҢЫҢ Ш§ЫҢЩҶ Щ…ЩӮШ§Щ„ЩҮШҢ ЩҶЫҢШ§ШІ ШЁЩҮ ШўШҙЩҶШ§ЫҢЫҢ ШЁШ§ Щ…ШЁШ§ШӯШ« Щ…Ш·ШұШӯ ШҙШҜЩҮ ШҜШұ ЩӮШіЩ…ШӘ ЩӮШЁЩ„ ЩҲШ¬ЩҲШҜ ШҜШ§ШұШҜ. ЩҫШі Ш§ШІ ШӘЩҲШ¶ЫҢШӯШ§ШӘ Ш§Ш¶Ш§ЩҒЩҮ ШЁШұ ШұЩҲЫҢ ЩӮШіЩ…ШӘвҖҢЩҮШ§ЫҢ ШІЫҢШұ ЪҜШ°ШҙШӘЩҮ ЩҲ ЩҒШұШ¶ ШЁШұ ШўЩҶ Ш§ШіШӘ Ъ©ЩҮ ШўШҙЩҶШ§ЫҢЫҢ ШЁШ§ Ш§ЫҢЩҶ ЩӮШіЩ…ШӘвҖҢЩҮШ§ ЩҲШ¬ЩҲШҜ ШҜШ§ШұШҜ.
Ш§ШІ Ш§ЫҢЩҶ Щ…ШҜЩ„ Щ…ЫҢШӘЩҲШ§ЩҶ ШҜШұ ШІШЁШ§ЩҶвҖҢЩҮШ§ЫҢ Щ…Ш®ШӘЩ„ЩҒ ШЁШұЩҶШ§Щ…ЩҮ ЩҶЩҲЫҢШіЫҢ ЩҲ ЩҮЩ…ЪҶЩҶЫҢЩҶ ШіЫҢШіШӘЩ…вҖҢЩҮШ§ЫҢ Щ…Ш®ШӘЩ„ЩҒ Ш§Ш№Щ… Ш§ШІ ЩҲШЁ Ш§ЩҫЩ„ЫҢЪ©ЫҢШҙЩҶ ЩҲ ... Ш§ШіШӘЩҒШ§ШҜЩҮ ЩҶЩ…ЩҲШҜ. ЩҮЩ…ЪҶЩҶЫҢЩҶ ШЁШұШ§ЫҢ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Ш§ЫҢЩҶ Щ…ШҜЩ„ ЩҶЫҢШ§ШІ ЩӮШ·Ш№ЫҢ ШЁЩҮ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ЩҒШұЫҢЩ… ЩҲШұЪ© Ш®Ш§ШөЫҢ ЩҶЫҢШіШӘ. ШҜШұ ШөЩҲШұШӘ ЩҶЫҢШ§ШІ Щ…ЫҢШӘЩҲШ§ЩҶЫҢШҜ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ ШіЩҒШ§ШұШҙЫҢ Ш®Ш§Шө Ш®ЩҲШҜ ШұШ§ ШҜШ§ШҙШӘЩҮ ШЁШ§ШҙЫҢШҜ. Ш§Щ…Ш§ ШЁШұШ§ЫҢ ШіШ§ШҜЩҮвҖҢШӘШұ ШҙШҜЩҶ ЩҲ ЩҮШұЪҶЩҮ ШіШұЫҢШ№ШӘШұ ШҙШҜЩҶ Щ…ШұШ§ШӯЩ„ Ш§ШІ ЩҒШұЫҢЩ…ЩҲШұЪ© SimpleCqrs Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢЪ©ЩҶЫҢЩ…. ЩҮШұ ЪҶЩҶШҜ ШЁШұ Ш®Щ„Ш§ЩҒ ЩҶШ§Щ…Шҙ Ш§Щ…Ъ©Ш§ЩҶШ§ШӘ ЩҒШұШ§ЩҲШ§ЩҶЫҢ ШұШ§ ШҜШұ Ш§Ш®ШӘЫҢШ§Шұ ШЁШұЩҶШ§Щ…ЩҮ ЩҶЩҲЫҢШіШ§ЩҶ ЩӮШұШ§Шұ Щ…ЫҢШҜЩҮШҜ ЩҲ ШӯШӘЫҢ ШҜШұ ЩҫШұЩҲЪҳЩҮвҖҢЩҮШ§ЫҢ ЩҲШ§ЩӮШ№ЫҢ ЩҶЫҢШІ Щ…ЫҢШӘЩҲШ§ЩҶ Ш§ШІ ШўЩҶ Ш§ШіШӘЩҒШ§ШҜЩҮ ЩҶЩ…ЩҲШҜ.
ШЁШұШ§ЫҢ ШіШұЫҢШ№ШӘШұ ШҙШҜЩҶ Ъ©Ш§Шұ Щ…ЫҢШ®ЩҲШ§ЩҮЫҢЩ… ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ Ш§ЫҢЩҶ Щ…ШҜЩ„ ШұШ§ ШҜШұ ЫҢЪ© ЩҫШұЩҲЪҳЩҮвҖҢЫҢ Console Ш§ЩҶШ¬Ш§Щ… ШҜЩҮЫҢЩ… ЩҲ ЩҮЩ…ЪҶЩҶЫҢЩҶ ЩҫШі Ш§ШІ Ш§ЫҢШ¬Ш§ШҜШҢ ЩҫЪ©ЫҢШ¬вҖҢЩҮШ§ЫҢ ШІЫҢШұ ШұШ§ ЩҶШөШЁ Щ…ЫҢЩҶЩ…Ш§ЫҢЫҢЩ…:
Unity, SimpleCqrs, SimpleCqrs.Unity
Щ…ЫҢШ®ЩҲШ§ЩҮЫҢЩ… Ш·ШЁЩӮ Щ…ШұШ§ШӯЩ„ ЪҜЩҒШӘЩҮ ШҙШҜЩҮвҖҢЫҢ ШҜШұ ЩӮШіЩ…ШӘ ЩӮШЁЩ„ШҢ ШЁЩҮ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ Ш§ЫҢЩҶ Щ…ШҜЩ„ ШЁЩҫШұШҜШ§ШІЫҢЩ… ЩҲ ЩҮШҜЩҒШҢ Ш§Ш¶Ш§ЩҒЩҮ Ъ©ШұШҜЩҶ ЫҢЪ© Account ШЁЩҮ ШіЫҢШіШӘЩ… Ш®ЩҲШ§ЩҮШҜ ШЁЩҲШҜ.
Ш§ШЁШӘШҜШ§ ШЁШ§ЫҢШҜ DomainObject Щ…ЩҲШұШҜ ЩҶШёШұ ЩҶЩҲШҙШӘЩҮ ШҙЩҲШҜ:
using System;
using SimpleCqrs.Domain;

namespace CqrsPattern.Cqrs.Command
{
    public class Account : AggregateRoot
    {
        public Account(Guid id)
        {
            Apply(new AccountCreatedEvent { AggregateRootId = id });
        }

        public void SetName(string firstName, string lastName)
        {
            Apply(new AccountNameSetEvent { FirstName = firstName, LastName = lastName });
        }

        public void OnAccountCreated(AccountCreatedEvent evt)
        {
            Id = evt.AggregateRootId;
        }
    }
}
ЩҶЪ©ШӘЩҮ: Щ…ЫҢШ®ЩҲШ§ЩҮЫҢЩ… Ш№Щ…Щ„ЫҢШ§ШӘ Ш§Ш¶Ш§ЩҒЩҮ Ъ©ШұШҜЩҶ ЫҢЪ© AccountШҢ ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ШҜЩҲ event Щ…ШұШЁЩҲШ·ЩҮ ШЁЩҮ ЩҶШ§Щ… AccountCreatedEvent ЩҲ Щ…ЩӮШҜШ§Шұ ШҜЩҮЫҢ ШўЩҶ ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ AccountNameSetEvent Ш§ЩҶШ¬Ш§Щ… ШҙЩҲШҜ.
eventЩҮШ§ЫҢ ЩҒЩҲЩӮ ШұШ§ ШҜШұ Ш§ШҜШ§Щ…ЩҮ Ш§Ш¶Ш§ЩҒЩҮ Ш®ЩҲШ§ЩҮЫҢЩ… ШҜШ§ШҜ (Ш§ШІ ШӘЩҲШ¶ЫҢШӯШ§ШӘ ШЁЫҢШҙШӘШұ ШөШұЩҒЩҶШёШұ ШҙШҜЩҮ ЩҲ ШЁЩҮ Щ…ЩӮШ§Щ„ЩҮвҖҢЫҢ ЩӮШіЩ…ШӘ ЩӮШЁЩ„ ШұШ¬ЩҲШ№ ШҙЩҲШҜ).
ШӯШ§Щ„ Ш§ШӯШӘЫҢШ§Ш¬ ШЁЩҮ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ Command Щ…ШұШЁЩҲШ·ЩҮ ШЁШұШ§ЫҢ Ш§ЩҶШ¬Ш§Щ… ЩҲШёЫҢЩҒЩҮвҖҢЫҢ Ш®ЩҲШҜ ШҜШ§ШұЫҢЩ… Ъ©ЩҮ ЩҮШҜЩҒ ШўЩҶШҢ Ш§Ш¶Ш§ЩҒЩҮ Ъ©ШұШҜЩҶ ЫҢЪ© Account  ШЁЩҮ ШіЫҢШіШӘЩ… Щ…ЩҲШұШҜ ЩҶШёШұ Щ…ЫҢШЁШ§ШҙШҜ.
ЩҒШұШ¶ Ъ©ЩҶЫҢШҜ ШЁШұШ§ЫҢ Ш§Ш¶Ш§ЩҒЩҮ ШҙШҜЩҶ AccountШҢ ЩҫШұШ§ЩҫШұШӘЫҢвҖҢЩҮШ§ЫҢ FirstName ЩҲ LastName ШЁШ§ЫҢШҜ Щ…ЩӮШҜШ§Шұ ШҜЩҮЫҢ ШҙЩҲЩҶШҜ:
using SimpleCqrs.Commanding;

namespace CqrsPattern.Cqrs.Command
{
    public class CreateAccountCommand : ICommand
    {
        public string FirstName { get; set; }
        public string LastName { get; set; }
    }
}

ШӯШ§Щ„ CommandHandler Ъ©ЩҮ ЩҲШёЫҢЩҒЩҮвҖҢЫҢ ШӘЩҒШіЫҢШұ Ъ©ШұШҜЩҶ Command Щ…ШұШЁЩҲШ·ЩҮ ШұШ§ ШЁЩҮ Ш№ЩҮШҜЩҮ ШҜШ§ШұШҜШҢ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ Ш®ЩҲШ§ЩҮШҜ ШҙШҜ:
using System;
using SimpleCqrs.Commanding;
using SimpleCqrs.Domain;

namespace CqrsPattern.Cqrs.Command
{
    public class CreateAccountCommandHandler : CommandHandler<CreateAccountCommand>
    {
        private readonly IDomainRepository repository;

        public CreateAccountCommandHandler(IDomainRepository repository)
        {
            this.repository = repository;
        }

        public override void Handle(CreateAccountCommand command)
        {
            var account = new Account(Guid.NewGuid());
            account.SetName(command.FirstName, command.LastName);

            repository.Save(account);
        }
    }
}
ЩҶЪ©ШӘЩҮ: Ш§ШІ Ш·ШұЫҢЩӮ account.SetName ЩҒШұШ§Ш®ЩҲШ§ЩҶЫҢ Event Щ…ШұШЁЩҲШ·ЩҮ Ш§ЩҶШ¬Ш§Щ… ШҙШҜЩҮвҖҢШ§ШіШӘ ЩҲ ЩҮЩ…ЪҶЩҶЫҢЩҶ repository.Save ШЁЩҮ raise Ъ©ШұШҜЩҶ EventHandler Щ…ЫҢЩҫШұШҜШ§ШІШҜ.
event Щ…ШұШЁЩҲШ· ШЁЩҮ Ш§Ш¶Ш§ЩҒЩҮ ШҙШҜЩҶ Account ШұШ§ ШЁЩҮ ШөЩҲШұШӘ ШІЫҢШұ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ Щ…ЫҢЩҶЩ…Ш§ЫҢЫҢЩ…:
using SimpleCqrs.Eventing;

namespace CqrsPattern.Cqrs.Command
{
    public class AccountCreatedEvent : DomainEvent { }
}
ЩҲ ЩҮЩ…ЪҶЩҶЫҢЩҶ event Щ…ШұШЁЩҲШ· ШЁЩҮ Щ…ЩӮШҜШ§Шұ ШҜЩҮЫҢ ЩҫШұШ§ЩҫШұШӘЫҢвҖҢЩҮШ§ ЩҶЫҢШІ ШЁЩҮ ШөЩҲШұШӘ ШІЫҢШұ Ш®ЩҲШ§ЩҮШҜ ШЁЩҲШҜ:
using SimpleCqrs.Eventing;

namespace CqrsPattern.Cqrs.Command
{
    public class AccountNameSetEvent : DomainEvent
    {
        public string FirstName { get; set; }

ШҜШұ Ш§ЫҢЩҶ ШЁШ®ШҙШҢ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ EventHandler ШұШ§ Ш®ЩҲШ§ЩҮЫҢЩ… ШҜШ§ШҙШӘ. Ш·ШЁЩӮ Щ…Ш·Щ„ШЁ ЩҫЫҢШҙЫҢЩҶ ЩҮШұ Domain ШЁШ§ЫҢШҜ EventHnadler ЫҢ ШҜШ§ШҙШӘЩҮ ШЁШ§ШҙШҜ Ъ©ЩҮ Ш§ШІ Event ЩҮШ§ЫҢШҙ Ш§ШұШ« ШЁШұЫҢ Ъ©ШұШҜЩҮ ЩҲ ЩҮШұ Ъ©ШҜШ§Щ… Ш§ШІ EventвҖҢЩҮШ§ Ш№Щ…Щ„Ш§ ШҜШұ ЩӮШіЩ…ШӘ Handle Щ…ШұШЁЩҲШ· ШЁЩҮ Ш®ЩҲШҜШҙ ЩҫШұШҜШ§ШІШҙ Ш®ЩҲШ§ЩҮШҜ ШҙШҜ.
using System.Linq;
using SimpleCqrs.Eventing;
using CqrsPattern.Cqrs.Db;

namespace CqrsPattern.Cqrs.Command
{
    public class AccountEventHandler : IHandleDomainEvents<AccountCreatedEvent>,
                                             IHandleDomainEvents<AccountNameSetEvent>
    {
        private readonly FakeAccountTable accountTable;

        public AccountEventHandler(FakeAccountTable accountTable)
        {
            this.accountTable = accountTable;
        }

        public void Handle(AccountCreatedEvent domainEvent)
        {
            accountTable.Add(new FakeAccountTableRow { Id = domainEvent.AggregateRootId });
        }

        public void Handle(AccountNameSetEvent domainEvent)
        {
            var account = accountTable.Single(x => x.Id == domainEvent.AggregateRootId);
            account.Name = domainEvent.FirstName + " " + domainEvent.LastName;
        }
    }
}
