Щ…Ш§ЪҳЩҲЩ„ Http ШҜШұ AngularШҢ ШЁШұШ§ЫҢ ШЁШұЩӮШұШ§ШұЫҢ Ш§ШұШӘШЁШ§Ш· ШЁЫҢЩҶ Ъ©Щ„Ш§ЫҢЩҶШӘ ЩҲ ШіЩ…ШӘ ШіШұЩҲШұШҢ Щ…ЩҲШұШҜ Ш§ШіШӘЩҒШ§ШҜЩҮ ЩӮШұШ§Шұ Щ…ЫҢвҖҢЪҜЫҢШұШҜ. Щ…Ш№Щ…ЩҲЩ„Ш§ ЩҮЩҶЪҜШ§Щ… ШіШ§Ш®ШӘ ШҜШұШ®ЩҲШ§ШіШӘвҖҢЩҮШ§ЫҢ HttpШҢ ЫҢЪ©ШіШұЫҢ Ъ©ШҜЩҮШ§ЫҢ ШӘЪ©ШұШ§ШұЫҢ ШЁШұШ§ЫҢ ШӘЩҶШёЫҢЩ… ЩҮШҜШұ (ШЁШұШ§ЫҢ Ш§Ш№ШӘШЁШ§ШұШіЩҶШ¬ЫҢ ЩҲ ЩҮЩ…ЪҶЩҶЫҢЩҶ ШӘЩҶШёЫҢЩ…Ш§ШӘ ШҜЫҢЪҜШұ) ЩҶЩҲШҙШӘЩҮ Щ…ЫҢвҖҢШҙЩҲЩҶШҜ Ъ©ЩҮ ШҜШұ ЩҮШұ ШҜШұШ®ЩҲШ§ШіШӘ ЫҢЪ©ШіШ§ЩҶ ЩҮШіШӘЩҶШҜ. ЩҮЩ…ЪҶЩҶЫҢЩҶ ШЁШ№ШҜ Ш§ШІ ШўЩ…ШҜЩҶ Ш¬ЩҲШ§ШЁ (Response) Ш§ШІ ШіЩ…ШӘ ШіШұЩҲШұ ЩҶЫҢШІ ЫҢЪ©ШіШұЫҢ Ъ©ШҜЩҮШ§ЫҢ ШӘЪ©ШұШ§ШұЫҢ Ш¬ЩҮШӘ ШЁШұШіЫҢ Ъ©ШҜ response ЩҲ ЫҢШ§ ШӘШәЫҢЫҢШұ ЩҒШұЩ…ШӘ Ш§Ш·Щ„Ш§Ш№Ш§ШӘ ШұШіЫҢШҜЩҮШҢ ШЁЩҮ ШіШ§Ш®ШӘШ§Шұ Щ…ЩҲШұШҜ ШӘЩҲШ§ЩҒЩӮ ЩҶЩҲШҙШӘЩҮ Ш®ЩҲШ§ЩҮЩҶШҜ ШҙШҜ.

ШЁШұШ§ЫҢ Щ…Ш«Ш§Щ„ ШҜШұ ШөЩҲШұШӘЫҢЪ©ЩҮ ШҜШұ ШЁШұЩҶШ§Щ…ЩҮ Ш®ЩҲШҜ Ш§ШІ Ш§Ш№ШӘШЁШ§Шұ ШіЩҶШ¬ЫҢ Щ…ШЁШӘЩҶЫҢ ШЁШұ ШӘЩҲЪ©ЩҶ (Token Base Authentication) Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶЫҢШҜШҢ ЩӮШЁЩ„ Ш§ШІ Ш§ШұШіШ§Щ„ ЩҮШұ ШҜШұШ®ЩҲШ§ШіШӘ (request)ШҢ Ъ©ШҜЩҮШ§ЫҢЫҢ Щ…ШҙШ§ШЁЩҮ Ъ©ШҜ ШІЫҢШұ ШЁШ§ЫҢШҜ ЩҶЩҲШҙШӘЩҮ ШҙЩҲЩҶШҜ:
let headers = new Headers();
let token = localStorage.getItem('token');
headers.append('Authorization', 'bearer ' + token);
this.http.get('/api/controller/action', { headers: headers })

ЩҮЩ…ЪҶЩҶЫҢЩҶ ЩҒШұШ¶ Ъ©ЩҶЫҢШҜ ШЁШ№ШҜ Ш§ШІ ШұШіЫҢШҜЩҶ Ш¬ЩҲШ§ШЁ ЩҮШұ ШҜШұШ®ЩҲШ§ШіШӘШҢ Щ…ЫҢвҖҢШ®ЩҲШ§ЩҮЫҢШҜ response code ШұШ§ ЪҶЪ© Ъ©ЩҶЫҢШҜ ЩҲ Ш®Ш·Ш§ЩҮШ§ЫҢ Ш§ШӯШӘЩ…Ш§Щ„ЫҢ ШұШ§ Щ…ШҜЫҢШұЫҢШӘ Ъ©ЩҶЫҢШҜ. Щ…Ш«Щ„Ш§ ШҜШұШөЩҲШұШӘ ШҜШұЫҢШ§ЩҒШӘ Ъ©ШҜ 401ШҢ Ъ©Ш§ШұШЁШұ ШұШ§ ШЁЩҮ ШөЩҒШӯЩҮ В«ЩҲШұЩҲШҜВ» ЩҲ ШЁШ§ ШҜШұЫҢШ§ЩҒШӘ Ъ©ШҜ 404 ШўЩҶШұШ§ ШұШ§ ШЁЩҮ ШөЩҒШӯЩҮ В«ЫҢШ§ЩҒШӘ ЩҶШҙШҜВ» ЩҮШҜШ§ЫҢШӘ Ъ©ЩҶЫҢШҜ ЩҲ ЫҢШ§ ШЁШ§ ШҜШұЫҢШ§ЩҒШӘ Ъ©ШҜ 403 ЫҢШ§ 500 ЩҫЫҢШәШ§Щ… Щ…ЩҶШ§ШіШЁЫҢ ШұШ§ ЩҶЩ…Ш§ЫҢШҙ ШҜЩҮЫҢШҜ. ШЁШҜЫҢЩҮЫҢ Ш§ШіШӘ ШҜШұ Ш§ЫҢЩҶ ШөЩҲШұШӘ ШЁШ№ШҜ Ш§ШІ ЩҮШұ ШўЩ…ШҜЩҶ ЩҫШ§ШіШ® Ш§ШІ ШіЩ…ШӘ ШіШұЩҲШұ (response)ШҢ Ш§ЫҢЩҶ Ъ©ШҜЩҮШ§ ШЁШ§ЫҢШіШӘЫҢ ШӘЪ©ШұШ§Шұ ШҙЩҲЩҶШҜ.

Ш¬ЩҮШӘ ЩҫШұЩҮЫҢШІ Ш§ШІ Ш§ЫҢЩҶ Ъ©ШҜЩҮШ§ЫҢ ШӘЪ©ШұШ§ШұЫҢШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ ШЁШұШ§ЫҢ Щ…Ш§ЪҳЩҲЩ„ HttpШҢ ЫҢЪ© interceptor ЩҲШ§ШӯШҜ ШҜШұЩҶШёШұ ЪҜШұЩҒШӘ Ъ©ЩҮ ШӘЩ…Ш§Щ…ЫҢ Ъ©ШҜЩҮШ§ЫҢ ШӘЪ©ШұШ§ШұЫҢ ШұШ§ ШӘЩҶЩҮШ§ ЫҢЪ©ШЁШ§Шұ ШҜШ§Ш®Щ„ ШўЩҶ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ Ъ©ШұШҜ. Щ…ШІЫҢШӘ Ш§ЫҢЩҶ ШұЩҲШҙШҢ Щ…ШҜЫҢШұЫҢШӘ ШұШ§ШӯШӘ Ъ©ШҜШҢ Ъ©Ш§ЩҮШҙ ЩҫЫҢЪҶЫҢШҜЪҜЫҢвҖҢЩҮШ§ ЩҲ ЩҮЩ…ЪҶЩҶЫҢЩҶ ШӯШ°ЩҒ Ъ©ШҜЩҮШ§ЫҢ ШӘЪ©ШұШ§ШұЫҢ ЩҲ ЫҢЪ©ШіШ§ЩҶ ШіШ§ШІЫҢ ШўЩҶЩҮШ§ Ш§ШіШӘ.
ЩҮШұЪҶЩҶШҜ ШҜШұ Angular ШҜЫҢЪҜШұ ШЁЩҮ Щ…Ш§ЩҶЩҶШҜ Angular 1.x Щ…ЩҒЩҮЩҲЩ… intercept ШЁШұ ШұЩҲЫҢ Http ШұШ§ ШЁЩҮ ШөЩҲШұШӘ ШӘЩҲЪ©Ш§Шұ ЩҶШҜШ§ШұЫҢЩ…ШҢ ЩҲЩ„ЫҢ ШЁЩҮ ШҜЩ„Ш§ЫҢЩ„ ШІЫҢШұ Щ…Ш§ ЩҶЫҢШ§ШІ ШЁЩҮ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ interceptor ШЁШұШ§ЫҢ Щ…Ш§ЪҳЩҲЩ„ Http ШұШ§ ШҜШ§ШұЫҢЩ…:
- ШӘЩҶШёЫҢЩ… ЩҮШҜШұЩҮШ§ЫҢ ШіЩҒШ§ШұШҙЫҢ ЩҲ Ш§ШөЩ„Ш§Шӯ ШўШҜШұШіШҢ ЩӮШЁЩ„ Ш§ШІ Ш§ШұШіШ§Щ„ ШҜШұШ®ЩҲШ§ШіШӘ ШЁЩҮ ШіЩ…ШӘ ШіШұЩҲШұ
- ШӘЩҶШёЫҢЩ… token ШҜШұ ЩҮШҜШұ ШҜШұШ®ЩҲШ§ШіШӘШҢ Ш¬ЩҮШӘ Ш§Ш№ШӘШЁШ§Шұ ШіЩҶШ¬ЫҢ
- Щ…ШҜЫҢШұЫҢШӘ ШіШұШ§ШіШұЫҢ Ш®Ш·Ш§ЩҮШ§ЫҢ Http
- Ш§ЩҶШ¬Ш§Щ… ЩҮШұЪҜЩҲЩҶЩҮ Ш№Щ…Щ„ЫҢШ§ШӘ crosscutting

ШӯШ§Щ„Ш§ Ъ©ЩҮ Angular Щ…ЩҒЩҮЩ…ЩҲЩ… intercept ШұШ§ ШЁШұШ§ЫҢ Щ…Ш§ЪҳЩҲЩ„ Http Ш®ЩҲШҜ ШЁЩҮ ШөЩҲШұШӘ ШӘЩҲЪ©Ш§Шұ ШҜШұЩҶШёШұ ЩҶЪҜШұЩҒШӘЩҮ Ш§ШіШӘШҢ ШұШ§ЩҮвҖҢШӯЩ„ ЪҶЫҢШіШӘШҹ ШЁЩҮШӘШұЫҢЩҶ ШұШ§ЩҮвҖҢШӯЩ„ ШЁШұШ§ЫҢ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ Щ…ЩҲШ§ШұШҜ Щ…Ш·ШұШӯ ШҙШҜЩҮ ШҜШұ ШЁШ§Щ„Ш§ШҢ Ш§ШұШ« ШЁШұЫҢ ЩҲ ЫҢШ§ ЪҜШіШӘШұШҙ (extend) Щ…ШіШӘЩӮЫҢЩ… Щ…Ш§ЪҳЩҲЩ„ Http Ш§ШіШӘ:
import { Injectable } from "@angular/core";
import { ConnectionBackend, RequestOptions, Request, RequestOptionsArgs, Response, Http, Headers } from "@angular/http";
import { Observable } from "rxjs/Rx";
import 'rxjs/add/operator/catch';
import 'rxjs/add/observable/throw';

@Injectable()
export class InterceptedHttp extends Http {
    constructor(backend: ConnectionBackend, defaultOptions: RequestOptions) {
        super(backend, defaultOptions);
    }

    request(url: string | Request, options?: RequestOptionsArgs): Observable<Response> {
        // Ш§ШөЩ„Ш§Шӯ url
        if (typeof (url) === 'string')
            url = this.updateUrl((url as string));
        else
            url.url = this.updateUrl((url as Request).url);

        return super.request(url, this.getRequestOptionArgs(options)).catch((err: Response) => {
            // Exception handling
            switch (err.status) {
                case 400:
                    console.log('interceptor: 400');
                    console.log(err);
                    break;
                case 404:
                    console.log('interceptor: 404');
                    console.log(err);
                    break;
                case 500:
                    console.log('interceptor: 500');
                    console.log(err);
                    break;
                case 401:
                    console.log('interceptor: 401');
                    console.log(err);
                    break;
                case 403:
                    console.log('interceptor: 403');
                    console.log(err);
                    break;
                default:
                    console.log('interceptor: ' + err.status);
                    console.log(err);
            }
            return Observable.throw(err);

        });
    }

    private updateUrl(req: string) {
        return `http://localhost:61366/api/${req}`
    }

    private getRequestOptionArgs(options?: RequestOptionsArgs): RequestOptionsArgs {
        if (options == null) {
            options = new RequestOptions();
        }
        if (options.headers == null) {
            options.headers = new Headers();
        }
        // ЩҮШҜШұ ШҜШұШ®ЩҲШ§ШіШӘ ШӘЩҶШёЫҢЩ…
        let token = localStorage.getItem('token');
        options.headers.append('Authorization', 'bearer ' + token);


        return options;
    }
}
ШӘЩ…Ш§Щ…ЫҢ Ш§ЩҒШ№Ш§Щ„ HttpШҢ Ш§ШІ Ш¬Щ…Щ„ЩҮ get ШҢpost ШҢput ШҢdelete ШҢpatch ШҢhead ЩҲ options ШҜШұ ЩҶЩҮШ§ЫҢШӘ Ш§ШІ Щ…ШӘШҜ request Щ…ЩҲШ¬ЩҲШҜ ШҜШұ Щ…Ш§ЪҳЩҲЩ„ http ШЁШұШ§ЫҢ Ш§ШұШіШ§Щ„ ШҜШұШ®ЩҲШ§ШіШӘ Ш®ЩҲШҜ Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶЩҶШҜ. ШЁЩҮ ЩҮЩ…ЫҢЩҶ Ш¬ЩҮШӘ ШӘЩ…Ш§Щ…ЫҢ Ш№Щ…Щ„ЫҢШ§ШӘ crosscutting ШұШ§ ШҜШұ Ш§ЫҢЩҶ Щ…ШӘШҜ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ Ъ©ШұШҜЩҮвҖҢШ§ЫҢЩ…. ШЁЩҮ Ш№Щ„Ш§ЩҲЩҮ ЩӮШЁЩ„ Ш§ШІ Ш§ШұШіШ§Щ„ ШҜШұШ®ЩҲШ§ШіШӘШҢ ШӘЩҲШіШ· Щ…ШӘШҜ updateUrl ШўШҜШұШі url Ш®ЩҲШҜ ШұШ§ Ш§ШөЩ„Ш§Шӯ Ъ©ШұШҜЩҮвҖҢШ§ЫҢЩ…. ЩҮЩ…ЪҶЩҶЫҢЩҶ ШӘЩҲШіШ· Щ…ШӘШҜ getRequestOptionArgs ЩҮШҜШұ ШҜШұШ®ЩҲШ§ШіШӘ ШұШ§ Ш¬ЩҮШӘ Ш§Ш№ШӘШЁШ§Шұ ШіЩҶШ¬ЫҢ Щ…ЩӮШҜШ§ШұШҜЩҮЫҢ Ъ©ШұШҜЩҮвҖҢШ§ЫҢЩ…. ШҜШұ Ш§ЫҢЩҶШ¬Ш§ ШЁШ№ШҜ Ш§ШІ Ш§ШұШіШ§Щ„ ШҜШұШ®ЩҲШ§ШіШӘ ЩҲ ШўЩ…ШҜЩҶ response Ш§ШІ ШіЩ…ШӘ ШіШұЩҲШұШҢ ШӘЩҲШіШ· catch Ш®Ш·Ш§ЩҮШ§ЫҢ Ш§ШӯШӘЩ…Ш§Щ„ЫҢ ШұШ§ ШЁШұШіЫҢ Ъ©ШұШҜЩҮвҖҢШ§ЫҢЩ….

ЩҶЪ©ШӘЩҮ: ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ Щ…Ш«Ш§Щ„ШҢ ШҜШұ ШөЩҲШұШӘЫҢЪ©ЩҮ ЩӮШөШҜ ШҜШ§ШұЫҢШҜШҢ ШЁШұШ§ЫҢ ШҜШұШ®ЩҲШ§ШіШӘвҖҢЩҮШ§ЫҢЫҢ Ш§ШІ Ш¬ЩҶШі getШҢ ЩҮШҜШұ Щ…ШӘЩҒШ§ЩҲШӘЫҢ ЩҶШіШЁШӘ ШЁЩҮ ШҜЫҢЪҜШұ ШҜШұШ®ЩҲШ§ШіШӘвҖҢЩҮШ§ ШҜШ§ШҙШӘЩҮ ШЁШ§ШҙЫҢШҜШҢ ШўЩҶЪҜШ§ЩҮ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ Ш№Щ…Щ„ЫҢШ§ШӘ Ш§ШөЩ„Ш§Шӯ ЩҮШҜШұ ШҜШұ Щ…ШӘШҜ request Ш¬ЩҲШ§ШЁ Ъ©Ш§Шұ ШұШ§ ЩҶШ®ЩҲШ§ЩҮШҜ ШҜШ§ШҜ. ШЁШұШ§ЫҢ ШӯЩ„ Ш§ЫҢЩҶ Щ…ЩҲШ¶ЩҲШ№ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЫҢШҜ ШЁЩҮ Ш¬Ш§ЫҢ Ш§ШөЩ„Ш§Шӯ header ШҜШұ Щ…ШӘШҜ requestШҢ ШӘЩ…Ш§Щ…ЫҢ Щ…ШӘШҜЩҮШ§ЫҢ get ШҢpostШҢ put ШҢdelete ШҢpatch ШҢhead ЩҲ options ШұШ§ ШЁШ§ШІ ЩҶЩҲЫҢШіЫҢ Ъ©ШұШҜЩҮ ЩҲ ШҜШұ ЩҮШұЪ©ШҜШ§Щ… Ш§ШІ Ш§ЫҢЩҶ Щ…ШӘШҜЩҮШ§ Ш§ЫҢЩҶЪ©Ш§Шұ ШұШ§ Ш§ЩҶШ¬Ш§Щ… ШҜЩҮЫҢШҜ.

ШӯШ§Щ„Ш§ ШЁШ§ ШӘШәЫҢЫҢШұ ЩӮШіЩ…ШӘ providers ШҜШұ Щ…Ш§ЪҳЩҲЩ„ Ш§ШөЩ„ЫҢ ШЁШұЩҶШ§Щ…ЩҮ ШЁЩҮ ШҙЪ©Щ„ ШІЫҢШұШҢ Angular ШұШ§ Щ…Ш¬ШЁЩҲШұ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ… ШЁШ¬Ш§ЫҢ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Щ…Ш§ЪҳЩҲЩ„ Http ШӘЩҲЪ©Ш§Шұ Ш®ЩҲШҜШҢ Ш§ШІ Щ…Ш§ЪҳЩҲЩ„ Ш¬ШҜЫҢШҜ InterceptedHttp Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ЩҶШҜ: 
//вҖҰ
providers: [{
        provide: Http,
        useFactory: (backend: XHRBackend, options: RequestOptions) => {
            return new InterceptedHttp(backend, options);
        },
        deps: [XHRBackend, RequestOptions],
    }],
//вҖҰ

ЩҮЩ…ЩҮ ЪҶЫҢШІ ШўЩ…Ш§ШҜЩҮ Ш§ШіШӘ. Ш§Ъ©ЩҶЩҲЩҶ Ъ©Ш§ЩҒЫҢ Ш§ШіШӘ Щ…Ш§ЪҳЩҲЩ„ Http ШұШ§ ШҜШұ ШіШұЩҲЫҢШі ЫҢШ§ Ъ©Ш§Щ…ЩҫЩҲЩҶЩҶШӘвҖҢЩҮШ§ЫҢ Ш®ЩҲШҜ ШӘШІШұЫҢЩӮ Ъ©ШұШҜЩҮ ЩҲ ШҜШұШ®ЩҲШ§ШіШӘвҖҢЩҮШ§ЫҢ Http ШұШ§ ШЁШҜЩҲЩҶ ЩҮЫҢЪҶЪҜЩҲЩҶЩҮ ЩҶЩҲШҙШӘЩҶ Ъ©ШҜ Ш§Ш¶Ш§ЩҒЫҢ ШЁШұШ§ЫҢ ШӘЩҶШёЫҢЩ… ЩҮШҜШұ ЩҲ ШәЫҢШұЩҮ (ШЁШ§ ЩҒШұШ¶ Ш§ЫҢЩҶЪ©ЩҮ ШӘЩ…Ш§Щ…ЫҢ ШўЩҶЩҮШ§ ШҜШұ Щ…ШӘШҜ request Ш§ШІ Щ…Ш§ЪҳЩҲЩ„ http ЩҶЩҲШҙШӘЩҮ ШҙШҜЩҮвҖҢШ§ЩҶШҜ)ШҢ ШЁЩҮ Щ…Ш§ЩҶЩҶШҜ ЩӮШЁЩ„ ШөШ§ШҜШұ Ъ©ЩҶЫҢШҜ. ШЁШұШ§ЫҢ ЩҶЩ…ЩҲЩҶЩҮ Ъ©ШҜ ШІЫҢШұ ШұШ§ ШЁШЁЫҢЩҶЫҢШҜ.
import { Http, URLSearchParams } from '@angular/http';

//вҖҰ
constructor(private _http: Http) { }

ngOnInit() {
    let urlSearchParams: URLSearchParams = new URLSearchParams();
    urlSearchParams.append('page', page.toString());
    urlSearchParams.append('count', count.toString());
    let params = urlSearchParams.toString();
    this._http.get(`/cars`, { params: params })
        .subscribe(result => {
            console.log('service: Succ');
            this.cars = result.json();
        }, err => {
            console.log('service: error');
        });
}
//вҖҰ
ШЁШ§ Ш§ЫҢЩҶЪ©ЩҮ Angular Ш§ШІ interceptor ЩҫШҙШӘЫҢШЁШ§ЩҶЫҢ ЩҶЩ…ЫҢвҖҢЪ©ЩҶШҜШҢ ЩҲЩ„ЫҢ Ъ©ШӘШ§ШЁШ®Ш§ЩҶЩҮвҖҢЩҮШ§ЫҢЫҢ ШЁШұШ§ЫҢ Ш§ЫҢШ¬Ш§ШҜ ЩӮШ§ШЁЩ„ЫҢШӘ Щ…ШҙШ§ШЁЩҮ interceptor ШЁЩҮ ЩҲШ¬ЩҲШҜ ШўЩ…ШҜЩҮвҖҢШ§ЩҶШҜ Ъ©ЩҮ ШЁШұШ®ЫҢ Ш§ШІ ШўЩҶЩҮШ§ Ш№ШЁШ§ШұШӘЩҶШҜ Ш§ШІ:  angular2-cool-http ШҢ ng2-http-interceptor ШҢ ng2-interceptors . ШЁЩҮ Ш¬Ш§ЫҢ extend Щ…ШіШӘЩӮЫҢЩ… Щ…Ш§ЪҳЩҲЩ„ Http ШӘЩҲШіШ· Ш®ЩҲШҜШӘШ§ЩҶШҢ Ш§ЫҢЩҶЪ©Ш§Шұ ШұШ§ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЫҢШҜ ШЁЩҮ Ш§ЫҢЩҶ Ъ©ШӘШ§ШЁШ®Ш§ЩҶЩҮвҖҢЩҮШ§ ШЁШіЩҫШ§ШұЫҢШҜ.

    ЫҢЪ© ЩҶЪ©ШӘЩҮвҖҢЫҢ ШӘЪ©Щ…ЫҢЩ„ЫҢ
    ШЁЩҮ ЩҮЩ…ШұШ§ЩҮ ЩҶЪҜШ§ШұШҙ Angular 4.3ШҢ ШұЩҲШҙ Ш¬ШҜЫҢШҜЫҢ ШЁШұШ§ЫҢ Ъ©Ш§Шұ ШЁШ§ HTTPШҢ ШӘЩҲШіШ· Щ…Ш§ЪҳЩҲЩ„ Ш¬ШҜЫҢШҜ HTTP Client ШўЩҶ Ш§ШұШ§ШҰЩҮ ШҙШҜЩҮвҖҢШ§ШіШӘ Ъ©ЩҮ ШіШ§Ш®ШӘШ§Шұ ШўЩҶ ШЁШіЫҢШ§Шұ ШҙШЁЫҢЩҮ ШЁЩҮ Щ…Ш§ЪҳЩҲЩ„ ЩҒШ№Щ„ЫҢ HTTP ШўЩҶ Ш§ШіШӘ ЩҲ Ъ©ШҜЩҮШ§ЫҢ ЩҒШ№Щ„ЫҢ ШұШ§ ШЁЩҮ ШіШ§ШҜЪҜЫҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ ШЁЩҮ ШўЩҶ Ш§ЩҶШӘЩӮШ§Щ„ ШҜШ§ШҜ. ЫҢЪ©ЫҢ Ш§ШІ ШӘШәЫҢЫҢШұШ§ШӘ ШўЩҶ ШҜШ§ШҙШӘЩҶ HttpInterceptor ШЁЩҮ ШөЩҲШұШӘ ШӘЩҲЪ©Ш§Шұ Ш§ШіШӘ:
    import { HttpRequest, HttpHandler, HttpEvent } from '@angular/common/http';

@Injectable()
class JWTInterceptor implements HttpInterceptor {
    constructor(private userService: UserService) {}

    intercept(req: HttpRequest<any>, next: HttpHandler): Observable<HttpEvent<any>> {
          const JWT = `Bearer ${this.userService.getToken()}`;
          req = req.clone({
                     setHeaders: {
                        Authorization: JWT
                     }
                });
          return next.handle(req);
    }
}
