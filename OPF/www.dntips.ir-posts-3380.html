ШіШ§Ш®ШӘШ§Шұ Ш§ШЁШӘШҜШ§ЫҢЫҢ Щ…ЩҲШ¬ЩҲШҜЫҢШӘ ЩҶЩҲЫҢШіЩҶШҜЪҜШ§ЩҶ Щ…Ш·Ш§Щ„ШЁ ШЁЩ„Ш§ЪҜ

Ш§ЫҢЩҶ Щ…ЩҲШ¬ЩҲШҜЫҢШӘвҖҢЩҮШ§ ШҜШұ ЩҫЩҲШҙЩҮвҖҢЫҢ Ш¬ШҜЫҢШҜЫҢ ШЁЩҮ ЩҶШ§Щ… Model ЩҫШұЩҲЪҳЩҮвҖҢЫҢ MinimalBlog.Domain Ш§Ш¶Ш§ЩҒЩҮ Ш®ЩҲШ§ЩҮЩҶШҜ ШҙШҜ:

ШҜШұ Ш§ЫҢЩҶШ¬Ш§ ШӘШ№Ш§ШұЫҢЩҒЫҢ Щ…Ш§ЩҶЩҶШҜ !default ШұШ§ ЩҮЩ… Щ…ШҙШ§ЩҮШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶЫҢШҜ Ъ©ЩҮ Щ…ШұШӘШЁШ· Ш§ШіШӘ ШЁШ§ ЩҒШ№Ш§Щ„ ШЁЩҲШҜЩҶ nullable reference types ШҜШұ Ш§ЫҢЩҶ ЩҫШұЩҲЪҳЩҮ Ъ©ЩҮ ШҜШұ ЩҒШ§ЫҢЩ„ Directory.Build.props ШЁЩҮ ШөЩҲШұШӘ ШіШұШ§ШіШұЫҢ ШЁЩҮ ШӘЩ…Ш§Щ… ЩҫШұЩҲЪҳЩҮвҖҢЩҮШ§ Ш§Ш№Щ…Ш§Щ„ Щ…ЫҢвҖҢШҙЩҲШҜ. ШЁШ§ ШӘШ№ШұЫҢЩҒ !default ШЁЩҮ Ъ©Ш§Щ…ЩҫШ§ЫҢЩ„Шұ Ш§Ш№Щ„Ш§Щ… Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ… Ъ©ЩҮ Ш§ЫҢЩҶ Ш®ЩҲШ§Шө ЩҮЫҢЪҶЪҜШ§ЩҮ ЩҶШ§Щ„ ЩҶШ®ЩҲШ§ЩҮЩҶШҜ ШЁЩҲШҜ. ЩҮШҜЩҒ Ш§ШөЩ„ЫҢ Ш§ШІ nullable reference typesШҢ ШЁШ§Щ„Ш§ ШЁШұШҜЩҶ ЩӮШ§ШЁЩ„ЫҢШӘ ЩҫЫҢШҙ ШЁЫҢЩҶЫҢ ЩҶШ§Щ„ ШЁЩҲШҜЩҶШҢ ЫҢШ§ ЩҶШЁЩҲШҜЩҶ ШўЩҶвҖҢЩҮШ§ Ш§ШіШӘШӣ ШӘШ§ ШЁШұЩҶШ§Щ…ЩҮ ЩҶЩҲЫҢШі ШҜШұ ЩӮШіЩ…ШӘвҖҢЩҮШ§ЫҢ Щ…Ш®ШӘЩ„ЩҒ ШЁШұЩҶШ§Щ…ЩҮ ШЁШҜШ§ЩҶШҜ Ъ©ЩҮ ШўЫҢШ§ ЩҲШ§ЩӮШ№Ш§Щ„ ЩҶЫҢШ§ШІ Ш§ШіШӘ ЩҮЩҶЪҜШ§Щ… Ъ©Ш§Шұ ШЁШ§ Ш®Ш§ШөЫҢШӘ FirstNameШҢ ЩҶШ§Щ„ ШЁЩҲШҜЩҶ ШўЩҶвҖҢШұШ§ ЩҫЫҢШҙ Ш§ШІ Ш§ШіШӘЩҒШ§ШҜЩҮ ШЁШұШұШіЫҢ Ъ©ЩҶШҜ ЫҢШ§ Ш®ЫҢШұШҹ

ШіШ§Ш®ШӘШ§Шұ Ш§ШЁШӘШҜШ§ЫҢЫҢ Щ…ЩҲШ¬ЩҲШҜЫҢШӘ Щ…ЩӮШ§Щ„Ш§ШӘ ШЁЩ„Ш§ЪҜ

Щ„Ш§ЫҢЩҮвҖҢЫҢ Dal Ш¬Ш§ЫҢЫҢ Ш§ШіШӘ Ъ©ЩҮ DbContext ШЁШұЩҶШ§Щ…ЩҮ ШӘШ№ШұЫҢЩҒ Щ…ЫҢвҖҢШҙЩҲШҜ ЩҲ Щ…ЩҲШ¬ЩҲШҜЫҢШӘвҖҢЩҮШ§ ЩҒЩҲЩӮ ШӘЩҲШіШ· DbSetЩҮШ§ ШҜШұ Щ…Ш№ШұШ¶ ШҜЫҢШҜ EF-Core ЩӮШұШ§Шұ Щ…ЫҢвҖҢЪҜЫҢШұЩҶШҜ. ШЁЩҮ ЩҮЩ…ЫҢЩҶ Ш¬ЩҮШӘ Ш§ШЁШӘШҜШ§ ЩҒШ§ЫҢЩ„ MinimalBlog.Dal.csproj ШұШ§ ШЁЩҮ ШөЩҲШұШӘ ШІЫҢШұ ШӘШәЫҢЫҢШұ Щ…ЫҢвҖҢШҜЩҮЫҢЩ…:

ШҜШұ Ш§ЫҢЩҶШ¬Ш§ ШҜШұ Ш§ШЁШӘШҜШ§ Ш§ШұШ¬Ш§Ш№ЫҢ ШЁЩҮ ЩҫШұЩҲЪҳЩҮвҖҢЫҢ MinimalBlog.Domain.csproj Ш§Ш¶Ш§ЩҒЩҮ ШҙШҜЩҮвҖҢШ§ШіШӘ. ШіЩҫШі ШЁШіШӘЩҮвҖҢЩҮШ§ЫҢ Щ…ЩҲШұШҜ ЩҶЫҢШ§ШІ Ш§ШІ EF-Core ЩҶЫҢШІ Ш¬ЩҮШӘ ШӘШ№ШұЫҢЩҒ DbSetЩҮШ§ ЩҲ Ш§Ш¬ШұШ§ЫҢ MigrationsШҢ Ш§Ш¶Ш§ЩҒЩҮ ШҙШҜЩҮвҖҢШ§ЩҶШҜ.
ШЁЩҮ Ш№Щ„Ш§ЩҲЩҮ ШӘШ№Ш§ШұЫҢЩҒЫҢ Щ…Ш§ЩҶЩҶШҜ ImplicitUsings ЩҲ Nullable ШұШ§ ЩҮЩ… Щ…ШҙШ§ЩҮШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶЫҢШҜ Ъ©ЩҮ ШЁШ§ ШӘЩҲШ¬ЩҮ ШЁЩҮ Ш§ШіШӘЩҒШ§ШҜЩҮвҖҢЫҢ Ш§ШІ ЩҒШ§ЫҢЩ„ Directory.Build.props ШәЫҢШұШ¶ШұЩҲШұЫҢ ЩҲ ШӘЪ©ШұШ§ШұЫҢ ЩҮШіШӘЩҶШҜШӣ ЪҶЩҲЩҶ Ш§ЫҢЩҶ ЩҒШ§ЫҢЩ„ Щ…Ш®ШөЩҲШө ШЁЩҮ ЩҮЩ…ШұШ§ЩҮ Ш§ЫҢЩҶ ШӘШ№Ш§ШұЫҢЩҒ ШіШұШ§ШіШұЫҢ ЩҶЫҢШІ ЩҮШіШӘ.

ШіЩҫШі ШЁЩҮ ЩҫШұЩҲЪҳЩҮвҖҢЫҢ DalШҢ Ъ©Щ„Ш§Ші Ш¬ШҜЫҢШҜ MinimalBlogDbContext ШұШ§ ШЁЩҮ ШөЩҲШұШӘ ШІЫҢШұ Ш§Ш¶Ш§ЩҒЩҮ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ… ШӘШ§ Щ…ШҜЩ„вҖҢЩҮШ§ЫҢ ШЁШұЩҶШ§Щ…ЩҮ ШұШ§ ШҜШұ Щ…Ш№ШұШ¶ ШҜЫҢШҜ EF-Core ЩӮШұШ§Шұ ШҜЩҮШҜ:
using Microsoft.EntityFrameworkCore;
using MinimalBlog.Domain.Model;

namespace MinimalBlog.Dal;

public class MinimalBlogDbContext : DbContext
{
    public MinimalBlogDbContext(DbContextOptions options) : base(options)
    {
    }

    public DbSet<Article> Articles { get; set; } = default!;
    public DbSet<Category> Categories { get; set; } = default!;
    public DbSet<Author> Authors { get; set; } = default!;
    public DbSet<Blog> Blogs { get; set; } = default!;
}
ШҜШұ Ш§ЫҢЩҶШ¬Ш§ ЩҶЫҢШІ ШӘШ№Ш§ШұЫҢЩҒ !default ШұШ§ Щ…ШҙШ§ЩҮШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶЫҢШҜ Ъ©ЩҮ Ш®Ш§ШөЫҢШӘ ШұШ§ЩҮЩҶЩ…Ш§ЫҢ Ъ©Ш§Щ…ЩҫШ§ЫҢЩ„Шұ ШұШ§ ШҜШұ ШӯШ§Щ„ШӘ ЩҒШ№Ш§Щ„ ШЁЩҲШҜЩҶ <Nullable>enable</Nullable> ШҜШ§ШұЩҶШҜ ЩҲ ЩҮШҜЩҒ Ш§ШІ ШўЩҶШҢ Ш§Ш№Щ„Ш§Щ… ЩҶШ§Щ„ ЩҶШЁЩҲШҜЩҶ Ш§ЫҢЩҶ Ш®ЩҲШ§ШөШҢ ШҜШұ ШӯЫҢЩҶ Ш§ШіШӘЩҒШ§ШҜЩҮвҖҢЫҢ Ш§ШІ ШўЩҶвҖҢЩҮШ§ ШҜШұ ЩӮШіЩ…ШӘвҖҢЩҮШ§ЫҢ Щ…Ш®ШӘЩ„ЩҒ ШЁШұЩҶШ§Щ…ЩҮ Ш§ШіШӘ.


Ш§ЫҢШ¬Ш§ШҜ ЩҲ Ш§Ш¬ШұШ§ЫҢ Migrations

ЩҫШі Ш§ШІ ШӘШ№ШұЫҢЩҒ MinimalBlogDbContextШҢ Ш§Ъ©ЩҶЩҲЩҶ ЩҶЩҲШЁШӘ ШЁЩҮ Ш§ЫҢШ¬Ш§ШҜ Ъ©Щ„Ш§ШівҖҢЩҮШ§ЫҢ Migrations ЩҲ Ш§Ш¬ШұШ§ЫҢ ШўЩҶвҖҢЩҮШ§ Ш¬ЩҮШӘ Ш§ЫҢШ¬Ш§ШҜ ШЁШ§ЩҶЪ© Ш§Ш·Щ„Ш§Ш№Ш§ШӘЫҢ Щ…ШӘЩҶШ§ШёШұ ШЁШ§ Щ…ШҜЩ„вҖҢЩҮШ§ЫҢ ШЁШұЩҶШ§Щ…ЩҮ Ш§ШіШӘ. ШЁШұШ§ЫҢ Ш§ЫҢЩҶ Щ…ЩҶШёЩҲШұ ШҜШұ Ш§ШЁШӘШҜШ§ ШЁЩҮ ЩҫШұЩҲЪҳЩҮвҖҢЫҢ Api Щ…ШұШ§Ш¬Ш№ЩҮ Ъ©ШұШҜЩҮ ЩҲ ЩҒШ§ЫҢЩ„ appsettings.json ШұШ§ ШЁЩҮ ШөЩҲШұШӘ ШІЫҢШұ ШӘЪ©Щ…ЫҢЩ„ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ…:

var connectionString = Configuration.GetConnectionString(connectionStringKey);

switch (connectionStringKey)
{
    case "SqlServerConnection":
        services.AddScoped<MinimalBlogDbContext, SqlServerDbContext>();
        services.AddDbContext<SqlServerDbContext>(options => opt.UseSqlServer(connectionString));
        break;
    case "SqliteConnection":
        services.AddScoped<MinimalBlogDbContext, SqliteDbContext>();
        services.AddDbContext<SqliteDbContext>(options => options.UseSqlite(connectionString));
        break;
    default:
        throw new NotImplementedException($"`{connectionStringKey}` is not defined in `appsettings.json` file.");
