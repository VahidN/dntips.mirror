      $(".from-date").pDatepicker({
                format: "YYYY/MM/DD dddd",
                autoClose: true,
                onSelect: function (unixDate) {
                    console.log(unixDate);
                    $("#start").val(unixDate);
                    return this;
                },
                navigator: {
                    enabled: true,
                    text: {
                        btnNextText: ">",
                        btnPrevText: "<"
                    },
                },
            });
