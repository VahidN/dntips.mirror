ШұЩҲШҙ Щ…ШұШіЩҲЩ… Щ…ЩӮШ§ШЁЩ„ЩҮ ШЁШ§ ШӯЩ…Щ„Ш§ШӘ CSRF ШҜШұ ASP.NET MVCШҢ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ЩҒЫҢЩ„ШӘШұ Ш§Щ…ЩҶЫҢШӘЫҢ ValidateAntiForgeryToken ШЁШұ ШұЩҲЫҢ Ш§Ъ©ШҙЩҶ Щ…ШӘШҜЩҮШ§ЫҢЫҢ Ш§ШІ ЩҶЩҲШ№ Post Ш§ШіШӘ ЩҲ ШіЩҫШі ЩҒШұШ§Ш®ЩҲШ§ЩҶЫҢ Html.AntiForgeryToken ШҜШұ View Щ…ШӘЩҶШ§ШёШұ.
ШЁШ§ ШЁШ§Щ„Ш§ ШұЩҒШӘЩҶ ШӘШ№ШҜШ§ШҜ Ш§Ъ©ШҙЩҶ Щ…ШӘШҜЩҮШ§ЫҢ ЫҢЪ© ЩҫШұЩҲЪҳЩҮШҢ Щ…Щ…Ъ©ЩҶ Ш§ШіШӘ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ValidateAntiForgeryToken ЩҒШұШ§Щ…ЩҲШҙ ШҙЩҲШҜ. ШҜШұ Ш§ШҜШ§Щ…ЩҮ Щ…Ш«Ш§Щ„ЫҢ ШұШ§ Щ…Щ„Ш§ШӯШёЩҮ Щ…ЫҢвҖҢЪ©ЩҶЫҢШҜ Ъ©ЩҮ ЫҢЪ© ЩҫШұЩҲЪҳЩҮвҖҢЫҢ ASP.NET MVC ШұШ§ Ш¬ЩҮШӘ ЫҢШ§ЩҒШӘЩҶ Ш§Ъ©ШҙЩҶ Щ…ШӘШҜЩҮШ§ЫҢ Post Ш§ЫҢЫҢ Ъ©ЩҮ ЩҒЫҢЩ„ШӘШұ ValidateAntiForgeryToken ЩҶШҜШ§ШұЩҶШҜШҢ Ш§ШіЪ©ЩҶ Щ…ЫҢвҖҢЪ©ЩҶШҜ:

using System;
using System.Linq;
using System.Reflection;
// Add a ref. to \Program Files\Microsoft ASP.NET\ASP.NET MVC 4\Assemblies\System.Web.Mvc.dll
using System.Web.Mvc;
// Add a ref. to System.Web
using System.Web.UI;

namespace FindOutputCaches
{
    class Program
    {
        static void Main(string[] args)
        {
            var path = @"D:\path\bin\site.dll";
            var asmTarget = Assembly.LoadFrom(path);

            checkCsrfTokens(asmTarget);

            Console.WriteLine("Press a key...");
            Console.Read();
        }


        private static void checkCsrfTokens(Assembly asmTarget)
        {
            // ЫҢШ§ЩҒШӘЩҶ Ъ©Щ„ЫҢЩҮ Ъ©ЩҶШӘШұЩ„ШұЩҮШ§
            var controllers = asmTarget.GetTypes()
                                       .Where(type => typeof(IController).IsAssignableFrom(type) &&
                                                     !type.Name.StartsWith("T4MVC"))
                                       .ToList();

            foreach (var controller in controllers)
            {
                // ЫҢШ§ЩҒШӘЩҶ Ъ©Щ„ЫҢЩҮ Ш§Ъ©ШҙЩҶ Щ…ШӘШҜЩҮШ§ЫҢ Ъ©ЩҶШӘШұЩ„Шұ Ш¬Ш§ШұЫҢ
                var actionMethods = controller.GetMethods(BindingFlags.Public | BindingFlags.Instance | BindingFlags.DeclaredOnly)
                                              .Where(method => typeof(ActionResult).IsAssignableFrom(method.ReturnType))
                                              .ToList();

                foreach (var method in actionMethods)
                {                    
                    var httpPostAttributes = method.GetCustomAttributes(typeof(HttpPostAttribute), true);
                    if (httpPostAttributes == null || !httpPostAttributes.Any())
                        continue;

                    var csrfTokens = method.GetCustomAttributes(typeof(ValidateAntiForgeryTokenAttribute), true);
                    if (csrfTokens == null || !csrfTokens.Any())
                    {
                        Console.WriteLine("Detected [HttpPost] without [ValidateAntiForgeryToken] in:\n {0}-->{1}",
                                               controller.FullName, method.Name);
                    }
                }
            }
        }
    }
}
Ш§ШЁШӘШҜШ§ Щ…ШіЫҢШұ Ш§ШіЩ…ШЁЩ„ЫҢ Ъ©Ш§Щ…ЩҫШ§ЫҢЩ„ ШҙШҜЩҮ ЩҫШұЩҲЪҳЩҮ ASP.NET MVC Ъ©ЩҮ ШӯШ§ЩҲЫҢ Ъ©ЩҶШӘШұЩ„ШұЩҮШ§ЫҢ ШЁШұЩҶШ§Щ…ЩҮ Ш§ШіШӘШҢ ШЁШ§ЫҢШҜ Щ…ШҙШ®Шө ЪҜШұШҜШҜ.
ШіЩҫШі ШҜШұ Ш§ЫҢЩҶ Ш§ШіЩ…ШЁЩ„ЫҢШҢ Ъ©Щ„ЫҢЩҮ ЩҶЩҲШ№вҖҢЩҮШ§ЫҢ ШӘШ№ШұЫҢЩҒ ШҙШҜЩҮШҢ ЫҢШ§ЩҒШӘ ЪҜШұШҜЫҢШҜЩҮ ЩҲ ШўЩҶвҖҢЩҮШ§ЫҢЫҢ Ъ©ЩҮ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ Ъ©ЩҶЩҶШҜЩҮ IController ЩҮШіШӘЩҶШҜ (ЫҢШ№ЩҶЫҢ Ъ©Щ„Ш§ШівҖҢЩҮШ§ЫҢ Ъ©ЩҶШӘШұЩ„Шұ ЩҲШ§ЩӮШ№ЫҢ ШЁШұЩҶШ§Щ…ЩҮ)ШҢ Ш¬ШҜШ§ Ш®ЩҲШ§ЩҮЩҶШҜ ШҙШҜ.
ШҜШұ Ш§ШҜШ§Щ…ЩҮ ШҜШұ Ш§ЫҢЩҶ Ъ©ЩҶШӘШұЩ„ШұЩҮШ§ШҢ Щ…ШӘШҜЩҮШ§ЫҢЫҢ ШұШ§ ШЁШұШұШіЫҢ Ш®ЩҲШ§ЩҮЫҢЩ… Ъ©ШұШҜ Ъ©ЩҮ ШҜШ§ШұШ§ЫҢ Ш®ШұЩҲШ¬ЫҢ Ш§ШІ ЩҶЩҲШ№ ActionResult ШЁШ§ШҙЩҶШҜ (ЩҒЩӮШ· Ш§Ъ©ШҙЩҶ Щ…ШӘШҜЩҮШ§ Щ…ШҜЩҶШёШұ ЩҮШіШӘЩҶШҜ). Ш§ЪҜШұ Ш§ЫҢЩҶ Ш§Ъ©ШҙЩҶ Щ…ШӘШҜ ЫҢШ§ЩҒШӘ ШҙШҜЩҮ ШҜШ§ШұШ§ЫҢ ЩҲЫҢЪҳЪҜЫҢ HttpPost ШЁЩҲШҜ ЩҲ ЩҮЩ…ЪҶЩҶЫҢЩҶ ValidateAntiForgeryToken ЩҶШҜШ§ШҙШӘШҢ ЫҢШ№ЩҶЫҢ ЫҢЪ© Щ…ШҙЪ©Щ„ Ш§Щ…ЩҶЫҢШӘЫҢ Ъ©ЩҮ ШЁШ§ЫҢШҜ ШЁШұШ·ШұЩҒ ШҙЩҲШҜ.
