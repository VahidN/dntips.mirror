ЩҮШұ Ъ©Щ„Ш§ШіЫҢ ШҜШұ #C Ъ©ЩҮ Ш§ШІ Ъ©Щ„Ш§Ші ЩҫШ§ЫҢЩҮвҖҢЫҢ System.Attribute Щ…ШҙШӘЩӮ ШҙЩҲШҜШҢ ЫҢЪ© Attribute ЩҶШ§Щ…ЫҢШҜЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ ЩҲ Щ…ЩҮЩ…вҖҢШӘШұЫҢЩҶ ЩҲ ЩҮШҜЩҒ ЩҲ Ъ©Ш§ШұШЁШұШҜ ШўЩҶвҖҢЩҮШ§ШҢ Щ…ШІЫҢЩҶ Ъ©ШұШҜЩҶ ЩҲ Ш№Щ„Ш§Щ…ШӘЪҜШ°Ш§ШұЫҢ ШіШ§ЫҢШұ ЩҶЩҲШ№вҖҢЩҮШ§ ЩҲ ЩҒЫҢЩ„ШҜЩҮШ§ ЩҮШіШӘЩҶШҜ ШӘШ§ ШЁШұ Ш§ШіШ§Ші ШўЩҶвҖҢЩҮШ§ ШЁШӘЩҲШ§ЩҶ Ъ©Ш§ШұЪ©ШұШҜЩҮШ§ЫҢ ШЁЫҢШҙШӘШұЫҢ ШұШ§ ШҜШұ Ш§Ш®ШӘЫҢШ§Шұ ШўЩҶ ЩҶЩҲШ№вҖҢЩҮШ§ ЩӮШұШ§Шұ ШҜШ§ШҜ. ШЁШұШ§ЫҢ Щ…Ш«Ш§Щ„ШҢ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ  ЩҲЫҢЪҳЪҜЫҢвҖҢвҖҢЩҮШ§ЫҢ JsonProperty ЩҲ ЫҢШ§ JsonPropertyName ШҜШұ ШӯЫҢЩҶ Ш§Ш№Щ…Ш§Щ„ serializations ЩҲ ЫҢШ§ ШҜШұ Ъ©Ш§ШұШЁШұШҜЩҮШ§ЫҢ Ш§Ш№ШӘШЁШ§ШұШіЩҶШ¬ЫҢ Щ…Ш§ЩҶЩҶШҜ ЩҲЫҢЪҳЪҜЫҢвҖҢЩҮШ§ЫҢ RequiredШҢ Range ЩҲ Ш§Щ…Ш«Ш§Щ„ ШўЩҶвҖҢЩҮШ§:
public class Student
{
    [JsonPropertyName("id")]
    public int Id { get; set; }

    [JsonPropertyName("name")]
    public string Name { get; set; }
}

public class WeatherForecast
{
    [Required]
    public int TemperatureC { get; set; }

    [MinLength(50)]
    public string Summary { get; set; }
}

ШұЩҲШҙ Щ…ШӘШҜШ§ЩҲЩ„ Ш§ШұШіШ§Щ„ ЩҶЩҲШ№вҖҢЩҮШ§ ШЁЩҮ attributes ШӘШ§ ЩҫЫҢШҙ Ш§ШІ C# 11

ШӘШ§ ЩҫЫҢШҙ Ш§ШІ C# 11ШҢ ШұЩҲШҙ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ ЫҢЪ© attribute Ш¬ЩҶШұЫҢЪ© Ъ©ЩҮ ШЁШӘЩҲШ§ЩҶШҜ ШЁШ§ Ш§ЩҶЩҲШ§Ш№ ЩҲ Ш§ЩӮШіШ§Щ… ЩҶЩҲШ№вҖҢЩҮШ§ Ъ©Ш§Шұ Ъ©ЩҶШҜШҢ ШЁЩҮ ШөЩҲШұШӘ ШІЫҢШұ Ш§ШіШӘ:
- Ш§ШұШіШ§Щ„ ЫҢЪ© ЩҫШ§ШұШ§Щ…ШӘШұ Ш§ШІ ЩҶЩҲШ№ System.Type ШЁЩҮ ШіШ§ШІЩҶШҜЩҮвҖҢЫҢ attribute
- ШӘШ№ШұЫҢЩҒ Ш®Ш§ШөЫҢШӘЫҢ Щ…Ш§ЩҶЩҶШҜ ParamType ШҜШұ ШөЩҲШұШӘ ЩҶЫҢШ§ШІШӣ ШӘШ§ Щ…ШҙШ®Шө Ъ©ЩҶШҜ Ъ©ЩҮ ЪҶЩҮ ЩҶЩҲШ№ЫҢ ШЁЩҮ ШіШ§ШІЩҶШҜЩҮвҖҢЫҢ attribute Ш§ШұШіШ§Щ„ ШҙШҜЩҮвҖҢШ§ШіШӘ. Щ…Ш§ЩҶЩҶШҜ Щ…Ш«Ш§Щ„ ЩҒШұШ¶ЫҢ ШІЫҢШұ:
[AttributeUsage(AttributeTargets.Class)]
public class CustomDoNothingAttribute: Attribute
{
    // Note the type parameter in the constructor
    public CustomDoNothingAttribute(Type t)
    {
        ParamType = t;
    }

    public Type ParamType { get; }
}
ЩҲ ШіЩҫШі ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Ш№Щ…Щ„ЪҜШұ typeofШҢ ЩҶЩҲШ№ Щ…ШҜЩҶШёШұ ШұШ§ ШЁЩҮ ШіШ§ШІЩҶШҜЩҮвҖҢЫҢ ЩҲЫҢЪҳЪҜЫҢ ШӘШ№ШұЫҢЩҒ ШҙШҜЩҮШҢ Ш§ШұШіШ§Щ„ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ…:
[CustomDoNothing(typeof(string))]
public class Student
{
    public int Id { get; set; }

    public string Name { get; set; }
}
ЫҢЪ© ЩҶЩ…ЩҲЩҶЩҮ Щ…Ш«Ш§Щ„ ШҜЩҶЫҢШ§ЫҢ ЩҲШ§ЩӮШ№ЫҢ ШўЩҶШҢ [ServiceFilter(typeof(ResponseLoggerFilter))] ШҜШұ ASP.NET Core Ш§ШіШӘ Ъ©ЩҮ ШҜЫҢЪҜШұ ШЁШ§ ЩҲШ¬ЩҲШҜ Ш¬ЩҶШұЫҢЪ©вҖҢЩҮШ§ШҢ ШўЩҶЪҶЩҶШ§ЩҶ ЩҮЩ…Ш§ЩҮЩҶЪҜ ЩҲ ЫҢЪ©вҖҢШҜШіШӘ ШЁШ§ ШіШ§ЫҢШұ Ш§Ш¬ШІШ§ЫҢ ШІШЁШ§ЩҶ ШЁЩҮ ЩҶШёШұ ЩҶЩ…ЫҢвҖҢШұШіШҜ. ЩҶЩ…ЩҲЩҶЩҮвҖҢЫҢ ЩҮЩ…Ш§ЩҮЩҶЪҜ ШЁШ§ ЩҫЫҢШҙШұЩҒШӘвҖҢЩҮШ§ЫҢ ШІШЁШ§ЩҶШҢ ШЁШ§ЫҢШҜ ЪҶЩҶЫҢЩҶ ШҙЪ©Щ„ЫҢ ШұШ§ ШҜШ§ШҙШӘЩҮ ШЁШ§ШҙШҜ: [<ServiceFilter<ResponseLoggerFilter]


Ш§Щ…Ъ©Ш§ЩҶ ШӘШ№ШұЫҢЩҒ ЩҲЫҢЪҳЪҜЫҢвҖҢЩҮШ§ЫҢ Ш¬ЩҶШұЫҢЪ© ШҜШұ C# 11

вҖҚC# 11 ШЁЩҮ ЩҮЩ…ШұШ§ЩҮ ЩҫЫҢШҙШӘЫҢШЁШ§ЩҶЫҢ Ш§ШІ generic attributes Ш§ШұШ§ШҰЩҮ ШҙШҜЩҮвҖҢШ§ШіШӘ. ШЁЩҶШ§ШЁШұШ§ЫҢЩҶ Ш§ЫҢЩҶШЁШ§Шұ ШЁШ¬Ш§ЫҢ Ш§ШұШіШ§Щ„ ЩҫШ§ШұЩ…ШӘШұЫҢ Ш§ШІ ЩҶЩҲШ№ Type ШЁЩҮ ШіШ§ШІЩҶШҜЩҮвҖҢЫҢ ЩҲЫҢЪҳЪҜЫҢвҖҢШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ Ъ©Щ„Ш§Ші ШўЩҶ attribute ШұШ§ ШЁЩҮ ШөЩҲШұШӘ Ш¬ЩҶШұЫҢЪ© ШӘШ№ШұЫҢЩҒ Ъ©ЩҶЫҢЩ… Ъ©ЩҮ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ ЫҢЪ© ЫҢШ§ ЪҶЩҶШҜЫҢЩҶ ЩҶЩҲШ№ ШұШ§ ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ ЩҫШ§ШұШ§Щ…ШӘШұ ШЁЩҫШ°ЫҢШұШҜ. ШЁЩҶШ§ШЁШұШ§ЫҢЩҶ Щ…Ш«Ш§Щ„ ЩӮШЁЩ„ ШҜШұ C# 11 ШЁЩҮ ШөЩҲШұШӘ ШІЫҢШұ ШЁШ§ШІЩҶЩҲЫҢШіЫҢ Щ…ЫҢвҖҢШҙЩҲШҜ:
[AttributeUsage(AttributeTargets.Class)]
public class CustomDoNothingAttribute<T> : Attribute
    where T : class
{
    public T ParamType { get; }
}

[CustomDoNothing<string>]
public class Student
{
    public int Id { get; set; }

    public string Name { get; set; }
}
ЫҢЪ© Щ…ШІЫҢШӘ Щ…ЩҮЩ… Ш§ЫҢЩҶ ШұЩҲШҙ ЩҶШіШЁШӘ ШЁЩҮ ЩӮШЁЩ„ШҢ Ш§Щ…Ъ©Ш§ЩҶ ШӘШ№ШұЫҢЩҒ ЩӮЫҢЩҲШҜ ЩҲ type safety Ш§ШіШӘ. ШЁШұШ§ЫҢ ЩҶЩ…ЩҲЩҶЩҮ ШҜШұ Щ…Ш«Ш§Щ„ ЩҒЩҲЩӮШҢ ЩҶЩҲШ№ T ШЁЩҮ Ъ©Щ„Ш§ШівҖҢЩҮШ§ Щ…ШӯШҜЩҲШҜ ШҙШҜЩҮвҖҢШ§ШіШӘ ЩҲ ЩҶЩҲШ№вҖҢЩҮШ§ЫҢ ШҜЫҢЪҜШұ ШұШ§ ЩҶЩ…ЫҢвҖҢЩҫШ°ЫҢШұШҜ. ШЁЩҮ Ш§ЫҢЩҶ ШӘШұШӘЫҢШЁ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ Ш§ЫҢЩҶ ЩҶЩҲШ№ ШЁШұШұШіЫҢвҖҢЩҮШ§ ШұШ§ ШЁШ¬Ш§ЫҢ ШІЩ…Ш§ЩҶ Ш§Ш¬ШұШ§ ЩҲ ШөШҜЩҲШұ Ш§ШіШӘШ«ЩҶШ§ШЎЩҮШ§ШҢ ШҜЩӮЫҢЩӮШ§ ШҜШұ ШІЩ…Ш§ЩҶ Ъ©Ш§Щ…ЩҫШ§ЫҢЩ„ Ш§ЩҶШ¬Ш§Щ… ШҜШ§ШҜ.
ЩҲ Ш§ЪҜШұ ЩҶЫҢШ§ШІ ШЁЩҮ ШӘШ№ЫҢЫҢЩҶ ЪҶЩҶШҜ ЩҶЩҲШ№ ШЁЩҲШҜШҢ ШЁШ§ЫҢШҜ Ш®Ш§ШөЫҢШӘ AllowMultiple ЩҶШӯЩҲЩҮвҖҢЫҢ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ЩҲЫҢЪҳЪҜЫҢ ШұШ§ ШЁЩҮ true ШӘЩҶШёЫҢЩ… Ъ©ШұШҜ:
[AttributeUsage(AttributeTargets.Class, AllowMultiple = true)]
public class DecorateAttribute<T> : Attribute where T : class
{
    // ....
}
ШӘШ§ ШЁШӘЩҲШ§ЩҶ ШЁЩҮ ШӘШ№ШұЫҢЩҒ ШІЫҢШұ ШұШіЫҢШҜ:
[Decorate<LoggerDecorator>]
[Decorate<TimerDecorator>]
public class SimpleWorker
{
    // ....
}


Щ…ШӯШҜЩҲШҜЫҢШӘвҖҢЩҮШ§ЫҢ Ш§ЩҶШӘШ®Ш§ШЁ ЩҶЩҲШ№вҖҢЩҮШ§ ШҜШұ ЩҲЫҢЪҳЪҜЫҢвҖҢЩҮШ§ЫҢ Ш¬ЩҶШұЫҢЪ© C# 11

ШҜШұ ЩҲЫҢЪҳЪҜЫҢвҖҢЩҮШ§ЫҢ Ш¬ЩҶШұЫҢЪ© ЩҶЩ…ЫҢвҖҢШӘЩҲШ§ЩҶ Ш§ШІ ЩҶЩҲШ№вҖҢЩҮШ§ЫҢ ШІЫҢШұ Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ШұШҜ (ЩҮЩ…Ш§ЩҶ Щ…ШӯШҜЩҲШҜЫҢШӘвҖҢЩҮШ§ЫҢ typeofШҢ ШҜШұ Ш§ЫҢЩҶШ¬Ш§ ЩҮЩ… ШЁШұЩӮШұШ§Шұ ЩҮШіШӘЩҶШҜ):
- ЩҶЩҲШ№вҖҢЩҮШ§ЫҢ dynamic
- nullable reference types Щ…Ш§ЩҶЩҶШҜ ?string
- ЩҶЩҲШ№вҖҢЩҮШ§ЫҢ tuple ШӘШ№ШұЫҢЩҒ ШҙШҜЩҮвҖҢЫҢ ШЁЩҮ Ъ©Щ…Ъ© C# tuple syntax Щ…Ш§ЩҶЩҶШҜ (int x, int y)

ЪҶЩҲЩҶ Ш§ЫҢЩҶ ЩҶЩҲШ№вҖҢЩҮШ§ ШЁЩҮ ЩҮЩ…ШұШ§ЩҮ ЫҢЪ©ШіШұЫҢ metadata annotations ЩҮШіШӘЩҶШҜ Ъ©ЩҮ ШөШұЩҒШ§ ШЁЫҢШ§ЩҶЪҜШұ ШӘЩҲШ¶ЫҢШӯЫҢ Ш§Ш¶Ш§ЩҒЫҢ ШҜШұ Щ…ЩҲШұШҜ ЩҶЩҲШ№ ШЁЪ©Ш§ШұЪҜШұЩҒШӘЩҮ ШҙШҜЩҮ ЩҮШіШӘЩҶШҜ ЩҲ ШҜШұ ШөЩҲШұШӘ ЩҶЫҢШ§ШІШҢ ШЁШ¬Ш§ЫҢ ШўЩҶвҖҢЩҮШ§ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЫҢШҜ Ш§ШІ ЩҶЩҲШ№вҖҢЩҮШ§ЫҢ ШІЫҢШұ Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ЩҶЫҢШҜ:
- Ш§ШІ object ШЁШ¬Ш§ЫҢ dynamic
- Ш§ШІ string ШЁШ¬Ш§ЫҢ ?string
- Ш§ШІ <ValueTuple<int, int ШЁШ¬Ш§ЫҢ (int X, int Y)

ЩҮЩ…ЪҶЩҶЫҢЩҶ ШҜШұ ШІЩ…Ш§ЩҶ Ш§ШіШӘЩҒШ§ШҜЩҮвҖҢЫҢ Ш§ШІ ЫҢЪ© ЩҲЫҢЪҳЪҜЫҢ Ш¬ЩҶШұЫҢЪ©ШҢ ШЁШ§ЫҢШҜ ЩҶЩҲШ№ Щ…ЩҲШұШҜ Ш§ШіШӘЩҒШ§ШҜЩҮШҢ Ъ©Ш§Щ…Щ„Ш§ Щ…ШҙШ®Шө ЩҲ ШҜШұ Ш§ШөШ·Щ„Ш§Шӯ fully constructed ШЁШ§ШҙШҜ:
public class GenericAttribute<T> : Attribute { }

public class GenericType<T>
{
    [GenericAttribute<T>] // Not allowed! generic attributes must be fully constructed types.
    public string Method1() => default;

    [GenericAttribute<string>]
    public string Method2() => default;
}

    ШЁШ§ШІШ®ЩҲШұШҜЩҮШ§ЫҢ ШҜЩҲШұЩҮШ§ЩҶШӘЩӮШ§Щ„ Ш®ЩҲШҜЪ©Ш§Шұ Data Annotations Ш§ШІ Щ…ШҜЩ„вҖҢЩҮШ§ ШЁЩҮ ViewModelЩҮШ§ЫҢ ASP.NET MVC ШЁЩҮ Ъ©Щ…Ъ© AutoMapperЩ…Ш·Ш§Щ„ШЁЩҶЪҜШ§ШҙШӘ Ш§ШҙЫҢШ§ШЎ ШҜШұ AutoMapper ШӘЩҲШіШ· Attribute ЩҮШ§ #1Щ…Ш·Ш§Щ„ШЁШ§ЫҢШ¬Ш§ШҜ Щ„ЫҢШіШӘЫҢ Ш§ШІ Ъ©Щ„Ш§ШіЫҢ Ш¬ЩҶШұЫҢЪ©Щ…Ш·Ш§Щ„ШЁЪ©Ш§Щ…ЩҫЩҲЩҶЩҶШӘвҖҢЩҮШ§ЫҢ Ш¬ЩҶШұЫҢЪ© ШҜШұ BlazorЩ…Ш·Ш§Щ„ШЁC# 7 - Tuple return types and deconstructionЩ…Ш·Ш§Щ„ШЁЩҶЪҜШ§ШҙШӘ Ш§ШҙЫҢШ§ШЎ ШҜШұ AutoMapper ШӘЩҲШіШ· Attribute ЩҮШ§ #2 - ШӘШЁШҜЫҢЩ„ ЩҲЫҢЪҳЪҜЫҢвҖҢЩҮШ§ ШЁЩҮ ЩҶЪҜШ§ШҙШӘЩ…Ш·Ш§Щ„ШЁШўШҙЩҶШ§ЫҢЫҢ ШЁШ§ Ш¬ЩҶШұЫҢЪ©вҖҢЩҮШ§ #3Щ…Ш·Ш§Щ„ШЁШіШ§Ш®ШӘ Attribute ЩҮШ§ЫҢ ШҜЩ„Ш®ЩҲШ§ЩҮ ЫҢШ§ Ш®ШөЩҲШөЫҢ ШіШ§ШІЫҢ ШҙШҜЩҮЩ…Ш·Ш§Щ„ШЁШ§ЩҶШ¬Ш§Щ… Ш§Ш№Щ…Ш§Щ„ ШұЫҢШ§Ш¶ЫҢ ШЁШұ ШұЩҲЫҢ GenericsЩ…Ш·Ш§Щ„ШЁШ§Щ…Ъ©Ш§ЩҶ ШӘШ№ШұЫҢЩҒ Ш§Ш№Ш¶Ш§ЫҢ static abstract ШҜШұ Ш§ЫҢЩҶШӘШұЩҒЫҢШівҖҢЩҮШ§ЫҢ C# 11

ЩҶШёШұШ§ШӘ

  • ЩҲШӯЫҢШҜ ЩҶШөЫҢШұЫҢ ШҜШұ ЫұЫҙЫ°Ыұ/Ы°Ы№/ЫұЫё Ы·:ЫөЫұ
    ШұЩҲШҙ ШӘШЁШҜЫҢЩ„ ServiceFilterAttribute ШҜШұ ASP.NET Core ШЁЩҮ ЫҢЪ© ЩҶЩ…ЩҲЩҶЩҮвҖҢЫҢ Type-Safe

    Ш§ЫҢЩҶ ЩҲЫҢЪҳЪҜЫҢ ШҜШұ Ш§ШөЩ„ ШЁЩҮ ШөЩҲШұШӘ ШІЫҢШұ ШӘШ№ШұЫҢЩҒ ШҙШҜЩҮвҖҢШ§ШіШӘ: 
    /// <summary>

     ServiceType = type ?? throw new ArgumentNullException(nameof(type));

    ЩҲ ШЁЩҮ ЩҮЩ…ЫҢЩҶ Ш¬ЩҮШӘ ШҜШұ ШІЩ…Ш§ЩҶ Ъ©Ш§Щ…ЩҫШ§ЫҢЩ„ШҢ Щ…ШӯШҜЩҲШҜЫҢШӘЫҢ ШҜШұ Щ…ЩҲШұШҜ ЩҶЩҲШ№ Ш§ШұШіШ§Щ„ЫҢ ШЁЩҮ ШўЩҶ ЩҲШ¬ЩҲШҜ ЩҶШҜШ§ШұШҜ. Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ Ш§ЫҢЩҶ Щ…ШҙЪ©Щ„ ШұШ§ ШҜШұ C# 11 ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ЩӮЫҢЩҲШҜ Ш¬ЩҶШұЫҢЪ©ШҢ ШЁЩҮ ЩҶШӯЩҲ ШІЫҢШұ ШЁШұШ·ШұЩҒ Ъ©ШұШҜ:

    • ЩҲШӯЫҢШҜ ЩҶШөЫҢШұЫҢ ШҜШұ ЫұЫҙЫ°ЫІ/Ы°Ыі/ЫіЫ° Ы·:ЫІЫІ
      ЫҢЪ© ЩҶЪ©ШӘЩҮвҖҢЫҢ ШӘЪ©Щ…ЫҢЩ„ЫҢ: ШЁЪ©Ш§ШұЪҜЫҢШұЫҢ ЪҜШіШӘШұШҜЩҮвҖҢЫҢ Generic attributes ШҜШұ ASP.NET Core 8x

      ЩҲЫҢЪҳЪҜЫҢ Generic attributes ШЁЩҮ C# 11 Ш§Ш¶Ш§ЩҒЩҮ ШҙШҜШҢ Ш§Щ…Ш§ ШҜШұ ШўЩҶ ШІЩ…Ш§ЩҶШҢ ШӘШәЫҢЫҢШұЫҢ ШҜШұ ШӘШ№Ш§ШұЫҢЩҒ ЩҲЫҢЪҳЪҜЫҢвҖҢЩҮШ§ЫҢ Щ…ШЁШӘЩҶЫҢ ШЁШұ System.Type Щ…ЩҲШ¬ЩҲШҜ ШҜШұ ASP.NET Core 7x ШұШ® ЩҶШҜШ§ШҜ.
