ШЁШұШ§Щү Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Ъ©Щ„Ш§ШівҖҸЩҮШ§Щү Entity Ъ©ЩҮ ШҜШұ ЩҶЩҲШҙШӘШ§Шұ ЩҫЫҢШҙЫҢЩҶ Ш§ЫҢШ¬Ш§ШҜ Ъ©ШұШҜЫҢЩ… ШҜШұ WCF ШЁШ§ЫҢШҜ ШўЩҶ Ъ©Щ„Ш§ШівҖҺЩҮШ§ ШұШ§ ШҜШіШӘвҖҺЪ©Ш§ШұЩү Ъ©ЩҶЫҢЩ…. Щ…ШӘЩҶ Ъ©Щ„Ш§Ші tblNews ШұШ§ ШҜШұ ЩҶШёШұ ШЁЪҜЫҢШұЫҢШҜ:
namespace MyNewsWCFLibrary
{
    using System;
    using System.Collections.Generic;
    
    public partial class tblNews
    {
        public int tblNewsId { get; set; }

        public string Description { get; set; }

    }

 Щ…ШҙШ§ЩҮШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶЫҢШҜ Ъ©ЩҮ ШЁШұШ§ЫҢ ШӘШ№ШұЫҢЩҒ Ъ©Щ„Ш§ШівҖҢЩҮШ§ Ш§ШІ Ъ©Щ„Щ…ЩҮ Ъ©Щ„ЫҢШҜЫҢ partial Ш§ШіШӘЩҒШ§ШҜЩҮ ШҙШҜЩҮ Ш§ШіШӘ.  Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Ъ©Щ„Щ…ЩҮ Ъ©Щ„ЫҢШҜЫҢ partial ШЁЩҮ ШҙЩ…Ш§ Ш§Ш¬Ш§ШІЩҮ Щ…ЫҢвҖҢШҜЩҮШҜ Ъ©ЩҮ ЫҢЪ© Ъ©Щ„Ш§Ші ШұШ§ ШҜШұ ЪҶЩҶШҜЫҢЩҶ ЩҒШ§ЫҢЩ„ Ш¬ШҜШ§ЪҜШ§ЩҶЩҮ ШӘШ№ШұЫҢЩҒ Ъ©ЩҶЫҢШҜ. ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ Щ…Ш«Ш§Щ„ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЫҢШҜ ЩҒЫҢЩ„ШҜЩҮШ§ШҢ ЩҲЫҢЪҳЪҜЩү ЩҮШ§ ЩҲ ШіШ§ШІЩҶШҜЩҮвҖҢЩҮШ§ ШұШ§ ШҜШұ ЫҢЪ© ЩҒШ§ЫҢЩ„ ЩҲ Щ…ШӘШҜЩҮШ§ ШұШ§ ШҜШұ ЩҒШ§ЫҢЩ„ ШҜЫҢЪҜШұ ЩӮШұШ§Шұ ШҜЩҮЫҢШҜ. 
ШЁЩҮ ШөЩҲШұШӘ Ш®ЩҲШҜЪ©Ш§Шұ Ъ©Щ„ЫҢЩҮвҖҢЩү ЩҲЫҢЪҳЪҜЩүвҖҢЩҮШ§ ШЁЩҮ ШӘЩҲШ¬ЩҮ ШЁЩҮ ЩҫШ§ЫҢЪҜШ§ЩҮ ШҜШ§ШҜЩҮ ШіШ§Ш®ШӘЩҮ ШҙШҜЩҮ Ш§ЩҶШҜ. ШЁШұШ§Щү ЩҶЩ…ЩҲЩҶЩҮ Щ…Ш§ ШЁШұШ§Щү ЩҒЫҢЩ„ШҜ IsDeleted ШҜШұ SQL Server ШіШӘЩҲЩҶ Allow Nulls ШұШ§ Ъ©Щ„ЫҢЪ© Ъ©ШұШҜЩҮ ШЁЩҲШҜЫҢЩ… Ъ©ЩҮ ШҜШұ ЩҶШӘЫҢШ¬ЩҮ ШҜШұ Ш§ЫҢЩҶШ¬Ш§ Ш№ШЁШ§ШұШӘ Nullable ЩҫЫҢШҙ Ш§ШІ ЩҶЩҲШ№ ЩҒЫҢЩ„ШҜ ЩҶШҙШ§ЩҶ ШҜШ§ШҜЩҮ ШҙШҜЩҮ Ш§ШіШӘ. ШЁШұШ§Щү Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Ш§ЫҢЩҶ Ъ©Щ„Ш§Ші ШҜШұ WCF ШЁШ§ЫҢШҜ ШөЩҒШӘ  DataContract ШұШ§ ШЁЩҮ Ъ©Щ„Ш§Ші ШҜШ§ШҜ. Ш§ЫҢЩҶ ЩӮШұШ§ШұШҜШ§ШҜ ШЁЩҮ Щ…Ш§ Ш§Ш¬Ш§ШІЩҮ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ЩҲЫҢЪҳЪҜЩүвҖҢЩҮШ§ЫҢЩү Ъ©ЩҮ ШөЩҒШӘ DataMember ШұШ§ Щ…ЩүвҖҢЪҜЫҢШұЩҶШҜ ШұШ§ Щ…ЩүвҖҢШҜЩҮШҜ.
Ъ©Щ„Ш§Ші ШЁШ§Щ„Ш§ ШұШ§ ШЁЩҮ ШҙЪ©Щ„ ШІЫҢШұ ШЁШ§ШІЩҶЩҲЫҢШіЩү Ъ©ЩҶЫҢШҜ:
using System.Runtime.Serialization;

namespace MyNewsWCFLibrary
{
    using System;
    using System.Collections.Generic;
    
    [DataContract]
    public partial class tblNews
    {
        [DataMember]
        public int tblNewsId { get; set; }
        [DataMember]
        public int tblCategoryId { get; set; }
        [DataMember]
        public string Title { get; set; }
        [DataMember]
        public string Description { get; set; }
        [DataMember]
        public System.DateTime RegDate { get; set; }
        [DataMember]
        public Nullable<bool> IsDeleted { get; set; }

        public virtual tblCategory tblCategory { get; set; }
    }
}
ЩҮЩ…вҖҢЪҶЩҶЫҢЩҶ Ъ©Щ„Ш§Ші tblCategory ШұШ§ ШЁЩҮ ШөЩҲШұШӘ ШІЫҢШұ ШӘШәЫҢЫҢШұ ШҜЩҮЫҢШҜ:
namespace MyNewsWCFLibrary

ЩҲЩ„ЫҢ ЩҒШұШ¶ Ъ©ЩҶЫҢШҜ ШҜШұ Ш§ЫҢЩҶШ¬Ш§ ШӘШөЩ…ЫҢЩ… ШЁЩҮ ШӘШәЫҢЫҢШұЫҢ ШҜШұ ЩҫШ§ЫҢЪҜШ§ЩҮ ШҜШ§ШҜЩҮ Щ…ЫҢвҖҢЪҜЫҢШұЫҢШҜ. ЩҫШі Ш§ШІ ШўЩҶ Щ…ШҜЩ„ Entity Framework ШұШ§ ШЁШ§ШІ Ъ©ЩҶЫҢШҜ ЩҲ ШҜШұ Ш¬Ш§ЫҢЫҢ Ш§ШІ ШөЩҒШӯЩҮ ШұШ§ШіШӘвҖҢЪ©Щ„ЫҢЪ© Ъ©ШұШҜЩҮ ЩҲ Ш§ШІ Щ…ЩҶЩҲЫҢ ШЁШ§ШІШҙШҜЩҮ ЪҜШІЫҢЩҶЩҮ Update Model from Database ШұШ§ Ш§ЩҶШӘШ®Ш§ШЁ Ъ©ЩҶЫҢШҜ. ШіЩҫШі ШҜШұ ЩҫЩҶШ¬ШұЩҮ ШЁШ§ШІШҙШҜЩҮШҢ ЪҶЩҲЩҶ ЩҮЫҢЪҶ Ш¬ШҜЩҲЩ„ШҢ ЩҶЩ…Ш§ ЫҢШ§ ШұЩҲШ§Щ„ЫҢ ШЁЩҮ ЩҫШ§ЫҢЪҜШ§ЩҮ ШҜШ§ШҜЩҮвҖҢЩҮШ§ ЩҶЫҢЩҒШІЩҲШҜЩҮ Ш§ЫҢЩ…Шӣ ШҜЪҜЩ…ЩҮвҖҢЫҢ Finish ШұШ§ Ъ©Щ„ЫҢЪ© Ъ©ЩҶЫҢШҜ. ШҜЩҲШЁШ§ШұЩҮ Ъ©Щ„Ш§Ші tblNews ШұШ§ ШЁШ§ШІЪ©ЩҶЫҢШҜ. Щ…ШӘЩҲШ¬ЩҮ Ш®ЩҲШ§ЩҮЫҢШҜ ШҙШҜ Ъ©ЩҮ ЩҮЩ…ЩҮвҖҢЫҢ DataContractЩҮШ§ ЩҲ DataMemberЩҮШ§ ШұШ§ ШӯШ°ЩҒ ШҙШҜЩҮ Ш§ШіШӘ. Щ…Щ…Ъ©ЩҶ Ш§ШіШӘ ШЁЪҜЩҲЫҢЫҢШҜ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШіШӘЫҢЩ… Ъ©Щ„Ш§Ші ЫҢШ§ Щ…ШҜЩ„ ШұШ§ ШӘШәЫҢЫҢШұ ШҜЩҮЫҢЩ… ЩҲ ШЁЩҮ ЩҲШіЫҢЩ„ЩҮвҖҢЫҢ Generate Database from Model ШЁЩҮвҖҢЩҮЩҶЪҜШ§Щ… Ъ©ЩҶЫҢЩ…. ЩҲЩ„ЫҢ ШҜШұ ЩҶШёШұ ШЁЪҜЫҢШұЫҢШҜ Ъ©ЩҮ ЩҶЫҢШ§ШІ ШЁЩҮ Ш§ЫҢШ¬Ш§ШҜ ЪҶЩҶШҜЫҢЩҶ Ш¬ШҜЩҲЩ„ ШҜЫҢЪҜШұ ШҜШ§ШұЫҢЩ… ЩҲ Щ…ШҜЩ„ЫҢ ШЁШ§ ШҜЩҮвҖҢЩҮШ§ Entity ШҜШ§ШұЫҢШҜ. ШҜШұ Ш§ЫҢЩҶ ШөЩҲШұШӘ ЩҮЩ…ЩҮвҖҢЫҢ ШӘШәЫҢЫҢШұШ§ШӘЫҢ Ъ©ЩҮ ШҜШұ Ъ©Щ„Ш§Ші ШҜШ§ШҜЩҮ Ш§ЫҢЩ… ШІШҜЩҲШҜЩҮ Ш®ЩҲШ§ЩҮШҜ ШҙШҜ. 
ШҜШұ ШЁШ®Шҙ ЩҫШіЫҢЩҶШҢ ШҜШұШЁШ§ШұЩҮвҖҢЫҢ Ш§ЫҢЩҶвҖҢЪ©ЩҮ ЪҶЩҮ Ъ©ЩҶЫҢЩ… Ъ©ЩҮ Ш№ШЁШ§ШұШӘвҖҢЩҮШ§ЫҢЫҢ Ъ©ЩҮ ШЁЩҮ Ъ©Щ„Ш§ШівҖҢЩҮШ§ Щ…ЫҢвҖҢШ§ЩҒШІШ§ЫҢЫҢЩ… ШӯШ°ЩҒ ЩҶШҙЩҲШҜШӣ Ш®ЩҲШ§ЩҮЩ… ЩҶЩҲШҙШӘ.
