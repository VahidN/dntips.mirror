Managed Extensibility Framework ЫҢШ§ MEF Ъ©Ш§Щ…ЩҫЩҲЩҶЩҶШӘЫҢ Ш§ШІ Framework 4 Ш§ШіШӘ Ъ©ЩҮ ШЁШұШ§ЫҢ Ш§ЫҢШ¬Ш§ШҜ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҮШ§ЫҢ ШӘЩҲШіШ№ЩҮ ЩҫШ°ЫҢШұ (Extensible) ШЁШ§ ШӯШ¬Щ… Ъ©Щ… Ъ©ШҜ Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢШҙЩҮ.Ш§ЫҢЩҶ ШӘЪ©ЩҶЩҲЩ„ЩҲЪҳЫҢ ШЁЩҮ ШЁШұЩҶШ§Щ…ЩҮ ЩҶЩҲЫҢШіШ§ЩҶ Ш§ЫҢЩҶ Ш§Щ…Ъ©Ш§ЩҶ ШұЩҲ Щ…ЫҢШҜЩҮ Ъ©ЩҮ ШӘЩҲШіШ№ЩҮвҖҢЩҮШ§ЫҢ (Extension) ШЁШұЩҶШ§Щ…ЩҮ ШұЩҲ ШЁШҜЩҲЩҶ ЩҫЫҢЪ©ШұШЁЩҶШҜЫҢ Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ЩҶЩҶШҜ. ЩҮЩ…ЪҶЩҶЫҢЩҶ ШЁЩҮ ШӘЩҲШіШ№ЩҮ ШҜЩҮЩҶШҜЪҜШ§ЩҶ Ш§ЫҢЩҶ Ш§Ш¬Ш§ШІЩҮ ШұЩҲ Щ…ЫҢвҖҢШҜЩҮ Ъ©ЩҮ ШЁЩҮ ШўШіШ§ЩҶЫҢ Ъ©ШҜЩҮШ§ ШұЩҲ Ъ©ЩҫШіЩҲЩ„ЩҮ Ъ©ЩҶЩҶШҜ .
MEF ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ ШЁШ®ШҙЫҢ Ш§ШІ 4 NET. ЩҲ Silverlight 4 Щ…Ш№ШұЩҒЫҢ ШҙШҜ. MEF ЫҢЪ© ШұШ§ЩҮ ШӯЩ„ ШіШ§ШҜЩҮ ШЁШұШ§ЫҢ Щ…ШҙЪ©Щ„ ШӘЩҲШіШ№ЩҮ ШҜШұ ШӯШ§Щ„ Ш§Ш¬ШұШ§ЫҢ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҮШ§ Ш§ШұШ§ШҰЩҮ Щ…ЫҢвҖҢЪ©ЩҶШҜ.ШӘШ§ ЩӮШЁЩ„ Ш§ШІ Ш§ЫҢЩҶ ШӘЪ©ЩҶЩҲЩ„ЩҲЪҳЫҢ ШҢ ЩҮШұ ШЁШұЩҶШ§Щ…ЩҮвҖҢШ§ЫҢ Ъ©ЩҮ Щ…ЫҢвҖҢШ®ЩҲШ§ШіШӘ ЫҢЪ© Щ…ШҜЩ„ Plugin ШұШ§ ЩҫШҙШӘЫҢШЁШ§ЩҶЫҢ Ъ©ЩҶЩҮ Щ„Ш§ШІЩ… ШЁЩҲШҜ Ъ©ЩҮ Ш®ЩҲШҜШҙ ШІЫҢШұ ШіШ§Ш®ШӘвҖҢЩҮШ§ ШұШ§ Ш§ШІ Ш§ШЁШӘШҜШ§ Ш§ЫҢШ¬Ш§ШҜ Ъ©ЩҶЩҮ . Ш§ЫҢЩҶ PluginвҖҢЩҮШ§ Ш§ШәЩ„ШЁ ШЁШұШ§ЫҢ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҮШ§ЫҢ Ш®Ш§ШөЫҢ ШЁЩҲШҜЩҶШҜ ЩҲ ЩҶЩ…ЫҢвҖҢШӘЩҲШ§ЩҶШіШӘЩҶШҜ ШҜШұ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢвҖҢЩҮШ§ЫҢ ЪҶЩҶШҜЪҜШ§ЩҶЩҮ ШҜЩҲШЁШ§ШұЩҮ Ш§ШіШӘЩҒШ§ШҜЩҮ ШҙЩҲЩҶШҜ. ЩҲЩ„ЫҢ MEF ШҜШұ ШұШ§ШіШӘШ§ЫҢ ШӯЩ„ Ш§ЫҢЩҶ Щ…ШҙЪ©Щ„Ш§ШӘ ШҢ ШұЩҲШҙ Ш§ШіШӘШ§ЩҶШҜШ§ШұШҜЫҢ ШұЩҲ ШЁШұШ§ЫҢ Щ…ЫҢШІШЁШ§ЩҶЫҢ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҮШ§ЫҢ Ъ©Ш§ШұШЁШұШҜЫҢ ЩҫЫҢШ§ШҜЩҮ Ъ©ШұШҜЩҮ Ш§ШіШӘ. 
ШЁШұШ§ЫҢ ЩҒЩҮЩ… ШЁЩҮШӘШұ Щ…ЩҒШ§ЩҮЫҢЩ… ЫҢЪ© Щ…Ш«Ш§Щ„ ШіШ§ШҜЩҮ ШұЩҲ ШЁШ§ MEF ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ Щ…ЫҢвҖҢЪ©ЩҶЩ….
Ш§ШЁШӘШҜШ§ ЫҢЪ© ЩҫШұЩҲЪҳЩҮ Ш§ШІ ЩҶЩҲШ№ Console Application Ш§ЫҢШ¬Ш§ШҜ Ъ©ЩҶЫҢШҜ . ШЁШ№ШҜ ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Add Reference ЫҢЪ© Ш§ШұШ¬Ш§Ш№ ШЁЩҮ System.ComponentModel.Composition ШЁШҜЫҢШҜ. ШіЩҫШі ЫҢЪ© Interface ШЁЩҮ ЩҶШ§Щ… IViewModel ШұШ§ ШЁЩҮ ШөЩҲШұШӘ ШІЫҢШұ Ш§ЫҢШ¬Ш§ШҜ Ъ©ЩҶЫҢШҜ:
public interface IViewModel
    {
        string Name { get; set; }
    }

ЫҢЪ© Ш®Ш§ШөЫҢШӘ ШЁЩҮ ЩҶШ§Щ… Name ШЁШұШ§ЫҢ ШҜШіШӘШұШіЫҢ ШЁЩҮ ЩҶШ§Щ… ViewModel Ш§ЫҢШ¬Ш§ШҜ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ….
ШіЩҫШі 2 ШӘШ§ ViewModel ШҜЫҢЪҜЩҮ Ш§ЫҢШ¬Ш§ШҜ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ… Ъ©ЩҮ IViewModel ШұШ§ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ Ъ©ЩҶЩҶШҜ. ШЁЩҮ ШөЩҲШұШӘ ШІЫҢШұ:

Export Attribute Ш§ШіШӘЩҒШ§ШҜЩҮ ШҙШҜЩҮ ШҜШұ ШЁШ§Щ„Ш§ЫҢ Ъ©Щ„Ш§ШівҖҢЩҮШ§ЫҢ ViewModel ШЁЩҮ Ш§ЫҢЩҶ Щ…Ш№ЩҶЫҢ Ш§ШіШӘ Ъ©ЩҮ Ш§ЫҢЩҶ Ъ©Щ„Ш§ШівҖҢЩҮШ§ Ш§ЫҢЩҶШӘШұЩҒЫҢШі IViewModel ШұЩҲ Export Ъ©ШұШҜЩҶШҜ ШӘШ§ ШҜШұ Ш¬Ш§ЫҢ Щ…ЩҶШ§ШіШЁ ШЁШӘЩҲЩҶЫҢЩ… Ш§ЫҢЩҶ ViewModel вҖҢЩҮШ§ Import Ъ©ЩҶЫҢЩ….(Import , Export Ш§ШІ Щ…ЩҒШ§ЩҮЫҢЩ… Ш§ШөЩ„ЫҢ ШҜШұ MEF ЩҮШіШӘЩҶШҜ)
ШӯШ§Щ„Ш§ ЩҶЩҲШЁШӘ ШЁЩҮ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ Ъ©Щ„Ш§Ші Plugin Щ…ЫҢвҖҢШұШіЩҮ. 
public class PluginManager
    {
        public PluginManager()
        {

        }

        public IList<IViewModel> ViewModels
        {
            get
            {
                return _viewModels;
            }
            private set
            {
                _viewModels = value;
            }
        }

        [ImportMany( typeof( IViewModel ) )]
        private IList<IViewModel> _viewModels = new List<IViewModel>();

        public void SetupManager()
        {
            AggregateCatalog aggregateCatalog = new AggregateCatalog();

            CompositionContainer container = new CompositionContainer( aggregateCatalog );

            CompositionBatch batch = new CompositionBatch();

            batch.AddPart( this );

            aggregateCatalog.Catalogs.Add( new AssemblyCatalog( Assembly.GetExecutingAssembly() ) );           

            container.Compose( batch );
        }

Ъ©Щ„Ш§Ші PluginManager ШЁШұШ§ЫҢ ШҙЩҶШ§ШіШ§ЫҢЫҢ ЩҲ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Ъ©Щ„Ш§Ші ЩҮШ§ЫҢЫҢ Ъ©ЩҮ ШөЩҒШӘвҖҢЩҮШ§ЫҢ Export ШұЩҲ ШҜШ§ШұЩҶШҜ ЩҶЩҲШҙШӘЩҮ ШҙШҜЩҮ(ШҜЩӮЫҢЩӮШ§ ШҙШЁЫҢЩҮ ЫҢЪ© UnityContainer ШҜШұ Microsoft Unity Application Block ЫҢШ§ IKernel ШҜШұ Ninject) Ш№Щ…Щ„ Щ…ЫҢвҖҢЪ©ЩҶЩҮ ШЁШ§ Ш§ЫҢЩҶ ШӘЩҒШ§ЩҲШӘ Ъ©ЩҮ ЩҶЫҢШ§ШІЫҢ ШЁЩҮ Register ШЁШ§ Bind Ъ©ШұШҜЩҶ ЩҶШҜШ§ШұЩҶШҜ)
Ш§ШЁШӘШҜШ§ ШЁЪ© Щ„ЫҢШіШӘ Ш§ШІ Ъ©Щ„Ш§Ші ЩҮШ§ЫҢЫҢ Ъ©ЩҮ IViewModel ШұЩҲ Export Ъ©ШұШҜЩҶШҜ ШҜШ§ШұЫҢЩ….
ШЁШ№ШҜ ШҜШұ Щ…ШӘШҜ SetupManager Ш§ШЁШӘШҜШ§ ЫҢЪ© AggregateCatalog ЩҶЫҢШ§ШІ ШҜШ§ШұЫҢЩ… ШӘШ§ ШЁШӘЩҲЩҶЫҢЩ… Composition PartвҖҢЩҮШ§ ШұЩҲ ШЁЩҮШҙ Ш§Ш¶Ш§ЩҒЩҮ Ъ©ЩҶЫҢЩ…. ШЁЩҮ Ъ©ШҜ ШІЫҢШұ ШӘЩҲШ¬ЩҮ Ъ©ЩҶЫҢШҜ:
 aggregateCatalog.Catalogs.Add( new AssemblyCatalog( Assembly.GetExecutingAssembly() ) );

ШӘЩҲ Ш§ЫҢЩҶ ЩӮШ·Ш№ЩҮ Ъ©ШҜ Щ…ЩҶ ЫҢЪ© Assembly Catalog ШұЩҲ Ъ©ЩҮ ШЁЩҮ Assembly Ш¬Ш§ШұЫҢ ШЁШұЩҶШ§Щ…ЩҮ Ш§ШҙШ§ШұЩҮ Щ…ЫҢвҖҢЪ©ЩҶЩҮ ШЁЩҮ AggregateCatalog Ш§Ш¶Ш§ЩҒЩҮ Ъ©ШұШҜЩ….
Щ…ШӘШҜ (batch.AddPart(this ШҜШұ ЩҲШ§ЩӮШ№ ШЁЩҮ Ш§ЫҢЩҶ Щ…Ш№ЩҶЫҢ Ш§ШіШӘ Ъ©ЩҮ ШЁЩҮ MEF ЪҜЩҒШӘЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ Ш§ЫҢЩҶ Ъ©Щ„Ш§Ші Щ…Щ…Ъ©ЩҶ Ш§ШіШӘ ШҙШ§Щ…Щ„ Export ЩҮШ§ЫҢЫҢ ШЁШ§ШҙШҜ Ъ©ЩҮ ШЁЩҮ ЫҢЪ© ЫҢШ§ ЪҶЩҶШҜ Import ЩҲШ§ШЁШіШӘЪҜЫҢ ШҜШ§ШұЩҶШҜ.
Щ…ШӘШҜ (AddExport(this ШҜШұ CompositionBatch ШЁЩҮ Ш§ЫҢЩҶ Щ…Ш№ЩҶЫҢ Ш§ШіШӘ Ъ©ЩҮ Ш§ЫҢЩҶ Ъ©Щ„Ш§Ші Щ…Щ…Ъ©ЩҶ Ш§ШіШӘ ШҙШ§Щ…Щ„ ExportЩҮШ§ЫҢЫҢ ШЁШ§ШҙШҜ Ъ©ЩҮ ШЁЩҮ Import ЩҲШ§ШЁШіШӘЪҜЫҢ ЩҶШҜШ§ШұЩҶШҜ.
ШӯШ§Щ„Ш§ ШЁШұШ§ЫҢ Щ…ШҙШ§ЩҮШҜЩҮ ЩҶШӘШ§ЫҢШ¬ Ъ©ШҜ ШІЫҢШұ ШұШ§ ШҜШұ Ъ©Щ„Ш§Ші Program Ш§Ш¶Ш§ЩҒЩҮ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ…:
static void Main( string[] args )
        {
            PluginManager plugin = new PluginManager();

            Console.WriteLine( string.Format( "Number Of ViewModels Before Plugin Setup Is [ {0} ]", plugin.ViewModels.Count ) );

            Console.WriteLine( Environment.NewLine );

            plugin.SetupManager();

            Console.WriteLine( string.Format( "Number Of ViewModels After Plugin Setup Is [ {0} ]", plugin.ViewModels.Count ) );

            Console.ReadLine();
        }

ШҜШұ Ъ©Щ„Ш§Ші ШЁШ§Щ„Ш§ Ш§ШЁШӘШҜШ§ ШӘШ№ШҜШ§ШҜ Ъ©Щ„Ш§ШівҖҢЩҮШ§ЫҢ Щ…ЩҲШ¬ЩҲШҜ ШҜШұ Щ„ЫҢШіШӘ ViewModels ШұЩҲ ЩӮШЁЩ„ Ш§ШІ Setup Ъ©ШұШҜЩҶ Plugin ЩҶЩ…Ш§ЫҢШҙ ШҜШ§ШҜЩҮ ШіЩҫШі ШЁШ№ШҜ Ш§ШІ Setup  Ъ©ШұШҜЩҶ Plugin  ШҜЩҲШЁШ§ШұЩҮ ШӘШ№ШҜШ§ШҜ Ъ©Щ„Ш§ШівҖҢЩҮШ§ЫҢ Щ…ЩҲШ¬ЩҲШҜ ШҜШұ Щ„ЫҢШіШӘ ViewModel ШұЩҲ Щ…ШҙШ§ЩҮШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ….Ъ©ЩҮ Ш®ШұЩҲШ¬ЫҢ ШЁЩҮ ШҙЪ©Щ„ ШІЫҢШұ ШӘЩҲЩ„ЫҢШҜ Ш®ЩҲШ§ЩҮШҜ ШҙШҜ.


Щ…ШӘШҜ SetupManager ШҜШұ Ъ©Щ„Ш§Ші Plugin (ШЁШ§ ШӘЩҲШ¬ЩҮ ШЁЩҮ AggregateCatalog) Ъ©ЩҮ ШҜШұ Ш§ЫҢЩҶ ШЁШұЩҶШ§Щ…ЩҮ ЩҒЩӮШ· Assembly  Ш¬Ш§ШұЫҢ ШұЩҲ ШЁЩҮШҙ Ш§Ш¶Ш§ЩҒЩҮ Ъ©ШұШҜЫҢЩ… ШӘЩ…Ш§Щ… Ъ©Щ„Ш§Ші ЩҮШ§ЫҢЫҢ ШұЩҲ Ъ©ЩҮ ЩҶЩҲШ№ IViewModel ШұЩҲ Export Ъ©ШұШҜЩҶШҜ ЩҫЫҢШҜШ§ Ъ©ШұШҜЩҮ ЩҲ ШҜШұ Щ„ЫҢШіШӘ Ш§Ш¶Ш§ЩҒЩҮ Щ…ЫҢвҖҢЪ©ЩҶЩҮ(Ш§ЫҢЩҶ Ъ©Ш§Шұ ШұЩҲ ШЁШ§ ШӘЩҲШ¬ЩҮ ШЁЩҮ ImportMany Attribute) Ш§ЩҶШ¬Ш§Щ… Щ…ЫҢШҜЩҮ.     • Щ…ШіШ№ЩҲШҜ ЩҫШ§Ъ©ШҜЩ„ ШҜШұ ЫұЫіЫ№ЫІ/Ы°Ы¶/ЫІЫ¶ ЫұЫі:ЫҙЫ·
       MEF ШЁШұШ§ЫҢ ЩҫЫҢШ§ШҜЩҮ ШіШ§ШІЫҢ Щ…ШЁШӯШ« Chaining Dependencies Ш§ШІ Щ…ЩҒЩҮЩҲЩ… Contract ШҜШұ Export Attribute Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶШҜ. ЩҫШ§ШұШ§Щ…ШӘШұ Ш§ЩҲЩ„ ШҜШұ Export ШЁШұШ§ЫҢ ContractName Ш§ШіШӘ. ШЁЩҮ ШөЩҲШұШӘ ШІЫҢШұ:
