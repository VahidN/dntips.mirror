Щ…Ш·Ш§Щ„ШЁ ЫіЫ¶Ы№Ы·
ЩҶЩҲЫҢШіЩҶШҜЪҜШ§ЩҶ ЫІЫ·Ы¶
ЪҜШұЩҲЩҮвҖҢЩҮШ§ЫҢ Щ…Ш·Ш§Щ„ШЁ ЫұЫ°ЫІЫҙ
ЩҶЩӮШҙЩҮвҖҢЩҮШ§ЫҢ ШұШ§ЩҮ ЫұЫұЫ№
ШҜЩҲШұЩҮвҖҢЩҮШ§ ЫұЫҙ
Ш§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§ ЫұЫ·Ы№ЫІЫІ
→ вҖ«ШЁЩҮШЁЩҲШҜ Ъ©ЫҢЩҒЫҢШӘ ШӘШіШӘвҖҢЩҶЩҲЫҢШіЫҢ ШЁШ§ Ш§Ш¬ШӘЩҶШ§ШЁ Ш§ШІ ШҙШЁЫҢЩҮвҖҢШіШ§ШІЫҢ (Mocking) Ъ©Щ„Ш§ШівҖҢЩҮШ§ЫҢ Щ…ШҙШ®Шө (Concrete Classes)
вҖ«Ш§Ш¬ШӘЩҶШ§ШЁ Ш§ШІ Ш§Щ„ЪҜЩҲЩҮШ§ЫҢ ЩҶШ§ШҜШұШіШӘ ШҜШұ ШӘШіШӘ ЩҲШ§ШӯШҜ ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ILogger ШҜШұ ШӘЩҲШіШ№ЩҮ .NET ←
Ш№ЩҶЩҲШ§ЩҶ:

вҖ«Ш§Ш¬ШӘЩҶШ§ШЁ Ш§ШІ Ш§Щ„ЪҜЩҲЩҮШ§ЫҢ ЩҶШ§ШҜШұШіШӘ Ъ©Ш§Шұ ШЁШ§ ШІЩ…Ш§ЩҶ ШҜШұ ШӘШіШӘвҖҢЩҶЩҲЫҢШіЫҢ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҮШ§ЫҢ ШҜШ§ШӘвҖҢЩҶШӘ


ЩҶЩҲЫҢШіЩҶШҜЩҮ: ЩҲШӯЫҢШҜ ЩҶШөЫҢШұЫҢ
ШӘШ§ШұЫҢШ®: ЫұЫҙЫ°Ыҙ/Ы°Ыұ/Ы°Ы№ ЫІЫІ:Ы°Ы·
ШўШҜШұШі: www.dntips.ir
Щ…ЩӮШҜЩ…ЩҮ
ШҜШұ ШҜЩҶЫҢШ§ЫҢ ШӘЩҲШіШ№ЩҮ ЩҶШұЩ…вҖҢШ§ЩҒШІШ§ШұШҢ ШЁЩҮвҖҢЩҲЫҢЪҳЩҮ ШҜШұ ЪҶШ§ШұЪҶЩҲШЁ Microsoft .NETШҢ ШӘШіШӘвҖҢЩҶЩҲЫҢШіЫҢ (Unit Testing) ЫҢЪ©ЫҢ Ш§ШІ Ш§ШұЪ©Ш§ЩҶ Ш§ШөЩ„ЫҢ ШӘШ¶Щ…ЫҢЩҶ Ъ©ЫҢЩҒЫҢШӘ Ъ©ШҜ Ш§ШіШӘ. ШӘШіШӘвҖҢЩҮШ§ЫҢ ЩҲШ§ШӯШҜ ШЁЩҮ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ Ъ©Щ…Ъ© Щ…ЫҢвҖҢЪ©ЩҶЩҶШҜ ШӘШ§ ШұЩҒШӘШ§Шұ Ъ©ШҜ ШұШ§ ШҜШұ ШҙШұШ§ЫҢШ· Щ…ШҙШ®Шө ШЁШұШұШіЫҢ Ъ©ШұШҜЩҮ ЩҲ Ш§ШІ ШөШӯШӘ Ш№Щ…Щ„Ъ©ШұШҜ ШўЩҶ Ш§Ш·Щ…ЫҢЩҶШ§ЩҶ ШӯШ§ШөЩ„ Ъ©ЩҶЩҶШҜ. ШЁШ§ Ш§ЫҢЩҶ ШӯШ§Щ„ШҢ Ш§ШіШӘЩҒШ§ШҜЩҮ ЩҶШ§ШҜШұШіШӘ Ш§ШІ ШЁШұШ®ЫҢ Ш§Щ„ЪҜЩҲЩҮШ§ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ ШЁЩҮ ШӘШіШӘвҖҢЩҮШ§ЫҢЫҢ ШҙЪ©ЩҶЩҶШҜЩҮШҢ ШәЫҢШұЩӮШ§ШЁЩ„ Ш§Ш№ШӘЩ…Ш§ШҜ ЩҲ ШҜШҙЩҲШ§Шұ ШҜШұ ЩҶЪҜЩҮШҜШ§ШұЫҢ Щ…ЩҶШ¬Шұ ШҙЩҲШҜ. ЫҢЪ©ЫҢ Ш§ШІ Ш§ЫҢЩҶ Ш§Щ„ЪҜЩҲЩҮШ§ЫҢ ЩҶШ§ШҜШұШіШӘ ЫҢШ§ Ш¶ШҜШ§Щ„ЪҜЩҲЩҮШ§ (Anti-Patterns)ШҢ ЩҶШӯЩҲЩҮ Ъ©Ш§Шұ ШЁШ§ ШІЩ…Ш§ЩҶ ШҜШұ ШӘШіШӘвҖҢЩҶЩҲЫҢШіЫҢ Ш§ШіШӘ. ЩҲШ§ШЁШіШӘЪҜЫҢ Щ…ШіШӘЩӮЫҢЩ… ШЁЩҮ ШІЩ…Ш§ЩҶ ЩҒШ№Щ„ЫҢ ШіЫҢШіШӘЩ…ШҢ Щ…Ш§ЩҶЩҶШҜ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ DateTime.NowШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ ЩҶШӘШ§ЫҢШ¬ЫҢ ШәЫҢШұЩӮШ§ШЁЩ„ ЩҫЫҢШҙвҖҢШЁЫҢЩҶЫҢ ШЁЩҮ ЩҮЩ…ШұШ§ЩҮ ШҜШ§ШҙШӘЩҮ ШЁШ§ШҙШҜШҢ ШІЫҢШұШ§ ШІЩ…Ш§ЩҶ ШҜШұ ЩҮШұ Ш§Ш¬ШұШ§ Щ…ШӘШәЫҢШұ Ш§ШіШӘ. Ш§ЫҢЩҶ Щ…ЩӮШ§Щ„ЩҮ ШЁЩҮ ШЁШұШұШіЫҢ Ш§ЫҢЩҶ Ш¶ШҜШ§Щ„ЪҜЩҲШҢ ШҜЩ„Ш§ЫҢЩ„ Щ…ШҙЪ©Щ„вҖҢШіШ§ШІ ШЁЩҲШҜЩҶ ШўЩҶ ЩҲ ШұШ§ЩҮвҖҢШӯЩ„вҖҢЩҮШ§ЫҢ Ш§ШөЩҲЩ„ЫҢ ШЁШұШ§ЫҢ Щ…ШҜЫҢШұЫҢШӘ ШІЩ…Ш§ЩҶ ШҜШұ ШӘШіШӘвҖҢЩҶЩҲЫҢШіЫҢ Щ…ЫҢвҖҢЩҫШұШҜШ§ШІШҜ. ЩҮШҜЩҒ Ш§ЫҢЩҶ Ш§ШіШӘ Ъ©ЩҮ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ ШҜШ§ШӘвҖҢЩҶШӘ ШЁШӘЩҲШ§ЩҶЩҶШҜ ШӘШіШӘвҖҢЩҮШ§ЫҢЫҢ ЩҫШ§ЫҢШҜШ§ШұШҢ ЩӮШ§ШЁЩ„ ШӘЪ©ШұШ§Шұ ЩҲ Щ…ЩҶШіШ¬Щ… ШЁЩҶЩҲЫҢШіЩҶШҜ.

ЪҶШұШ§ Ъ©Ш§Шұ ШЁШ§ ШІЩ…Ш§ЩҶ ЫҢЪ© Ш¶ШҜШ§Щ„ЪҜЩҲ Щ…ШӯШіЩҲШЁ Щ…ЫҢвҖҢШҙЩҲШҜШҹ
ШЁШіЫҢШ§ШұЫҢ Ш§ШІ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҮШ§ ШЁЩҮ ШІЩ…Ш§ЩҶ ЩҲШ§ШЁШіШӘЩҮ ЩҮШіШӘЩҶШҜШҢ Щ…Ш«Щ„Ш§ЩӢ ШЁШұШ§ЫҢ Щ…ШӯШ§ШіШЁЩҮ ШӘШ§ШұЫҢШ® Ш§ЩҶЩӮШ¶Ш§ШҢ ШІЩ…Ш§ЩҶвҖҢШЁЩҶШҜЫҢ ЩҲШёШ§ЫҢЩҒ ЫҢШ§ Ш«ШЁШӘ ШұЩҲЫҢШҜШ§ШҜЩҮШ§. Ш§Щ…Ш§ ШҜШұ ШӘШіШӘвҖҢЩҶЩҲЫҢШіЫҢШҢ ЩҲШ§ШЁШіШӘЪҜЫҢ Щ…ШіШӘЩӮЫҢЩ… ШЁЩҮ ШІЩ…Ш§ЩҶ ЩҒШ№Щ„ЫҢ ШіЫҢШіШӘЩ… ЪҶШ§Щ„ШҙвҖҢЩҮШ§ЫҢЫҢ Ш§ЫҢШ¬Ш§ШҜ Щ…ЫҢвҖҢЪ©ЩҶШҜ. ЩҒШұШ¶ Ъ©ЩҶЫҢШҜ Ъ©ШҜЫҢ ШҜШ§ШұЫҢШҜ Ъ©ЩҮ ШЁШұШұШіЫҢ Щ…ЫҢвҖҢЪ©ЩҶШҜ ШўЫҢШ§ ЫҢЪ© ШӘШ§ШұЫҢШ® Ш®Ш§Шө Ш§ШІ ШІЩ…Ш§ЩҶ ЩҒШ№Щ„ЫҢ ЪҜШ°ШҙШӘЩҮ Ш§ШіШӘ ЫҢШ§ Ш®ЫҢШұ. Ш§ЪҜШұ Ш§ШІ DateTime.Now Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ЩҶЫҢШҜШҢ ЩҶШӘЫҢШ¬ЩҮ ШӘШіШӘ ШЁЩҮ Щ„ШӯШёЩҮ Ш§Ш¬ШұШ§ЫҢ ШўЩҶ ШЁШіШӘЪҜЫҢ ШҜШ§ШұШҜ. Ш§ЫҢЩҶ ЫҢШ№ЩҶЫҢ ШӘШіШӘ ШҙЩ…Ш§ Щ…Щ…Ъ©ЩҶ Ш§ШіШӘ ШҜШұ ЫҢЪ© ШұЩҲШІ Щ…ЩҲЩҒЩӮ ЩҲ ШҜШұ ШұЩҲШІ ШҜЫҢЪҜШұ ЩҶШ§Щ…ЩҲЩҒЩӮ ШЁШ§ШҙШҜШҢ ШЁШҜЩҲЩҶ Ш§ЫҢЩҶЪ©ЩҮ ШӘШәЫҢЫҢШұЫҢ ШҜШұ Щ…ЩҶШ·ЩӮ Ъ©ШҜ Ш§ЫҢШ¬Ш§ШҜ ШҙШҜЩҮ ШЁШ§ШҙШҜ. Ш§ЫҢЩҶ ЩҶШ§ЩҫШ§ЫҢШҜШ§ШұЫҢШҢ ШӘШіШӘ ШұШ§ ШәЫҢШұЩӮШ§ШЁЩ„ Ш§Ш№ШӘЩ…Ш§ШҜ Щ…ЫҢвҖҢЪ©ЩҶШҜ ЩҲ Ш§ШөЩ„ Ъ©Щ„ЫҢШҜЫҢ ШӘШіШӘвҖҢЩҶЩҲЫҢШіЫҢШҢ ЫҢШ№ЩҶЫҢ ШӘЪ©ШұШ§ШұЩҫШ°ЫҢШұЫҢ (Repeatability)ШҢ ШұШ§ ЩҶЩӮШ¶ Щ…ЫҢвҖҢЪ©ЩҶШҜ.
ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ Щ…Ш«Ш§Щ„ШҢ Ъ©ШҜЫҢ ШұШ§ ШҜШұ ЩҶШёШұ ШЁЪҜЫҢШұЫҢШҜ Ъ©ЩҮ ШӘШ®ЩҒЫҢЩҒ ЫҢЪ© Щ…ШӯШөЩҲЩ„ ШұШ§ ШЁШұ Ш§ШіШ§Ші ШӘШ§ШұЫҢШ® Ш§ЩҶЩӮШ¶Ш§ Щ…ШӯШ§ШіШЁЩҮ Щ…ЫҢвҖҢЪ©ЩҶШҜ: 
public class DiscountCalculator
{
    public decimal CalculateDiscount(DateTime expiryDate)
    {
        return expiryDate > DateTime.Now ? 0.1m : 0m;
    }
}
ШӯШ§Щ„Ш§ ШӘШіШӘЫҢ ШЁШұШ§ЫҢ Ш§ЫҢЩҶ Ъ©ШҜ ШЁЩҶЩҲЫҢШіЫҢШҜ: 
[Fact]
public void CalculateDiscount_WhenExpiryDateIsFuture_ReturnsDiscount()
{
    var calculator = new DiscountCalculator();
    var expiryDate = new DateTime(2025, 12, 31);
    var discount = calculator.CalculateDiscount(expiryDate);
    Assert.Equal(0.1m, discount);
}
Ш§ЫҢЩҶ ШӘШіШӘ Ш§Щ…ШұЩҲШІ (Щ…Ш§ШұШі 2025) Щ…ЩҲЩҒЩӮ Ш§ШіШӘШҢ ШІЫҢШұШ§ DateTime.Now Ъ©Щ…ШӘШұ Ш§ШІ ШӘШ§ШұЫҢШ® Ш§ЩҶЩӮШ¶Ш§ (ШҜШіШ§Щ…ШЁШұ 2025) Ш§ШіШӘ. Ш§Щ…Ш§ Ш§ЪҜШұ ЩҮЩ…ЫҢЩҶ ШӘШіШӘ ШҜШұ ЪҳШ§ЩҶЩҲЫҢЩҮ 2026 Ш§Ш¬ШұШ§ ШҙЩҲШҜШҢ ШҙЪ©ШіШӘ Щ…ЫҢвҖҢШ®ЩҲШұШҜШҢ ЪҶЩҲЩҶ ШІЩ…Ш§ЩҶ ЩҒШ№Щ„ЫҢ Ш§ШІ ШӘШ§ШұЫҢШ® Ш§ЩҶЩӮШ¶Ш§ ЪҜШ°ШҙШӘЩҮ Ш§ШіШӘ. Ш§ЫҢЩҶ ШҙЪ©ЩҶЩҶШҜЪҜЫҢ ЩҶШҙШ§ЩҶвҖҢШҜЩҮЩҶШҜЩҮ ЫҢЪ© Ш¶ШҜШ§Щ„ЪҜЩҲШіШӘ.

ШұЩҲШҙвҖҢЩҮШ§ЫҢ ЩҶШ§ШҜШұШіШӘ Щ…ШҜЫҢШұЫҢШӘ ШІЩ…Ш§ЩҶ
ЫҢЪ©ЫҢ Ш§ШІ ШұЩҲЫҢЪ©ШұШҜЩҮШ§ЫҢ ШұШ§ЫҢШ¬ Ш§Щ…Ш§ ЩҶШ§ШҜШұШіШӘШҢ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Щ…ШӘШҜЩҮШ§ЫҢ Ш§ШіШӘШ§ШӘЫҢЪ© Щ…Ш§ЩҶЩҶШҜ DateTime.Now ЫҢШ§ ШӘШұЩҒЩҶШҜЩҮШ§ЫҢЫҢ Щ…Ш«Щ„ Щ…ШӘЩҲЩӮЩҒ Ъ©ШұШҜЩҶ Ш§Ш¬ШұШ§ЫҢ Ъ©ШҜ ШЁШ§ Thread.Sleep ШЁШұШ§ЫҢ ШҙШЁЫҢЩҮвҖҢШіШ§ШІЫҢ ЪҜШ°Шұ ШІЩ…Ш§ЩҶ Ш§ШіШӘ. Ш§ЫҢЩҶ ШұЩҲШҙвҖҢЩҮШ§ ЩҶЩҮвҖҢШӘЩҶЩҮШ§ ШӘШіШӘ ШұШ§ ЩҲШ§ШЁШіШӘЩҮ ШЁЩҮ ЩҲШ¶Ш№ЫҢШӘ ШіЫҢШіШӘЩ… Щ…ЫҢвҖҢЪ©ЩҶЩҶШҜШҢ ШЁЩ„Ъ©ЩҮ ЪҜШ§ЩҮЫҢ ШІЩ…Ш§ЩҶ Ш§Ш¬ШұШ§ЫҢ ШӘШіШӘ ШұШ§ ШЁЫҢвҖҢШҜЩ„ЫҢЩ„ Ш·ЩҲЩ„Ш§ЩҶЫҢ Щ…ЫҢвҖҢЪ©ЩҶЩҶШҜ. ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ Щ…Ш«Ш§Щ„: 
[Fact]
public void CalculateDiscount_AfterDelay_ReturnsNoDiscount()
{
    var calculator = new DiscountCalculator();
    var expiryDate = DateTime.Now.AddSeconds(1);
    Thread.Sleep(2000); // ШөШЁШұ ШЁШұШ§ЫҢ ЪҜШ°Шұ ШІЩ…Ш§ЩҶ
    var discount = calculator.CalculateDiscount(expiryDate);
    Assert.Equal(0m, discount);
}
Ш§ЫҢЩҶ ШӘШіШӘ ШЁЩҮ ШІЩ…Ш§ЩҶвҖҢШЁЩҶШҜЫҢ ШіЫҢШіШӘЩ… ЩҲШ§ШЁШіШӘЩҮ Ш§ШіШӘ ЩҲ Ш§ЪҜШұ Ш§Ш¬ШұШ§ ШЁЩҮ ЩҮШұ ШҜЩ„ЫҢЩ„ЫҢ Ъ©ЩҶШҜ ШҙЩҲШҜШҢ Щ…Щ…Ъ©ЩҶ Ш§ШіШӘ ЩҶШӘЫҢШ¬ЩҮ Щ…ШӘЩҒШ§ЩҲШӘЫҢ ШЁШҜЩҮШҜ. ЩҮЩ…ЪҶЩҶЫҢЩҶШҢ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Thread.Sleep Ъ©Ш§ШұШ§ЫҢЫҢ ШӘШіШӘ ШұШ§ Ъ©Ш§ЩҮШҙ Щ…ЫҢвҖҢШҜЩҮШҜ ЩҲ ШўЩҶ ШұШ§ ШәЫҢШұШӯШұЩҒЩҮвҖҢШ§ЫҢ Ш¬Щ„ЩҲЩҮ Щ…ЫҢвҖҢШҜЩҮШҜ.

ШұШ§ЩҮвҖҢШӯЩ„вҖҢЩҮШ§ЫҢ Ш§ШөЩҲЩ„ЫҢ
ШЁШұШ§ЫҢ ШұЩҒШ№ Ш§ЫҢЩҶ Щ…ШҙЪ©Щ„ШҢ ШЁШ§ЫҢШҜ ШІЩ…Ш§ЩҶ ШұШ§ ШЁЩҮвҖҢШ№ЩҶЩҲШ§ЩҶ ЫҢЪ© ЩҲШ§ШЁШіШӘЪҜЫҢ ШөШұЫҢШӯ (Explicit Dependency) ШҜШұ Ъ©ШҜ Щ…ШҜЫҢШұЫҢШӘ Ъ©ЩҶЫҢЩ…. ШҜЩҲ ШұЩҲШҙ Ш§ШөЩ„ЫҢ ШЁШұШ§ЫҢ Ш§ЫҢЩҶ Ъ©Ш§Шұ ЩҲШ¬ЩҲШҜ ШҜШ§ШұШҜ:
  1. ШӘШІШұЫҢЩӮ ЩҲШ§ШЁШіШӘЪҜЫҢвҖҢЩҮШ§ (Dependency Injection): ШІЩ…Ш§ЩҶ ШұШ§ Ш§ШІ Ш·ШұЫҢЩӮ ЫҢЪ© ШұШ§ШЁШ· (Interface) ШЁЩҮ Ъ©ШҜ ШӘШІШұЫҢЩӮ Ъ©ЩҶЫҢШҜ. Ш§ЫҢЩҶ Ъ©Ш§Шұ Ъ©ЩҶШӘШұЩ„ ШІЩ…Ш§ЩҶ ШұШ§ ШЁЩҮ Ш№ЩҮШҜЩҮ ШӘШіШӘ Щ…ЫҢвҖҢЪҜШ°Ш§ШұШҜ ЩҲ ЩҲШ§ШЁШіШӘЪҜЫҢ ШЁЩҮ ШіЫҢШіШӘЩ… ШұШ§ ШӯШ°ЩҒ Щ…ЫҢвҖҢЪ©ЩҶШҜ.
  2. Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ FakeTimeProvider ШҜШұ ШҜШ§ШӘвҖҢЩҶШӘ 8: Ш§ЫҢЩҶ Ш§ШЁШІШ§Шұ Ш¬ШҜЫҢШҜШҢ Ш§Щ…Ъ©Ш§ЩҶ ШҙШЁЫҢЩҮвҖҢШіШ§ШІЫҢ ШІЩ…Ш§ЩҶ ШұШ§ ШЁЩҮ ШҙЪ©Щ„ЫҢ ШіШ§ШҜЩҮ ЩҲ Ъ©Ш§ШұШўЩ…ШҜ ЩҒШұШ§ЩҮЩ… Щ…ЫҢвҖҢЪ©ЩҶШҜ.

ШұЩҲШҙ Ш§ЩҲЩ„: ШӘШІШұЫҢЩӮ ЩҲШ§ШЁШіШӘЪҜЫҢ
Ш§ШЁШӘШҜШ§ ЫҢЪ© ШұШ§ШЁШ· ШЁШұШ§ЫҢ ШҜШіШӘШұШіЫҢ ШЁЩҮ ШІЩ…Ш§ЩҶ ШӘШ№ШұЫҢЩҒ Ъ©ЩҶЫҢШҜ: 
public interface ITimeProvider
{
    DateTime GetCurrentTime();
}
ШіЩҫШі ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ ЩҲШ§ЩӮШ№ЫҢ ШўЩҶ ШұШ§ Ш§ЫҢШ¬Ш§ШҜ Ъ©ЩҶЫҢШҜ: 
public class SystemTimeProvider : ITimeProvider
{
    public DateTime GetCurrentTime() => DateTime.Now;
}
ШӯШ§Щ„Ш§ Ъ©ШҜ Ш§ШөЩ„ЫҢ ШұШ§ ШЁЩҮ Ш§ЫҢЩҶ ШҙЪ©Щ„ ШЁШ§ШІЩҶЩҲЫҢШіЫҢ Ъ©ЩҶЫҢШҜ: 
public class DiscountCalculator
{
    private readonly ITimeProvider _timeProvider;

    public DiscountCalculator(ITimeProvider timeProvider)
    {
        _timeProvider = timeProvider;
    }

    public decimal CalculateDiscount(DateTime expiryDate)
    {
        return expiryDate > _timeProvider.GetCurrentTime() ? 0.1m : 0m;
    }
}
ШҜШұ ШӘШіШӘШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЫҢШҜ ЫҢЪ© ЩҶШіШ®ЩҮ Ш¬Ш№Щ„ЫҢ (Mock) Ш§ШІ ITimeProvider Ш§ЫҢШ¬Ш§ШҜ Ъ©ЩҶЫҢШҜ: 
[Fact]
public void CalculateDiscount_WhenExpiryDateIsFuture_ReturnsDiscount()
{
    var timeProvider = new Mock<ITimeProvider>();
    timeProvider.Setup(tp => tp.GetCurrentTime()).Returns(new DateTime(2025, 1, 1));
    var calculator = new DiscountCalculator(timeProvider.Object);
    var expiryDate = new DateTime(2025, 12, 31);
    var discount = calculator.CalculateDiscount(expiryDate);
    Assert.Equal(0.1m, discount);
}
Ш§ЫҢЩҶ ШӘШіШӘ ЩҮЩ…ЫҢШҙЩҮ ЩҶШӘЫҢШ¬ЩҮ ЫҢЪ©ШіШ§ЩҶЫҢ ШҜШ§ШұШҜШҢ ШІЫҢШұШ§ ШІЩ…Ш§ЩҶ ШҜЫҢЪҜШұ Щ…ШӘШәЫҢШұ ЩҶЫҢШіШӘ.

ШұЩҲШҙ ШҜЩҲЩ…: Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ FakeTimeProvider
ШҜШұ ШҜШ§ШӘвҖҢЩҶШӘ 8ШҢ Ш§ШЁШІШ§Шұ FakeTimeProvider ШЁЩҮ ЩҒШ¶Ш§ЫҢ ЩҶШ§Щ… System.Threading Ш§Ш¶Ш§ЩҒЩҮ ШҙШҜЩҮ Ш§ШіШӘ. Ш§ЫҢЩҶ Ш§ШЁШІШ§Шұ ШЁШұШ§ЫҢ ШҙШЁЫҢЩҮвҖҢШіШ§ШІЫҢ ШІЩ…Ш§ЩҶ ШҜШұ ШӘШіШӘвҖҢЩҮШ§ Ш·ШұШ§ШӯЫҢ ШҙШҜЩҮ Ш§ШіШӘ. Щ…Ш«Ш§Щ„: 
[Fact]
public void CalculateDiscount_WithFakeTimeProvider_ReturnsDiscount()
{
    var fakeTime = new FakeTimeProvider();
    fakeTime.SetUtcNow(new DateTime(2025, 1, 1));
    var calculator = new DiscountCalculator(fakeTime);
    var expiryDate = new DateTime(2025, 12, 31);
    var discount = calculator.CalculateDiscount(expiryDate);
    Assert.Equal(0.1m, discount);
}
Ш§ЫҢЩҶ ШұЩҲШҙ ЩҶЫҢШ§ШІЫҢ ШЁЩҮ ШӘШ№ШұЫҢЩҒ ШұШ§ШЁШ· Ш¬ШҜШ§ЪҜШ§ЩҶЩҮ ЩҶШҜШ§ШұШҜ ЩҲ Щ…ШіШӘЩӮЫҢЩ…Ш§ЩӢ ШІЩ…Ш§ЩҶ ШұШ§ Ъ©ЩҶШӘШұЩ„ Щ…ЫҢвҖҢЪ©ЩҶШҜ.

Щ…ШІШ§ЫҢШ§ЫҢ ШұЩҲЫҢЪ©ШұШҜЩҮШ§ЫҢ ЩҫЫҢШҙЩҶЩҮШ§ШҜЫҢ
ЩҮШұ ШҜЩҲ ШұЩҲШҙ ШӘШ¶Щ…ЫҢЩҶ Щ…ЫҢвҖҢЪ©ЩҶЩҶШҜ Ъ©ЩҮ ШӘШіШӘвҖҢЩҮШ§:
  • ШӘЪ©ШұШ§ШұЩҫШ°ЫҢШұ ШЁШ§ШҙЩҶШҜ: ЩҶШӘЫҢШ¬ЩҮ ШӘШіШӘ ШЁЩҮ ШІЩ…Ш§ЩҶ ЩҲШ§ЩӮШ№ЫҢ ЩҲШ§ШЁШіШӘЩҮ ЩҶЫҢШіШӘ.
  • ШіШұЫҢШ№ Ш§Ш¬ШұШ§ ШҙЩҲЩҶШҜ: ЩҶЫҢШ§ШІЫҢ ШЁЩҮ ШӘШЈШ®ЫҢШұЩҮШ§ЫҢ Щ…ШөЩҶЩҲШ№ЫҢ ЩҶЫҢШіШӘ.
  • Щ…ЩҶШ№Ш·ЩҒ ШЁШ§ШҙЩҶШҜ: Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЫҢШҜ ШіЩҶШ§ШұЫҢЩҲЩҮШ§ЫҢ Щ…Ш®ШӘЩ„ЩҒ ШІЩ…Ш§ЩҶЫҢ ШұШ§ ШЁЩҮвҖҢШұШ§ШӯШӘЫҢ ШҙШЁЫҢЩҮвҖҢШіШ§ШІЫҢ Ъ©ЩҶЫҢШҜ.

ЩҶШӘЫҢШ¬ЩҮвҖҢЪҜЫҢШұЫҢ
Ъ©Ш§Шұ ШЁШ§ ШІЩ…Ш§ЩҶ ШҜШұ ШӘШіШӘвҖҢЩҶЩҲЫҢШіЫҢШҢ Ш§ЪҜШұ ШЁЩҮ ШҙЪ©Щ„ ЩҶШ§ШҜШұШіШӘ Ш§ЩҶШ¬Ш§Щ… ШҙЩҲШҜШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ ШЁЩҮ ЫҢЪ© Ш¶ШҜШ§Щ„ЪҜЩҲ ШӘШЁШҜЫҢЩ„ ШҙЩҲШҜ ЩҲ Ъ©ЫҢЩҒЫҢШӘ ШӘШіШӘвҖҢЩҮШ§ ШұШ§ Ъ©Ш§ЩҮШҙ ШҜЩҮШҜ. ЩҲШ§ШЁШіШӘЪҜЫҢ ШЁЩҮ Щ…ШӘШҜЩҮШ§ЫҢ Ш§ШіШӘШ§ШӘЫҢЪ© Щ…Ш«Щ„ DateTime.Now ЫҢШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ШӘШұЩҒЩҶШҜЩҮШ§ЫҢ ШәЫҢШұШӯШұЩҒЩҮвҖҢШ§ЫҢ Щ…Ш§ЩҶЩҶШҜ Thread.SleepШҢ ШӘШіШӘвҖҢЩҮШ§ ШұШ§ ШҙЪ©ЩҶЩҶШҜЩҮ ЩҲ ШәЫҢШұЩӮШ§ШЁЩ„ Ш§Ш№ШӘЩ…Ш§ШҜ Щ…ЫҢвҖҢЪ©ЩҶШҜ. ШҜШұ Щ…ЩӮШ§ШЁЩ„ШҢ ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ШӘЪ©ЩҶЫҢЪ©вҖҢЩҮШ§ЫҢЫҢ Щ…Ш§ЩҶЩҶШҜ ШӘШІШұЫҢЩӮ ЩҲШ§ШЁШіШӘЪҜЫҢ ЫҢШ§ Ш§ШЁШІШ§ШұЩҮШ§ЫҢ Щ…ШҜШұЩҶ Щ…Ш«Щ„ FakeTimeProvider ШҜШұ ШҜШ§ШӘвҖҢЩҶШӘ 8ШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ ШІЩ…Ш§ЩҶ ШұШ§ ШЁЩҮ ШҙЪ©Щ„ЫҢ Ъ©ЩҶШӘШұЩ„вҖҢШҙШҜЩҮ Щ…ШҜЫҢШұЫҢШӘ Ъ©ШұШҜ. Ш§ЫҢЩҶ ШұЩҲЫҢЪ©ШұШҜЩҮШ§ ЩҶЩҮвҖҢШӘЩҶЩҮШ§ ЩҫШ§ЫҢШҜШ§ШұЫҢ ШӘШіШӘвҖҢЩҮШ§ ШұШ§ Ш§ЩҒШІШ§ЫҢШҙ Щ…ЫҢвҖҢШҜЩҮЩҶШҜШҢ ШЁЩ„Ъ©ЩҮ ШЁЩҮ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ ШҜШ§ШӘвҖҢЩҶШӘ Ш§Ш¬Ш§ШІЩҮ Щ…ЫҢвҖҢШҜЩҮЩҶШҜ Ъ©ШҜЫҢ ШӘЩ…ЫҢШІШӘШұ ЩҲ ЩӮШ§ШЁЩ„ ЩҶЪҜЩҮШҜШ§ШұЫҢвҖҢШӘШұ ШЁЩҶЩҲЫҢШіЩҶШҜ. ШӘЩҲШөЫҢЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ ШҜШұ ЩҫШұЩҲЪҳЩҮвҖҢЩҮШ§ЫҢ Ш®ЩҲШҜШҢ ШІЩ…Ш§ЩҶ ШұШ§ ШЁЩҮвҖҢШ№ЩҶЩҲШ§ЩҶ ЫҢЪ© ЩҲШ§ШЁШіШӘЪҜЫҢ ШөШұЫҢШӯ ШҜШұ ЩҶШёШұ ШЁЪҜЫҢШұЫҢШҜ ЩҲ Ш§ШІ Ш§ШЁШІШ§ШұЩҮШ§ЫҢ Щ…ЩҶШ§ШіШЁ ШЁШұШ§ЫҢ ШӘШіШӘ ШўЩҶ ШЁЩҮШұЩҮ ШЁШЁШұЫҢШҜ ШӘШ§ Ш§ШІ Щ…ШҙЪ©Щ„Ш§ШӘ ШұШ§ЫҢШ¬ Ш§ЫҢЩҶ Ш¶ШҜШ§Щ„ЪҜЩҲ ШҜШұ Ш§Щ…Ш§ЩҶ ШЁЩ…Ш§ЩҶЫҢШҜ.


Щ…ШҙШ§ЩҮШҜЩҮ Щ…Ш·Щ„ШЁ Ш§ШөЩ„ЫҢ
Щ…Ш·Ш§Щ„ШЁ Щ…ШҙШ§ШЁЩҮ
    Ш§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§Ш¬Щ„ЩҲЪҜЫҢШұЫҢ Ш§ШІ Ш§ЩҒШҙШ§ЫҢ ШӯШ§Щ„ШӘ Ш®ШөЩҲШөЫҢ ШҜШұ ШӘШіШӘвҖҢЩҮШ§ЫҢ ЩҲШ§ШӯШҜШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ TimeProvider ШҜШ§ШӘвҖҢЩҶШӘ 8 ШЁШұШ§ЫҢ ШӘШіШӘ Ъ©ШҜЩҶЩҲЫҢШіЫҢ ЩҲШ§ШЁШіШӘЩҮ ШЁЩҮ ШІЩ…Ш§ЩҶШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§Ш¬Ш°Ш§ШЁЫҢШӘвҖҢШЁШ®ШҙЫҢ ШЁЩҮ ШӘШіШӘвҖҢЩҮШ§ЫҢ ЩҲШ§ШӯШҜШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§ШЁЩҮШЁЩҲШҜ Ъ©ЫҢЩҒЫҢШӘ ШӘШіШӘвҖҢЩҶЩҲЫҢШіЫҢ ШЁШ§ Ш§Ш¬ШӘЩҶШ§ШЁ Ш§ШІ ШҙШЁЫҢЩҮвҖҢШіШ§ШІЫҢ (Mocking) Ъ©Щ„Ш§ШівҖҢЩҮШ§ЫҢ Щ…ШҙШ®Шө (Concrete Classes)Ш§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§Ш¬Щ„ЩҲЪҜЫҢШұЫҢ Ш§ШІ ЩҶШҙШӘ ШҜШ§ЩҶШҙ ШҜШ§Щ…ЩҶЩҮ ШҜШұ ШӘШіШӘвҖҢЩҮШ§ЫҢ ЩҲШ§ШӯШҜ ШҜШұ ШӘЩҲШіШ№ЩҮ .NETШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§ШіШ§ШҜЩҮвҖҢШіШ§ШІЫҢ ШӘШіШӘ ЩҲШ§ШӯШҜ ШҜШұ ШіЫҢвҖҢШҙШ§ШұЩҫ ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ЩҒШұЫҢЩ…вҖҢЩҲШұЪ© Щ…Ш§Ъ©ЫҢЩҶЪҜШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§ЪҶШұШ§ ЩҶШЁШ§ЫҢШҜ Щ…ШӘШҜЩҮШ§ЫҢ Ш®ШөЩҲШөЫҢ ШұШ§ ШҜШұ ШўШІЩ…ЩҲЩҶ ЩҲШ§ШӯШҜ ШӘШіШӘ Ъ©ЩҶЫҢЩ…ШҹШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§Ш§Ш¬ШӘЩҶШ§ШЁ Ш§ШІ Ш§Щ„ЪҜЩҲЩҮШ§ЫҢ ЩҶШ§ШҜШұШіШӘ ШҜШұ ШӘШіШӘ ЩҲШ§ШӯШҜ ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ILogger ШҜШұ ШӘЩҲШіШ№ЩҮ .NETЩ…ШіЫҢШұШұШ§ЩҮвҖҢЩҮШ§ШұШ§ЩҮЪ©Ш§ШұЩҮШ§ЫҢЫҢ ШЁШұШ§ЫҢ ШЁЩҮШЁЩҲШҜ Ъ©ЫҢЩҒЫҢШӘ ШӘШіШӘвҖҢЩҮШ§ЫҢ ЩҲШ§ШӯШҜ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҮШ§ЫҢ ШҜШ§ШӘвҖҢЩҶШӘЩҶШёШұШ§ШӘ Щ…Ш·Ш§Щ„ШЁASP.NET MVC #18
