вҖ«вҖ«Ш§Ш¬ШӘЩҶШ§ШЁ Ш§ШІ Ш§Щ„ЪҜЩҲЩҮШ§ЫҢ ЩҶШ§ШҜШұШіШӘ ШҜШұ ШӘШіШӘ ЩҲШ§ШӯШҜ ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ILogger ШҜШұ ШӘЩҲШіШ№ЩҮ .NET
→ вҖ«Ш§Ш¬ШӘЩҶШ§ШЁ Ш§ШІ Ш§Щ„ЪҜЩҲЩҮШ§ЫҢ ЩҶШ§ШҜШұШіШӘ Ъ©Ш§Шұ ШЁШ§ ШІЩ…Ш§ЩҶ ШҜШұ ШӘШіШӘвҖҢЩҶЩҲЫҢШіЫҢ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҮШ§ЫҢ ШҜШ§ШӘвҖҢЩҶШӘ
вҖ«Ш§Ш¬ШӘЩҶШ§ШЁ Ш§ШІ Ш¶ШҜ Ш§Щ„ЪҜЩҲЫҢ В«ШўЩ„ЩҲШҜЪҜЫҢ Ъ©ШҜВ» ШҜШұ ШӘЩҲШіШ№ЩҮвҖҢЫҢ ШӘШіШӘвҖҢЩҮШ§ЫҢ ЩҲШ§ШӯШҜ ←

Щ…ЩӮШҜЩ…ЩҮ
ШӘШіШӘ ЩҲШ§ШӯШҜ (Unit Testing) ЫҢЪ©ЫҢ Ш§ШІ Ш§ШұЪ©Ш§ЩҶ Ш§ШөЩ„ЫҢ ШӘЩҲШіШ№ЩҮ ЩҶШұЩ…вҖҢШ§ЩҒШІШ§ШұЩҮШ§ЫҢ ШЁШ§Ъ©ЫҢЩҒЫҢШӘ ШҜШұ ЪҶШ§ШұЪҶЩҲШЁ .NET Ш§ШіШӘ. Ш§ЫҢЩҶ ЩҒШұШўЫҢЩҶШҜ ШЁЩҮ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ Ъ©Щ…Ъ© Щ…ЫҢвҖҢЪ©ЩҶШҜ ШӘШ§ Ш§ШІ ШөШӯШӘ Ш№Щ…Щ„Ъ©ШұШҜ Ъ©ШҜЩҮШ§ЫҢ Ш®ЩҲШҜ Ш§Ш·Щ…ЫҢЩҶШ§ЩҶ ШӯШ§ШөЩ„ Ъ©ЩҶЩҶШҜ ЩҲ Ш§ШҙЪ©Ш§Щ„Ш§ШӘ ШұШ§ ШҜШұ Щ…ШұШ§ШӯЩ„ Ш§ЩҲЩ„ЫҢЩҮ ШҙЩҶШ§ШіШ§ЫҢЫҢ Ъ©ЩҶЩҶШҜ. ШЁШ§ Ш§ЫҢЩҶ ШӯШ§Щ„ШҢ ЪҜШ§ЩҮЫҢ Ш§ЩҲЩӮШ§ШӘ Ш§ШіШӘЩҒШ§ШҜЩҮ ЩҶШ§ШҜШұШіШӘ Ш§ШІ Ш§ШЁШІШ§ШұЩҮШ§ ЩҲ Щ…ЩҒШ§ЩҮЫҢЩ…ШҢ Щ…Ш§ЩҶЩҶШҜ Щ„Ш§ЪҜШұ (Logger)ШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ ШЁЩҮ Ш§Щ„ЪҜЩҲЩҮШ§ЫҢ ЩҶШ§ШҜШұШіШӘ (Anti-Patterns) ШҜШұ ШӘШіШӘ ЩҲШ§ШӯШҜ Щ…ЩҶШ¬Шұ ШҙЩҲШҜ. ЫҢЪ©ЫҢ Ш§ШІ ЪҶШ§Щ„ШҙвҖҢЩҮШ§ЫҢ ШұШ§ЫҢШ¬ ШҜШұ Ш§ЫҢЩҶ ШӯЩҲШІЩҮШҢ ЩҶШӯЩҲЩҮ ШӘШ№Ш§Щ…Щ„ ШЁШ§ ILoggerШҢ ШұШ§ШЁШ· Ш§ШіШӘШ§ЩҶШҜШ§ШұШҜ Щ„Ш§ЪҜЫҢЩҶЪҜ ШҜШұ Ш§Ъ©ЩҲШіЫҢШіШӘЩ… .NETШҢ Ш§ШіШӘ. ЩҮШҜЩҒ Ш§ЫҢЩҶ Щ…ЩӮШ§Щ„ЩҮШҢ ШЁШұШұШіЫҢ Ш§ЫҢЩҶ ЪҶШ§Щ„ШҙвҖҢЩҮШ§ ЩҲ Ш§ШұШ§ШҰЩҮ ШұШ§ЩҮЪ©Ш§ШұЩҮШ§ЫҢЫҢ ШЁШұШ§ЫҢ Ш§Ш¬ШӘЩҶШ§ШЁ Ш§ШІ ЩҫЫҢЪҶЫҢШҜЪҜЫҢвҖҢЩҮШ§ЫҢ ШәЫҢШұШ¶ШұЩҲШұЫҢ ЩҲ ШЁЩҮШЁЩҲШҜ Ъ©ЫҢЩҒЫҢШӘ ШӘШіШӘвҖҢЩҮШ§ Ш§ШіШӘ. ШҜШұ Ш§ЫҢЩҶ Щ…ШіЫҢШұШҢ ШЁЩҮ Щ…ЩҒШ§ЩҮЫҢЩ…ЫҢ Щ…Ш§ЩҶЩҶШҜ ШӘШіШӘвҖҢЩҫШ°ЫҢШұЫҢ (Testability)ШҢ Ш¬ШҜШ§ШіШ§ШІЫҢ ЩҲШ§ШЁШіШӘЪҜЫҢвҖҢЩҮШ§ (Dependency Isolation) ЩҲ Ш§ЩҶШӘШ®Ш§ШЁ Ш§ШЁШІШ§Шұ Щ…ЩҶШ§ШіШЁ ЩҫШұШҜШ§Ш®ШӘЩҮ Ш®ЩҲШ§ЩҮШҜ ШҙШҜ ШӘШ§ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ ШЁШӘЩҲШ§ЩҶЩҶШҜ ШӘШіШӘвҖҢЩҮШ§ЫҢЫҢ Ъ©Ш§ШұШўЩ…ШҜШӘШұ ЩҲ ЩӮШ§ШЁЩ„ ЩҶЪҜЩҮШҜШ§ШұЫҢвҖҢШӘШұ ШЁЩҶЩҲЫҢШіЩҶШҜ.

ЪҶШ§Щ„Шҙ ШӘШіШӘ Щ„Ш§ЪҜЫҢЩҶЪҜ ШҜШұ ШӘЩҲШіШ№ЩҮ .NET
Щ„Ш§ЪҜЫҢЩҶЪҜ (Logging) ЫҢЪ© ЩҶЫҢШ§ШІ Ш¬Ш§ЩҶШЁЫҢ Ш§Щ…Ш§ ШӯЫҢШ§ШӘЫҢ ШҜШұ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҶЩҲЫҢШіЫҢ Ш§ШіШӘ Ъ©ЩҮ ШЁЩҮ ШұШҜЫҢШ§ШЁЫҢ ШұЩҒШӘШ§Шұ ШЁШұЩҶШ§Щ…ЩҮ ЩҲ Ш№ЫҢШЁвҖҢЫҢШ§ШЁЫҢ Ъ©Щ…Ъ© Щ…ЫҢвҖҢЪ©ЩҶШҜ. ШҜШұ .NETШҢ ШұШ§ШЁШ· ILogger ШЁЩҮвҖҢШ№ЩҶЩҲШ§ЩҶ ШЁШ®ШҙЫҢ Ш§ШІ ШЁШіШӘЩҮ Microsoft.Extensions.LoggingШҢ Ш§Щ…Ъ©Ш§ЩҶ Ш«ШЁШӘ ЩҫЫҢШ§Щ…вҖҢЩҮШ§ ШұШ§ ШЁЩҮвҖҢШөЩҲШұШӘ Ш§ШіШӘШ§ЩҶШҜШ§ШұШҜ ЩҒШұШ§ЩҮЩ… Щ…ЫҢвҖҢЪ©ЩҶШҜ. ШЁШ§ Ш§ЫҢЩҶ ШӯШ§Щ„ШҢ ЩҲЩӮШӘЫҢ ШөШӯШЁШӘ Ш§ШІ ШӘШіШӘ ЩҲШ§ШӯШҜ Щ…ЫҢвҖҢШҙЩҲШҜШҢ Ш§ЫҢЩҶ ШіШӨШ§Щ„ Щ…Ш·ШұШӯ Щ…ЫҢвҖҢШҙЩҲШҜ Ъ©ЩҮ ШўЫҢШ§ ШЁШ§ЫҢШҜ Щ„Ш§ЪҜЫҢЩҶЪҜ ШұШ§ ШӘШіШӘ Ъ©ШұШҜШҹ ЩҫШ§ШіШ® ШЁЩҮ Ш§ЫҢЩҶ ШіШӨШ§Щ„ ШЁЩҮ ЩҮШҜЩҒ Щ„Ш§ЪҜЫҢЩҶЪҜ ШЁШіШӘЪҜЫҢ ШҜШ§ШұШҜ. Ш§ЪҜШұ Щ„Ш§ЪҜвҖҢЩҮШ§ ШЁШ®ШҙЫҢ Ш§ШІ ЩҶЫҢШ§ШІЩҮШ§ЫҢ Ъ©ШіШЁвҖҢЩҲЪ©Ш§Шұ ШЁШ§ШҙЩҶШҜ ЩҲ ШЁШұШ§ЫҢ Ъ©Ш§ШұШЁШұШ§ЩҶ ЩҶЩҮШ§ЫҢЫҢ ЫҢШ§ Ш°ЫҢвҖҢЩҶЩҒШ№Ш§ЩҶ ШӘЩҲЩ„ЫҢШҜ ШҙЩҲЩҶШҜШҢ ШӘШіШӘ ШўЩҶвҖҢЩҮШ§ Ш¶ШұЩҲШұЫҢ Ш§ШіШӘ. Ш§Щ…Ш§ Ш§ЪҜШұ ШөШұЩҒШ§ЩӢ Ш§ШЁШІШ§ШұЫҢ ШЁШұШ§ЫҢ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ ШЁШ§ШҙЩҶШҜШҢ Щ…Щ…Ъ©ЩҶ Ш§ШіШӘ ЩҶЫҢШ§ШІЫҢ ШЁЩҮ ШӘШіШӘ Щ…ШіШӘЩӮЫҢЩ… ЩҶШҜШ§ШҙШӘЩҮ ШЁШ§ШҙЩҶШҜ.
ЫҢЪ©ЫҢ Ш§ШІ Ш§Щ„ЪҜЩҲЩҮШ§ЫҢ ЩҶШ§ШҜШұШіШӘ ШұШ§ЫҢШ¬ШҢ ШӘЩ„Ш§Шҙ ШЁШұШ§ЫҢ ШӘШіШӘ Щ…ШіШӘЩӮЫҢЩ… ЩҒШұШ§Ш®ЩҲШ§ЩҶЫҢвҖҢЩҮШ§ЫҢ ILogger ШҜШұ Ъ©ШҜ Ш§ШіШӘ. ШЁЩҮвҖҢШ№ЩҶЩҲШ§ЩҶ Щ…Ш«Ш§Щ„ШҢ ЩҒШұШ¶ Ъ©ЩҶЫҢШҜ ЫҢЪ© Ъ©Щ„Ш§Ші ШіШ§ШҜЩҮ Щ…Ш§ЩҶЩҶШҜ ШІЫҢШұ ШҜШ§ШұЫҢЩ…:
public class OrderPlacedEventHandler : IDomainEventHandler<OrderPlacedEvent>
{
    private readonly ILogger<OrderPlacedEventHandler> _logger;
    public OrderPlacedEventHandler(ILogger<OrderPlacedEventHandler> logger)
    {
        _logger = logger;
    }
    public Task Handle(OrderPlacedEvent domainEvent, CancellationToken cancellationToken)
    {
        _logger.LogInformation("Order placed: {OrderId} at {OccurredOn}", domainEvent.OrderId, domainEvent.OccurredOn);
        return Task.CompletedTask;
    }
}
ШҜШұ Ш§ЫҢЩҶШ¬Ш§ШҢ ШӘШіШӘ Щ…ШіШӘЩӮЫҢЩ… ЩҒШұШ§Ш®ЩҲШ§ЩҶЫҢ LogInformation Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ ЩҲШіЩҲШіЩҮвҖҢШ§ЩҶЪҜЫҢШІ ШЁШ§ШҙШҜШҢ Ш§Щ…Ш§ Ш§ЫҢЩҶ ШұЩҲЫҢЪ©ШұШҜ Ш§ШәЩ„ШЁ ШЁЩҮ ЩҫЫҢЪҶЫҢШҜЪҜЫҢ ШәЫҢШұШ¶ШұЩҲШұЫҢ Щ…ЩҶШ¬Шұ Щ…ЫҢвҖҢШҙЩҲШҜ.

Ш§Щ„ЪҜЩҲЫҢ ЩҶШ§ШҜШұШіШӘ: Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Щ…ЩҲЪ© ЩҫЫҢЪҶЫҢШҜЩҮ ШЁШұШ§ЫҢ ILogger
ЫҢЪ©ЫҢ Ш§ШІ ШұЩҲШҙвҖҢЩҮШ§ЫҢ ШұШ§ЫҢШ¬ ШЁШұШ§ЫҢ ШӘШіШӘ ЪҶЩҶЫҢЩҶ Ъ©ШҜЫҢШҢ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ЩҒШұЫҢЩ…вҖҢЩҲШұЪ©вҖҢЩҮШ§ЫҢ Щ…ЩҲЪ©ЫҢЩҶЪҜ (Mocking) Щ…Ш§ЩҶЩҶШҜ Moq Ш§ШіШӘ. ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ Щ…Щ…Ъ©ЩҶ Ш§ШіШӘ ЫҢЪ© ШҙЫҢШЎ Щ…ЩҲЪ© Ш§ШІ ILogger Ш§ЫҢШ¬Ш§ШҜ Ъ©ЩҶЩҶШҜ ЩҲ ШіЩҫШі ШЁШұШұШіЫҢ Ъ©ЩҶЩҶШҜ Ъ©ЩҮ ШўЫҢШ§ Щ…ШӘШҜ LogInformation ШЁШ§ ЩҫШ§ШұШ§Щ…ШӘШұЩҮШ§ЫҢ Щ…ЩҲШұШҜ Ш§ЩҶШӘШёШ§Шұ ЩҒШұШ§Ш®ЩҲШ§ЩҶЫҢ ШҙШҜЩҮ Ш§ШіШӘ. ШЁЩҮвҖҢШ№ЩҶЩҲШ§ЩҶ Щ…Ш«Ш§Щ„:
[Fact]
public async Task Handle_ShouldLogInformation()
{
    var logger = new Mock<ILogger<OrderPlacedEventHandler>>();
    var handler = new OrderPlacedEventHandler(logger.Object);
    var domainEvent = new OrderPlacedEvent(Guid.NewGuid());
    
    await handler.Handle(domainEvent, CancellationToken.None);
    
    logger.Verify(
        x => x.Log(
            LogLevel.Information,
            It.IsAny<EventId>(),
            It.IsAny<It.IsAnyType>(),
            It.IsAny<Exception>(),
            It.IsAny<Func<It.IsAnyType, Exception?, string>>()),
        Times.Once);
}
Ш§ЫҢЩҶ Ъ©ШҜШҢ ЩҒШұШ§Ш®ЩҲШ§ЩҶЫҢ LogInformation ШұШ§ ШӘШЈЫҢЫҢШҜ Щ…ЫҢвҖҢЪ©ЩҶШҜШҢ Ш§Щ…Ш§ ЪҶЩҶШҜ Щ…ШҙЪ©Щ„ ШҜШ§ШұШҜ:
  1. ЩҫЫҢЪҶЫҢШҜЪҜЫҢ ШЁШ§Щ„Ш§: ШӘЩҶШёЫҢЩ… ЩҲ ШӘШЈЫҢЫҢШҜ Щ…ЩҲЪ© ШЁШұШ§ЫҢ ILogger ШЁЩҮвҖҢШҜЩ„ЫҢЩ„ ШіШ§Ш®ШӘШ§Шұ ЩҫЫҢЪҶЫҢШҜЩҮ Щ…ШӘШҜ Log (Ъ©ЩҮ ШӘЩ…Ш§Щ… Щ…ШӘШҜЩҮШ§ЫҢ ШіШ·Шӯ ШЁШ§Щ„Ш§ ШЁЩҮ ШўЩҶ Ш§ШұШ¬Ш§Ш№ Щ…ЫҢвҖҢШҜЩҮЩҶШҜ) ШҜШҙЩҲШ§Шұ Ш§ШіШӘ.
  2. ** ШҙЪ©ЩҶЩҶШҜЪҜЫҢ (Fragility)**: Ш§ЪҜШұ ЩӮШ§Щ„ШЁ ЩҫЫҢШ§Щ… ЫҢШ§ ШіШ·Шӯ Щ„Ш§ЪҜ ШӘШәЫҢЫҢШұ Ъ©ЩҶШҜШҢ ШӘШіШӘ ШҙЪ©ШіШӘ Щ…ЫҢвҖҢШ®ЩҲШұШҜШҢ ШӯШӘЫҢ Ш§ЪҜШұ Щ…ЩҶШ·ЩӮ Ш§ШөЩ„ЫҢ ШЁШҜЩҲЩҶ ШӘШәЫҢЫҢШұ ШЁШ§ЩӮЫҢ ШЁЩ…Ш§ЩҶШҜ.
  3. ШӘЩ…ШұЪ©ШІ ЩҶШ§ШҜШұШіШӘ: ШӘШіШӘ ЩҲШ§ШӯШҜ ШЁШ§ЫҢШҜ ШұЩҒШӘШ§Шұ Ш§ШөЩ„ЫҢ Ъ©ШҜ (Щ…Ш§ЩҶЩҶШҜ ЩҫШұШҜШ§ШІШҙ ШұЩҲЫҢШҜШ§ШҜ) ШұШ§ ШЁШұШұШіЫҢ Ъ©ЩҶШҜШҢ ЩҶЩҮ Ш¬ШІШҰЫҢШ§ШӘ ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ Щ„Ш§ЪҜЫҢЩҶЪҜ.
Ш§ЫҢЩҶ Ш§Щ„ЪҜЩҲ ЩҶШҙШ§ЩҶвҖҢШҜЩҮЩҶШҜЩҮ ЫҢЪ© Ш¶ШҜШ§Щ„ЪҜЩҲШіШӘШҢ ШІЫҢШұШ§ ШӘШіШӘ ШұШ§ ШЁЫҢШҙ Ш§ШІ ШӯШҜ ШЁЩҮ Ш¬ШІШҰЫҢШ§ШӘ ЩҲШ§ШЁШіШӘЩҮ Щ…ЫҢвҖҢЪ©ЩҶШҜ ЩҲ Ш§ШІ ЩҮШҜЩҒ Ш§ШөЩ„ЫҢ ШӘШіШӘ ЩҲШ§ШӯШҜШҢ ЫҢШ№ЩҶЫҢ ШӘШЈЫҢЫҢШҜ ШұЩҒШӘШ§ШұШҢ ШҜЩҲШұ Щ…ЫҢвҖҢШҙЩҲШҜ.

ШұШ§ЩҮЪ©Ш§Шұ: Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ЫҢЪ© Fake Logger ШіШ§ШҜЩҮ
ШЁШұШ§ЫҢ ШұЩҒШ№ Ш§ЫҢЩҶ Щ…ШҙЪ©Щ„ШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ Ш§ШІ ЫҢЪ© ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ Ш¬Ш№Щ„ЫҢ (Fake) ЩҲ ШіШ§ШҜЩҮвҖҢШӘШұ ШЁШұШ§ЫҢ ILogger Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ШұШҜ Ъ©ЩҮ ШЁШҜЩҲЩҶ ЩҶЫҢШ§ШІ ШЁЩҮ Щ…ЩҲЪ©ЫҢЩҶЪҜ ЩҫЫҢЪҶЫҢШҜЩҮШҢ ШӘШіШӘ ШұШ§ ШӘШіЩҮЫҢЩ„ Ъ©ЩҶШҜ. ШЁЩҮвҖҢШ№ЩҶЩҲШ§ЩҶ Щ…Ш«Ш§Щ„ШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ Ш§ШІ NullLogger Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ШұШҜ Ъ©ЩҮ ШҜШұ ШЁШіШӘЩҮ Microsoft.Extensions.Logging.Abstractions Ш§ШұШ§ШҰЩҮ ШҙШҜЩҮ Ш§ШіШӘ. Ш§ЫҢЩҶ Щ„Ш§ЪҜШұШҢ ЩҫЫҢШ§Щ…вҖҢЩҮШ§ ШұШ§ ЩҶШ§ШҜЫҢШҜЩҮ Щ…ЫҢвҖҢЪҜЫҢШұШҜ ЩҲ ЩҶЫҢШ§ШІЫҢ ШЁЩҮ ШӘШЈЫҢЫҢШҜ ЩҒШұШ§Ш®ЩҲШ§ЩҶЫҢ ЩҶШҜШ§ШұШҜ:
[Fact]
public async Task Handle_ShouldCompleteSuccessfully()
{
    var logger = NullLogger<OrderPlacedEventHandler>.Instance;
    var handler = new OrderPlacedEventHandler(logger);
    var domainEvent = new OrderPlacedEvent(Guid.NewGuid());
    
    var result = await handler.Handle(domainEvent, CancellationToken.None);
    
    Assert.Equal(Task.CompletedTask, result);
}
ШҜШұ Ш§ЫҢЩҶШ¬Ш§ШҢ ШӘШіШӘ ШЁШұ ШұЩҒШӘШ§Шұ Ш§ШөЩ„ЫҢ (Ш§ШӘЩ…Ш§Щ… Щ…ЩҲЩҒЩӮЫҢШӘвҖҢШўЩ…ЫҢШІ Щ…ШӘШҜ Handle) ШӘЩ…ШұЪ©ШІ ШҜШ§ШұШҜ ЩҲ Щ„Ш§ЪҜЫҢЩҶЪҜ ШұШ§ ШЁЩҮвҖҢШ№ЩҶЩҲШ§ЩҶ ЫҢЪ© Ш¬ШІШҰЫҢШ§ШӘ ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ ЩҶШ§ШҜЫҢШҜЩҮ Щ…ЫҢвҖҢЪҜЫҢШұШҜ. Ш§ЫҢЩҶ ШұЩҲЫҢЪ©ШұШҜ:
  • ШіШ§ШҜЪҜЫҢ: ЩҶЫҢШ§ШІ ШЁЩҮ ШӘЩҶШёЫҢЩ…Ш§ШӘ ЩҫЫҢЪҶЫҢШҜЩҮ ШұШ§ ШӯШ°ЩҒ Щ…ЫҢвҖҢЪ©ЩҶШҜ.
  • ЩҫШ§ЫҢШҜШ§ШұЫҢ: ШӘШіШӘ ШұШ§ ШҜШұ ШЁШұШ§ШЁШұ ШӘШәЫҢЫҢШұШ§ШӘ ШҜШұ Щ„Ш§ЪҜЫҢЩҶЪҜ Щ…ЩӮШ§ЩҲЩ… Щ…ЫҢвҖҢЪ©ЩҶШҜ.
  • ШӘЩ…ШұЪ©ШІ: ЩҮШҜЩҒ Ш§ШөЩ„ЫҢ ШӘШіШӘ ЩҲШ§ШӯШҜ ШұШ§ ШӯЩҒШё Щ…ЫҢвҖҢЪ©ЩҶШҜ.

ШІЩ…Ш§ЩҶ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ILogger ЩҲШ§ЩӮШ№ЫҢ
ШҜШұ ШЁШұШ®ЫҢ ШіЩҶШ§ШұЫҢЩҲЩҮШ§ШҢ Щ…Щ…Ъ©ЩҶ Ш§ШіШӘ ШЁШ®ЩҲШ§ЩҮЫҢЩ… Щ„Ш§ЪҜЫҢЩҶЪҜ ШұШ§ ШЁЩҮвҖҢШөЩҲШұШӘ ЩҲШ§ЩӮШ№ЫҢ ШӘШіШӘ Ъ©ЩҶЫҢЩ…ШҢ ШЁЩҮвҖҢЩҲЫҢЪҳЩҮ ШҜШұ ШӘШіШӘвҖҢЩҮШ§ЫҢ ЫҢЪ©ЩҫШ§ШұЪҶЪҜЫҢ (Integration Tests). ШҜШұ Ш§ЫҢЩҶ Щ…ЩҲШ§ШұШҜШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ Ш§ШІ ILoggerFactory ШЁШұШ§ЫҢ Ш§ЫҢШ¬Ш§ШҜ ЫҢЪ© Щ„Ш§ЪҜШұ ЩҲШ§ЩӮШ№ЫҢ Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ШұШҜ Ъ©ЩҮ Ш®ШұЩҲШ¬ЫҢ ШўЩҶ ШЁЩҮ Ъ©ЩҶШіЩҲЩ„ ЫҢШ§ ЫҢЪ© ЩҒШ§ЫҢЩ„ ЩҮШҜШ§ЫҢШӘ ШҙЩҲШҜ. ШЁЩҮвҖҢШ№ЩҶЩҲШ§ЩҶ Щ…Ш«Ш§Щ„:
ILoggerFactory factory = LoggerFactory.Create(builder => builder.AddConsole());
ILogger<OrderPlacedEventHandler> logger = factory.CreateLogger<OrderPlacedEventHandler>();
Ш§ЫҢЩҶ ШұЩҲШҙ ШЁШұШ§ЫҢ ШӘШЈЫҢЫҢШҜ Ш§ЫҢЩҶЪ©ЩҮ Щ„Ш§ЪҜвҖҢЩҮШ§ ШЁЩҮвҖҢШҜШұШіШӘЫҢ Ш«ШЁШӘ Щ…ЫҢвҖҢШҙЩҲЩҶШҜ Щ…ЩҒЫҢШҜ Ш§ШіШӘШҢ Ш§Щ…Ш§ Щ…Ш№Щ…ЩҲЩ„Ш§ЩӢ ШҜШұ ШӘШіШӘ ЩҲШ§ШӯШҜ ШӘЩҲШөЫҢЩҮ ЩҶЩ…ЫҢвҖҢШҙЩҲШҜШҢ ШІЫҢШұШ§ ШӘШіШӘ ШұШ§ Ш§ШІ ШӯШ§Щ„ШӘ Ш§ЫҢШІЩҲЩ„ЩҮ Ш®Ш§ШұШ¬ Щ…ЫҢвҖҢЪ©ЩҶШҜ.

ШЁЩҮШӘШұЫҢЩҶ ШұЩҲШҙвҖҢЩҮШ§ ШЁШұШ§ЫҢ ШӘШіШӘвҖҢЩҫШ°ЫҢШұЫҢ
ШЁШұШ§ЫҢ ШЁЩҮШЁЩҲШҜ ШӘШіШӘвҖҢЩҫШ°ЫҢШұЫҢ ЩҮЩҶЪҜШ§Щ… Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ILoggerШҢ ЪҶЩҶШҜ Ш§ШөЩ„ Ъ©Щ„ЫҢШҜЫҢ ШұШ§ ШЁШ§ЫҢШҜ ШұШ№Ш§ЫҢШӘ Ъ©ШұШҜ:
  1. Ш¬ШҜШ§ШіШ§ШІЫҢ ЩҲШ§ШЁШіШӘЪҜЫҢвҖҢЩҮШ§: ILogger ШұШ§ Ш§ШІ Ш·ШұЫҢЩӮ ШӘШІШұЫҢЩӮ ЩҲШ§ШЁШіШӘЪҜЫҢ (Dependency Injection) ЩҲШ§ШұШҜ Ъ©ЩҶЫҢШҜ ШӘШ§ ШЁШӘЩҲШ§ЩҶЫҢШҜ ШЁЩҮвҖҢШұШ§ШӯШӘЫҢ ШўЩҶ ШұШ§ ШЁШ§ ЫҢЪ© ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ Ш¬Ш№Щ„ЫҢ Ш¬Ш§ЫҢЪҜШІЫҢЩҶ Ъ©ЩҶЫҢШҜ.
  2. ШӯШҜШ§ЩӮЩ„ Ъ©ШұШҜЩҶ ШӘШіШӘ Щ„Ш§ЪҜЫҢЩҶЪҜ: ЩҒЩӮШ· ШҜШұ ШөЩҲШұШӘЫҢ Щ„Ш§ЪҜЫҢЩҶЪҜ ШұШ§ ШӘШіШӘ Ъ©ЩҶЫҢШҜ Ъ©ЩҮ ШЁШ®ШҙЫҢ Ш§ШІ Щ…ЩҶШ·ЩӮ Ш§ШөЩ„ЫҢ ШЁШ§ШҙШҜШҢ ЩҶЩҮ ЫҢЪ© Ш§ШЁШІШ§Шұ Ш¬Ш§ЩҶШЁЫҢ.
  3. Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Ш§ШЁШІШ§ШұЩҮШ§ЫҢ Щ…ЩҶШ§ШіШЁ: Ш§ШІ NullLogger ЫҢШ§ FakeLogger (ШҜШұ ШЁШіШӘЩҮ Microsoft.Extensions.Diagnostics.Testing) ШЁШұШ§ЫҢ ШӘШіШӘвҖҢЩҮШ§ЫҢ ШіШ§ШҜЩҮвҖҢШӘШұ Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ЩҶЫҢШҜ.

ЩҶШӘЫҢШ¬ЩҮвҖҢЪҜЫҢШұЫҢ
ШӘШіШӘ ЩҲШ§ШӯШҜ ШҜШұ .NET ЩҒШұШўЫҢЩҶШҜЫҢ Ш§ШіШӘ Ъ©ЩҮ ЩҶЫҢШ§ШІЩ…ЩҶШҜ ШҜЩӮШӘ ШҜШұ Ш§ЩҶШӘШ®Ш§ШЁ Ш§ШЁШІШ§ШұЩҮШ§ ЩҲ ШұЩҲШҙвҖҢЩҮШ§ Ш§ШіШӘ. Ш§ШіШӘЩҒШ§ШҜЩҮ ЩҶШ§ШҜШұШіШӘ Ш§ШІ ILoggerШҢ Щ…Ш§ЩҶЩҶШҜ Щ…ЩҲЪ©ЫҢЩҶЪҜ ЩҫЫҢЪҶЫҢШҜЩҮ ЫҢШ§ ШӘЩ…ШұЪ©ШІ ШЁЫҢШҙ Ш§ШІ ШӯШҜ ШЁШұ Ш¬ШІШҰЫҢШ§ШӘ Щ„Ш§ЪҜЫҢЩҶЪҜШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ ШЁЩҮ Ш§Щ„ЪҜЩҲЩҮШ§ЫҢ ЩҶШ§ШҜШұШіШӘ Щ…ЩҶШ¬Шұ ШҙЩҲШҜ Ъ©ЩҮ ШӘШіШӘвҖҢЩҮШ§ ШұШ§ ШҙЪ©ЩҶЩҶШҜЩҮ ЩҲ ШәЫҢШұЩӮШ§ШЁЩ„ ЩҶЪҜЩҮШҜШ§ШұЫҢ Щ…ЫҢвҖҢЪ©ЩҶШҜ. ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ШұШ§ЩҮЪ©Ш§ШұЩҮШ§ЫҢЫҢ Щ…Ш§ЩҶЩҶШҜ NullLogger ЫҢШ§ ШӘЩ…ШұЪ©ШІ ШЁШұ ШұЩҒШӘШ§Шұ Ш§ШөЩ„ЫҢ ШЁЩҮвҖҢШ¬Ш§ЫҢ Ш¬ШІШҰЫҢШ§ШӘ ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢШҢ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЩҶШҜ ШӘШіШӘвҖҢЩҮШ§ЫҢЫҢ ШЁЩҶЩҲЫҢШіЩҶШҜ Ъ©ЩҮ ЩҮЩ… ШіШ§ШҜЩҮвҖҢШӘШұ ШЁШ§ШҙЩҶШҜ ЩҲ ЩҮЩ… Щ…ШӨШ«ШұШӘШұ. ШҜШұ ЩҶЩҮШ§ЫҢШӘШҢ ЩҮШҜЩҒ Ш§ЫҢЩҶ Ш§ШіШӘ Ъ©ЩҮ ШӘШіШӘ ЩҲШ§ШӯШҜ ШЁЩҮвҖҢШ№ЩҶЩҲШ§ЩҶ ЫҢЪ© Ш§ШЁШІШ§Шұ Ш§Ш№ШӘЩ…Ш§ШҜШіШ§ШІ Ш№Щ…Щ„ Ъ©ЩҶШҜШҢ ЩҶЩҮ ЫҢЪ© Щ…Ш§ЩҶШ№ ШҜШұ Щ…ШіЫҢШұ ШӘЩҲШіШ№ЩҮ.

    Ш§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§Ъ©Ш§ЩҮШҙ ЩҲШ§ШЁШіШӘЪҜЫҢ ШЁЩҮ Ъ©ШӘШ§ШЁШ®Ш§ЩҶЩҮвҖҢЩҮШ§ЫҢ Ш«Ш§Щ„Ш« ШЁЩҮ Ъ©Щ…Ъ© ШӘШІШұЫҢЩӮ ЩҲШ§ШЁШіШӘЪҜЫҢвҖҢЩҮШ§Ш§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§ШіШ§ШҜЩҮвҖҢШіШ§ШІЫҢ ШӘШіШӘ ЩҲШ§ШӯШҜ ШҜШұ ШіЫҢвҖҢШҙШ§ШұЩҫ ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ЩҒШұЫҢЩ…вҖҢЩҲШұЪ© Щ…Ш§Ъ©ЫҢЩҶЪҜШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§Ш¬Щ„ЩҲЪҜЫҢШұЫҢ Ш§ШІ Ш§ЩҒШҙШ§ЫҢ ШӯШ§Щ„ШӘ Ш®ШөЩҲШөЫҢ ШҜШұ ШӘШіШӘвҖҢЩҮШ§ЫҢ ЩҲШ§ШӯШҜШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§Ш§Ш¬ШӘЩҶШ§ШЁ Ш§ШІ Ш¶ШҜ Ш§Щ„ЪҜЩҲЫҢ В«ШўЩ„ЩҲШҜЪҜЫҢ Ъ©ШҜВ» ШҜШұ ШӘЩҲШіШ№ЩҮвҖҢЫҢ ШӘШіШӘвҖҢЩҮШ§ЫҢ ЩҲШ§ШӯШҜШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§ЪҶШұШ§ ЩҶШЁШ§ЫҢШҜ Щ…ШӘШҜЩҮШ§ЫҢ Ш®ШөЩҲШөЫҢ ШұШ§ ШҜШұ ШўШІЩ…ЩҲЩҶ ЩҲШ§ШӯШҜ ШӘШіШӘ Ъ©ЩҶЫҢЩ…ШҹШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§Ш¬Щ„ЩҲЪҜЫҢШұЫҢ Ш§ШІ ЩҶШҙШӘ ШҜШ§ЩҶШҙ ШҜШ§Щ…ЩҶЩҮ ШҜШұ ШӘШіШӘвҖҢЩҮШ§ЫҢ ЩҲШ§ШӯШҜ ШҜШұ ШӘЩҲШіШ№ЩҮ .NETШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§Ъ©ШҙЩҒ ЩҲЫҢЪҳЪҜЫҢвҖҢЩҮШ§ЫҢ Ш¬ШҜЫҢШҜ SDK ШЁШ§ Ъ©Щ…Ъ© ЩҮЩҲШҙ Щ…ШөЩҶЩҲШ№ЫҢ: NullLoggerШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§Ш¬Ш°Ш§ШЁЫҢШӘвҖҢШЁШ®ШҙЫҢ ШЁЩҮ ШӘШіШӘвҖҢЩҮШ§ЫҢ ЩҲШ§ШӯШҜЩ…ШіЫҢШұШұШ§ЩҮвҖҢЩҮШ§ШұШ§ЩҮЪ©Ш§ШұЩҮШ§ЫҢЫҢ ШЁШұШ§ЫҢ ШЁЩҮШЁЩҲШҜ Ъ©ЫҢЩҒЫҢШӘ ШӘШіШӘвҖҢЩҮШ§ЫҢ ЩҲШ§ШӯШҜ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҮШ§ЫҢ ШҜШ§ШӘвҖҢЩҶШӘЩ…Ш·Ш§Щ„ШЁШҜЫҢЪҜШұ ILogger ШұШ§ Mock ЩҶЪ©ЩҶЫҢШҜ: ШӘШіШӘ Щ„Ш§ЪҜЫҢЩҶЪҜ ШҜШұ ШҜШ§ШӘвҖҢЩҶШӘ ШЁШ§ FakeLogger
