ШҜШұ ШІЩ…Ш§ЩҶ ЩҶЩҲШҙШӘЩҶ ШӘШіШӘвҖҢЩҮШ§ЫҢ Щ…Ш®ШӘЩ„ЩҒ (Unit - Integration - UI) ЪҜШ§ЩҮЫҢ Ш§ЩҲЩӮШ§ШӘ ЩҫЫҢШҙ Щ…ЫҢвҖҢШўЫҢШҜ Ъ©ЩҮ ШЁШ®ЩҲШ§ЩҮЫҢШҜ ШӘЩ…Ш§Щ…ЫҢ Ш®ШөЩҲШөЫҢШӘвҖҢЩҮШ§ЫҢ ЫҢЪ© ШҙЫҢШЎ ШұШ§ ШӘШ§ЫҢЫҢШҜ Ъ©ЩҶЫҢШҜ. Щ…Ш№Щ…ЩҲЩ„Ш§ ЩҶЩҲШҙШӘЩҶ Ш§Ш№ШӘШЁШ§ШұШіЩҶШ¬ЫҢ ШЁШұШ§ЫҢ ЩҮЩ…ЩҮ Ш®ШөЩҲШөЫҢШӘвҖҢЩҮШ§ ЩҲ ЩҮЩ…ЫҢЩҶ Ш·ЩҲШұ ЩҫЫҢШ§Щ…вҖҢЩҮШ§ЫҢ Ш§ШіШӘШ«ЩҶШ§ШЎ ШЁШұШ§ЫҢ ЩҮШұ ЫҢЪ© ШҜШұ ШІЩ…Ш§ЩҶ Ш№ШҜЩ… ШӘШ§ЫҢЫҢШҜ Ш§Ш№ШӘШЁШ§ШұШҢ Ъ©Ш§Шұ ШЁШіЫҢШ§Шұ ШІЩ…Ш§ЩҶШЁШұЫҢ Ш§ШіШӘ. ШҜШұ Ш§ЫҢЩҶ Щ…ЩӮШ§Щ„ЩҮ ШЁЩҮ ШҙЩ…Ш§ ЩҶШҙШ§ЩҶ Ш®ЩҲШ§ЩҮЩ… ШҜШ§ШҜ Ъ©ЩҮ ЪҶЪҜЩҲЩҶЩҮ ШЁШ§ ЩҶЩҲШҙШӘЩҶ ЫҢЪ© Ш§Ш№ШӘШЁШ§ШұШіЩҶШ¬ Ш№Щ…ЩҲЩ…ЫҢ Ш§ШІ Ш§ШӘЩ„Ш§ЩҒ ШІЩ…Ш§ЩҶ ШІЫҢШ§ШҜЫҢ Ш¬Щ„ЩҲЪҜЫҢШұЫҢ Ъ©ЩҶЫҢШҜ.

ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Ъ©Щ„Ш§Ші ШІЫҢШұ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ Ъ©Ш§Шұ Ш§Ш№ШӘШЁШ§ШұШіЩҶШ¬ЫҢ ШұШ§ ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Reflection ШЁЩҮ ШұШ§ШӯШӘЫҢ Ш§ЩҶШ¬Ш§Щ… ШҜШ§ШҜ. ШҜШұ Ш§ЫҢЩҶШ¬Ш§ ШЁШұШ§ЫҢ Ш§Ш№ШӘШЁШ§ШұШіЩҶШ¬ЫҢ DateTime Ш§ШІ Ъ©Щ„Ш§Ші DateTimeAssert Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ШұШҜЩҮвҖҢШ§ЫҢЩ….
public class PropertiesValidator<TK, T> where T : new() where TK : new()
{
    static TK _instance;

    public static TK Instance
    {
        get
        {
            if (_instance == null)
            {
                _instance = new TK();
            }
            return _instance;
        }
    }

    public void Validate(T expectedObject, T realObject, params string[] propertiesNotToCompare)
    {
        var properties = realObject.GetType().GetProperties();
        foreach (var currentRealProperty in properties)
        {
            if (!propertiesNotToCompare.Contains(currentRealProperty.Name))
            {
                var currentExpectedProperty = expectedObject.GetType().GetProperty(currentRealProperty.Name);
                var exceptionMessage = $"The property {currentRealProperty.Name} of class {currentRealProperty.DeclaringType?.Name} was not as expected.";

                if (currentRealProperty.PropertyType != typeof(DateTime) && currentRealProperty.PropertyType != typeof(DateTime?))
                {
                    Assert.AreEqual( currentExpectedProperty.GetValue( expectedObject,
                                                                        null ),
                                        currentRealProperty.GetValue( realObject,
                                                                    null ),
                                        exceptionMessage );
                }
                else
                {
                    DateTimeAssert.Validate( currentExpectedProperty.GetValue( expectedObject,
                                                                                null ) as DateTime?,
                                                currentRealProperty.GetValue( realObject,
                                                                            null ) as DateTime?,
                                                TimeSpan.FromMinutes( 5 ) );
                }
            }
        }
    }
}
