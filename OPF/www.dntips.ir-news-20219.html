Щ…ЩӮШҜЩ…ЩҮ
ШҜШұ ШҜЩҶЫҢШ§ЫҢ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҶЩҲЫҢШіЫҢШҢ Щ…ШҜЫҢШұЫҢШӘ ШІЩ…Ш§ЩҶ ЩҲ ШӘШ§ШұЫҢШ® ЩҮЩ…ЫҢШҙЩҮ ЫҢЪ©ЫҢ Ш§ШІ ЪҶШ§Щ„ШҙвҖҢЩҮШ§ЫҢ Ш§ШөЩ„ЫҢ ШЁЩҲШҜЩҮ Ш§ШіШӘШҢ ШЁЩҮвҖҢЩҲЫҢЪҳЩҮ ШІЩ…Ш§ЩҶЫҢ Ъ©ЩҮ ЩҶЫҢШ§ШІ ШЁЩҮ ШӘШіШӘ ЩҲШ§ШӯШҜ (Unit Testing) Ъ©ШҜЩҮШ§ЫҢЫҢ ШҜШ§ШұЫҢЩ… Ъ©ЩҮ ШЁЩҮ ШІЩ…Ш§ЩҶ ЩҒШ№Щ„ЫҢ ЩҲШ§ШЁШіШӘЩҮ ЩҮШіШӘЩҶШҜ. ШҜШұ ЩҶШіШ®ЩҮвҖҢЩҮШ§ЫҢ ЩӮШҜЫҢЩ…ЫҢвҖҢШӘШұ ЩҒШұЫҢЩ…вҖҢЩҲШұЪ© .NETШҢ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ ШЁШұШ§ЫҢ ШӘШіШӘ ЪҶЩҶЫҢЩҶ Ъ©ШҜЩҮШ§ЫҢЫҢ Щ…Ш¬ШЁЩҲШұ ШЁЩҮ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ШұШ§ЩҮвҖҢШӯЩ„вҖҢЩҮШ§ЫҢ ЩҫЫҢЪҶЫҢШҜЩҮ Щ…Ш§ЩҶЩҶШҜ ШұШ§ШЁШ·вҖҢЩҮШ§ (Interfaces) ЫҢШ§ Ш§ШЁШІШ§ШұЩҮШ§ЫҢ ШҙШ®Шө Ш«Ш§Щ„Ш« ШЁЩҲШҜЩҶШҜ. ШЁШ§ Щ…Ш№ШұЩҒЫҢ .NET 8ШҢ Щ…Ш§ЫҢЪ©ШұЩҲШіШ§ЩҒШӘ ЫҢЪ© Ш§ШЁШІШ§Шұ ЩӮШҜШұШӘЩ…ЩҶШҜ ШЁЩҮ ЩҶШ§Щ… TimeProvider ШұШ§ Ш§ШұШ§ШҰЩҮ Ъ©ШұШҜЩҮ Ш§ШіШӘ Ъ©ЩҮ Ш§ЫҢЩҶ Щ…ШҙЪ©Щ„Ш§ШӘ ШұШ§ ШЁЩҮ ШҙЪ©Щ„ЫҢ ШіШ§ШҜЩҮ ЩҲ Ъ©Ш§ШұШўЩ…ШҜ ШӯЩ„ Щ…ЫҢвҖҢЪ©ЩҶШҜ. Ш§ЫҢЩҶ Ш§ШЁШІШ§Шұ ШЁЩҮ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ Ш§Ш¬Ш§ШІЩҮ Щ…ЫҢвҖҢШҜЩҮШҜ ШӘШ§ ШІЩ…Ш§ЩҶ ШұШ§ ШҜШұ ШӘШіШӘвҖҢЩҮШ§ Ъ©ЩҶШӘШұЩ„ Ъ©ЩҶЩҶШҜ ЩҲ Ш§ШІ ЩҲШ§ШЁШіШӘЪҜЫҢ Щ…ШіШӘЩӮЫҢЩ… ШЁЩҮ ШІЩ…Ш§ЩҶ ШіЫҢШіШӘЩ…вҖҢШ№Ш§Щ…Щ„ (Щ…Ш§ЩҶЩҶШҜ DateTime.Now) ШұЩҮШ§ ШҙЩҲЩҶШҜ. ШҜШұ Ш§ЫҢЩҶ Щ…ЩӮШ§Щ„ЩҮШҢ ШЁЩҮ ШЁШұШұШіЫҢ Щ…ЩҒЩҮЩҲЩ… TimeProviderШҢ ЩӮШ§ШЁЩ„ЫҢШӘвҖҢЩҮШ§ЫҢ ШўЩҶ ЩҲ ЩҶШӯЩҲЩҮ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ШўЩҶ ШҜШұ ШӘШіШӘ ЩҲШ§ШӯШҜ ЩҫШұШҜШ§Ш®ШӘЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ. ЩҮШҜЩҒ Ш§ЫҢЩҶ Ш§ШіШӘ Ъ©ЩҮ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ .NET ШЁШӘЩҲШ§ЩҶЩҶШҜ ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Ш§ЫҢЩҶ Ш§ШЁШІШ§ШұШҢ ШӘШіШӘвҖҢЩҮШ§ЫҢЫҢ ЩҫШ§ЫҢШҜШ§ШұШӘШұ ЩҲ ЩӮШ§ШЁЩ„ Ш§Ш№ШӘЩ…Ш§ШҜШӘШұ ШЁЩҶЩҲЫҢШіЩҶШҜ.

ЪҶШ§Щ„ШҙвҖҢЩҮШ§ЫҢ ШӘШіШӘ Ъ©ШҜЩҮШ§ЫҢ ЩҲШ§ШЁШіШӘЩҮ ШЁЩҮ ШІЩ…Ш§ЩҶ
ЩҫЫҢШҙ Ш§ШІ .NET 8ШҢ ШӘШіШӘ Ъ©ШҜЩҮШ§ЫҢЫҢ Ъ©ЩҮ ШЁЩҮ ШІЩ…Ш§ЩҶ ЩҲШ§ШЁШіШӘЩҮ ШЁЩҲШҜЩҶШҜШҢ ШЁШ§ Щ…ШҙЪ©Щ„Ш§ШӘЫҢ ЩҮЩ…ШұШ§ЩҮ ШЁЩҲШҜ. ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ Щ…Ш«Ш§Щ„ШҢ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ DateTime.Now ЫҢШ§ DateTimeOffset.UtcNow ШЁЩҮ ШҜЩ„ЫҢЩ„ Щ…Ш§ЩҮЫҢШӘ Ш§ШіШӘШ§ШӘЫҢЪ© ЩҲ ЩҲШ§ШЁШіШӘЪҜЫҢ ШЁЩҮ ШІЩ…Ш§ЩҶ ЩҲШ§ЩӮШ№ЫҢ ШіЫҢШіШӘЩ…ШҢ Ъ©ЩҶШӘШұЩ„ ШІЩ…Ш§ЩҶ ШҜШұ ШӘШіШӘвҖҢЩҮШ§ ШұШ§ ШҜШҙЩҲШ§Шұ Щ…ЫҢвҖҢЪ©ШұШҜ. ШұШ§ЩҮвҖҢШӯЩ„вҖҢЩҮШ§ЫҢ ШұШ§ЫҢШ¬ ШҙШ§Щ…Щ„ Ш§ЫҢШ¬Ш§ШҜ ШұШ§ШЁШ·вҖҢЩҮШ§ЫҢ ШіЩҒШ§ШұШҙЫҢ (Щ…Ш§ЩҶЩҶШҜ ITimeService) ЩҲ ШӘШІШұЫҢЩӮ ЩҲШ§ШЁШіШӘЪҜЫҢ (Dependency Injection) ШЁЩҲШҜЩҶШҜ. ЩҶЩ…ЩҲЩҶЩҮвҖҢШ§ЫҢ Ш§ШІ Ш§ЫҢЩҶ ШұЩҲЫҢЪ©ШұШҜ ШұШ§ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ ШҜШұ Ъ©ШҜ ШІЫҢШұ Щ…ШҙШ§ЩҮШҜЩҮ Ъ©ШұШҜ:
public interface ITimeService
{
    DateTime GetCurrentTime();
}

public class SystemTimeService : ITimeService
{
    public DateTime GetCurrentTime() => DateTime.Now;
}

public class MyService
{
    private readonly ITimeService _timeService;
    public MyService(ITimeService timeService)
    {
        _timeService = timeService;
    }

    public string GetGreeting()
    {
        var hour = _timeService.GetCurrentTime().Hour;
        return hour < 12 ? "ШөШЁШӯ ШЁШ®ЫҢШұ" : "Ш№ШөШұ ШЁШ®ЫҢШұ";
    }
}
Ш§ЫҢЩҶ ШұЩҲШҙ Ш§ЪҜШұЪҶЩҮ Ъ©Ш§ШұШўЩ…ШҜ ШЁЩҲШҜШҢ Ш§Щ…Ш§ ЩҶЫҢШ§ШІ ШЁЩҮ Ъ©ШҜ Ш§Ш¶Ш§ЩҒЫҢ ЩҲ ЩҫЫҢЪҶЫҢШҜЪҜЫҢ ШҜШұ Щ…ШҜЫҢШұЫҢШӘ ЩҲШ§ШЁШіШӘЪҜЫҢвҖҢЩҮШ§ ШҜШ§ШҙШӘ. Ш№Щ„Ш§ЩҲЩҮ ШЁШұ Ш§ЫҢЩҶШҢ Ш§ШЁШІШ§ШұЩҮШ§ЫҢЫҢ Щ…Ш§ЩҶЩҶШҜ .NET Fakes ШЁШұШ§ЫҢ ШҙШЁЫҢЩҮвҖҢШіШ§ШІЫҢ ШІЩ…Ш§ЩҶ ЩҲШ¬ЩҲШҜ ШҜШ§ШҙШӘЩҶШҜШҢ Ш§Щ…Ш§ Ш§ЫҢЩҶ Ш§ШЁШІШ§ШұЩҮШ§ ШӘЩҶЩҮШ§ ШҜШұ ЩҶШіШ®ЩҮ Enterprise ЩҲЫҢЪҳЩҲШ§Щ„ Ш§ШіШӘЩҲШҜЫҢЩҲ ШҜШұ ШҜШіШӘШұШі ШЁЩҲШҜЩҶШҜ ЩҲ Щ…ШӯШҜЩҲШҜЫҢШӘвҖҢЩҮШ§ЫҢЫҢ Щ…Ш§ЩҶЩҶШҜ ЩҲШ§ШЁШіШӘЪҜЫҢ ШЁЩҮ ЩҲЫҢЩҶШҜЩҲШІ ШҜШ§ШҙШӘЩҶШҜ.

Щ…Ш№ШұЩҒЫҢ TimeProvider ЩҲ ЩӮШ§ШЁЩ„ЫҢШӘвҖҢЩҮШ§ЫҢ ШўЩҶ
TimeProvider ЫҢЪ© Ъ©Щ„Ш§Ші Ш§ЩҶШӘШІШ§Ш№ЫҢ (Abstract Class) Ш§ШіШӘ Ъ©ЩҮ ШҜШұ .NET 8 Щ…Ш№ШұЩҒЫҢ ШҙШҜЩҮ ЩҲ ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ ЫҢЪ© Ш¬Ш§ЫҢЪҜШІЫҢЩҶ Ш§ШіШӘШ§ЩҶШҜШ§ШұШҜ ШЁШұШ§ЫҢ Щ…ШҜЫҢШұЫҢШӘ ШІЩ…Ш§ЩҶ Ш№Щ…Щ„ Щ…ЫҢвҖҢЪ©ЩҶШҜ. Ш§ЫҢЩҶ Ъ©Щ„Ш§Ші ШҙШ§Щ…Щ„ Щ…ШӘШҜЩҮШ§ЫҢЫҢ Щ…Ш§ЩҶЩҶШҜ GetUtcNow ШЁШұШ§ЫҢ ШҜШұЫҢШ§ЩҒШӘ ШІЩ…Ш§ЩҶ UTC ЩҲ GetLocalNow ШЁШұШ§ЫҢ ШІЩ…Ш§ЩҶ Щ…ШӯЩ„ЫҢШҢ ЩҲ CreateTimer ШЁШұШ§ЫҢ Ш§ЫҢШ¬Ш§ШҜ ШӘШ§ЫҢЩ…Шұ Ш§ШіШӘ. ЩҶШіШ®ЩҮ ЩҫЫҢШҙвҖҢЩҒШұШ¶ ШўЩҶШҢ ЫҢШ№ЩҶЫҢ TimeProvider.SystemШҢ Ш§ШІ ШІЩ…Ш§ЩҶ ШіЫҢШіШӘЩ… Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶШҜШҢ Ш§Щ…Ш§ Ш§Щ…Ъ©Ш§ЩҶ Ш§ЫҢШ¬Ш§ШҜ ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢвҖҢЩҮШ§ЫҢ ШіЩҒШ§ШұШҙЫҢ (Щ…Ш§ЩҶЩҶШҜ FakeTimeProvider) ШЁШұШ§ЫҢ ШӘШіШӘ ЩҲШ¬ЩҲШҜ ШҜШ§ШұШҜ. Ш§ЫҢЩҶ Ш§ШЁШІШ§Шұ ШЁЩҮ ЩҮЩ…ШұШ§ЩҮ ШЁШіШӘЩҮ Microsoft.Extensions.TimeProvider.Testing Ш§ШұШ§ШҰЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ Ъ©ЩҮ ШҙШ§Щ…Щ„ FakeTimeProvider Ш§ШіШӘ ЩҲ ШЁЩҮ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ Ш§Ш¬Ш§ШІЩҮ Щ…ЫҢвҖҢШҜЩҮШҜ ШІЩ…Ш§ЩҶ ШұШ§ ШҜШұ ШӘШіШӘвҖҢЩҮШ§ ШЁЩҮ ШҜЩ„Ш®ЩҲШ§ЩҮ ШӘЩҶШёЫҢЩ… Ъ©ЩҶЩҶШҜ.
ШіШ§Ш®ШӘШ§Шұ Ш§ШөЩ„ЫҢ TimeProvider ШЁЩҮ Ш§ЫҢЩҶ ШөЩҲШұШӘ Ш§ШіШӘ:
public abstract class TimeProvider
{
    public static TimeProvider System { get; }
    public virtual DateTimeOffset GetUtcNow();
    public DateTimeOffset GetLocalNow();
    public virtual TimeZoneInfo LocalTimeZone { get; }
    public virtual ITimer CreateTimer(TimerCallback callback, object? state, TimeSpan dueTime, TimeSpan period);
}

Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ TimeProvider ШҜШұ Ъ©ШҜ ЩҲ ШӘШіШӘ
ШЁШұШ§ЫҢ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ TimeProviderШҢ Ъ©Ш§ЩҒЫҢ Ш§ШіШӘ ШўЩҶ ШұШ§ Ш§ШІ Ш·ШұЫҢЩӮ ШӘШІШұЫҢЩӮ ЩҲШ§ШЁШіШӘЪҜЫҢ ШЁЩҮ Ъ©Щ„Ш§Ші Ш®ЩҲШҜ ЩҲШ§ШұШҜ Ъ©ЩҶЫҢШҜ. ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ Щ…Ш«Ш§Щ„ШҢ ЩҒШұШ¶ Ъ©ЩҶЫҢШҜ Щ…ЫҢвҖҢШ®ЩҲШ§ЩҮЫҢЩ… ШіШұЩҲЫҢШіЫҢ Ш·ШұШ§ШӯЫҢ Ъ©ЩҶЫҢЩ… Ъ©ЩҮ ШЁШұ Ш§ШіШ§Ші ШіШ§Ш№ШӘ ШұЩҲШІ ЩҫЫҢШ§Щ… Ш®ЩҲШҙШ§Щ…ШҜЪҜЩҲЫҢЫҢ ШӘЩҲЩ„ЫҢШҜ Ъ©ЩҶШҜ:
public class GreetingService
{
    private readonly TimeProvider _timeProvider;
    public GreetingService(TimeProvider timeProvider)
    {
        _timeProvider = timeProvider;
    }

    public string GetGreeting()
    {
        var hour = _timeProvider.GetLocalNow().Hour;
        return hour < 12 ? "ШөШЁШӯ ШЁШ®ЫҢШұ" : "Ш№ШөШұ ШЁШ®ЫҢШұ";
    }
}
ШҜШұ Щ…ШӯЫҢШ· ЩҲШ§ЩӮШ№ЫҢШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЫҢШҜ TimeProvider.System ШұШ§ ШӘШІШұЫҢЩӮ Ъ©ЩҶЫҢШҜ. Ш§Щ…Ш§ ШҜШұ ШӘШіШӘвҖҢЩҮШ§ШҢ Ш§ШІ FakeTimeProvider Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ… ШӘШ§ ШІЩ…Ш§ЩҶ ШұШ§ Ъ©ЩҶШӘШұЩ„ Ъ©ЩҶЫҢЩ…. ЩҶЩ…ЩҲЩҶЩҮвҖҢШ§ЫҢ Ш§ШІ ШӘШіШӘ ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ xUnit:
public class GreetingServiceTests
{
    [Fact]
    public void GetGreeting_Morning_ReturnsMorningGreeting()
    {
        var fakeTimeProvider = new FakeTimeProvider();
        fakeTimeProvider.SetUtcNow(new DateTime(2025, 3, 31, 8, 0, 0, DateTimeKind.Utc)); // 8 ШөШЁШӯ UTC
        fakeTimeProvider.SetLocalTimeZone(TimeZoneInfo.FindSystemTimeZoneById("Iran Standard Time"));

        var service = new GreetingService(fakeTimeProvider);
        var result = service.GetGreeting();

public class TimerService
{
    private readonly TimeProvider _timeProvider;
    public bool HasTimerTriggered { get; private set; }

    public TimerService(TimeProvider timeProvider)
    {
        _timeProvider = timeProvider;
        _timeProvider.CreateTimer(_ => HasTimerTriggered = true, null, TimeSpan.FromSeconds(5), Timeout.InfiniteTimeSpan);
    }
}

public class TimerServiceTests
{
    [Fact]
    public void Timer_TriggersAfter5Seconds()
    {
        var fakeTimeProvider = new FakeTimeProvider();
        var startTime = new DateTime(2025, 3, 31, 10, 0, 0);
        fakeTimeProvider.SetUtcNow(startTime);

        var service = new TimerService(fakeTimeProvider);
        Assert.False(service.HasTimerTriggered);

        fakeTimeProvider.SetUtcNow(startTime.AddSeconds(6));
        Assert.True(service.HasTimerTriggered);
    }
}
