Щ…ЩӮШҜЩ…ЩҮ
ШҜШұ ШҜЩҶЫҢШ§ЫҢ ШӘЩҲШіШ№ЩҮ ЩҶШұЩ…вҖҢШ§ЩҒШІШ§ШұШҢ ШӘШіШӘ ЩҲШ§ШӯШҜ (Unit Testing) ЫҢЪ©ЫҢ Ш§ШІ ШіШӘЩҲЩҶвҖҢЩҮШ§ЫҢ Ш§ШөЩ„ЫҢ ШӘШ¶Щ…ЫҢЩҶ Ъ©ЫҢЩҒЫҢШӘ Ъ©ШҜ Ш§ШіШӘ. ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ NET. Ш§ШәЩ„ШЁ Ш§ШІ ШӘШіШӘвҖҢЩҮШ§ЫҢ ЩҲШ§ШӯШҜ ШЁШұШ§ЫҢ Ш§Ш№ШӘШЁШ§ШұШіЩҶШ¬ЫҢ ШұЩҒШӘШ§Шұ Ш§Ш¬ШІШ§ЫҢ Ъ©ЩҲЪҶЪ© ШЁШұЩҶШ§Щ…ЩҮ Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶЩҶШҜ. ШЁШ§ Ш§ЫҢЩҶ ШӯШ§Щ„ШҢ ЪҜШ§ЩҮЫҢ Ш§ЩҲЩӮШ§ШӘ ШұЩҲШҙвҖҢЩҮШ§ЫҢ ЩҶШ§ШҜШұШіШӘ ШҜШұ ЩҶЩҲШҙШӘЩҶ Ш§ЫҢЩҶ ШӘШіШӘвҖҢЩҮШ§ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ ШЁЩҮ Щ…ШҙЪ©Щ„Ш§ШӘ Ш¬ШҜЫҢШҜЫҢ Щ…ЩҶШ¬Шұ ШҙЩҲШҜ. ЫҢЪ©ЫҢ Ш§ШІ Ш§ЫҢЩҶ Щ…ШҙЪ©Щ„Ш§ШӘШҢ Ш§Щ„ЪҜЩҲЫҢ Ш¶ШҜ ШӘШіШӘ ЩҲШ§ШӯШҜ ШЁЩҮ ЩҶШ§Щ… "ШўЩ„ЩҲШҜЪҜЫҢ Ъ©ШҜ" (Code Pollution) Ш§ШіШӘ. ШўЩ„ЩҲШҜЪҜЫҢ Ъ©ШҜ ШІЩ…Ш§ЩҶЫҢ ШұШ® Щ…ЫҢвҖҢШҜЩҮШҜ Ъ©ЩҮ Щ…ЩҶШ·ЩӮ Ш®Ш§ШөЫҢ ЩҒЩӮШ· ШЁШұШ§ЫҢ ШӘШіЩҮЫҢЩ„ ШӘШіШӘ ШҜШұ Ъ©ШҜ Ш§ШөЩ„ЫҢ (Production Code) Ш§Ш¶Ш§ЩҒЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ. Ш§ЫҢЩҶ Ъ©Ш§Шұ ЩҶЩҮвҖҢШӘЩҶЩҮШ§ ЩҫЫҢЪҶЫҢШҜЪҜЫҢ ШәЫҢШұШ¶ШұЩҲШұЫҢ Ш§ЫҢШ¬Ш§ШҜ Щ…ЫҢвҖҢЪ©ЩҶШҜШҢ ШЁЩ„Ъ©ЩҮ ЩҲШ§ШЁШіШӘЪҜЫҢ ШЁЫҢЩҶ Ъ©ШҜ Ш§ШөЩ„ЫҢ ЩҲ ШӘШіШӘвҖҢЩҮШ§ ШұШ§ Ш§ЩҒШІШ§ЫҢШҙ Щ…ЫҢвҖҢШҜЩҮШҜ ЩҲ ЩҶЪҜЩҮШҜШ§ШұЫҢ ЩҫШұЩҲЪҳЩҮ ШұШ§ ШҜШҙЩҲШ§ШұШӘШұ Щ…ЫҢвҖҢЪ©ЩҶШҜ. ШҜШұ Ш§ЫҢЩҶ Щ…ЩӮШ§Щ„ЩҮШҢ ШЁЩҮ ШЁШұШұШіЫҢ Щ…ЩҒЩҮЩҲЩ… ШўЩ„ЩҲШҜЪҜЫҢ Ъ©ШҜШҢ ШҜЩ„Ш§ЫҢЩ„ ЩҲЩӮЩҲШ№ ШўЩҶШҢ ЩҲ ШұШ§ЩҮЪ©Ш§ШұЩҮШ§ЫҢЫҢ ШЁШұШ§ЫҢ Ш§Ш¬ШӘЩҶШ§ШЁ Ш§ШІ ШўЩҶ ШҜШұ ШӘЩҲШіШ№ЩҮ NET. Щ…ЫҢвҖҢЩҫШұШҜШ§ШІЫҢЩ…. ЩҮШҜЩҒ Ш§ЫҢЩҶ Ш§ШіШӘ Ъ©ЩҮ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ ШЁШӘЩҲШ§ЩҶЩҶШҜ Ъ©ШҜЩҮШ§ЫҢЫҢ ШӘЩ…ЫҢШІШӘШұШҢ Щ…ШіШӘЩӮЩ„вҖҢШӘШұ ЩҲ ЩӮШ§ШЁЩ„ ЩҶЪҜЩҮШҜШ§ШұЫҢвҖҢШӘШұ ШЁЩҶЩҲЫҢШіЩҶШҜ.

ШўЩ„ЩҲШҜЪҜЫҢ Ъ©ШҜ ЪҶЫҢШіШӘ ЩҲ ЪҶШұШ§ Щ…ШҙЪ©Щ„вҖҢШіШ§ШІ Ш§ШіШӘШҹ
ШўЩ„ЩҲШҜЪҜЫҢ Ъ©ШҜ ШЁЩҮ ЩҲШ¶Ш№ЫҢШӘЫҢ Ш§ШҙШ§ШұЩҮ ШҜШ§ШұШҜ Ъ©ЩҮ ШҜШұ ШўЩҶШҢ Ъ©ШҜ Ш§ШөЩ„ЫҢ ШЁЩҮ Ш®Ш§Ш·Шұ ЩҶЫҢШ§ШІЩҮШ§ЫҢ ШӘШіШӘ ШӘШәЫҢЫҢШұ Щ…ЫҢвҖҢЪ©ЩҶШҜ ЫҢШ§ ШЁШ§ Щ…ЩҶШ·ЩӮ Ш§Ш¶Ш§ЩҒЫҢ ЩҫШұ Щ…ЫҢвҖҢШҙЩҲШҜ. Ш§ЫҢЩҶ ШӘШәЫҢЫҢШұШ§ШӘ Щ…Ш№Щ…ЩҲЩ„Ш§ЩӢ ШҙШ§Щ…Щ„ Ш§Ш¶Ш§ЩҒЩҮ Ъ©ШұШҜЩҶ ШіЩҲШҰЫҢЪҶвҖҢЩҮШ§ (Switches)ШҢ Щ…ШӘШҜЩҮШ§ ЫҢШ§ ШҙШұШ·вҖҢЩҮШ§ЫҢЫҢ Ш§ШіШӘ Ъ©ЩҮ ЩҒЩӮШ· ШҜШұ Щ…ШӯЫҢШ· ШӘШіШӘ Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢШҙЩҲЩҶШҜ. Ш§ЫҢЩҶ Ш§Щ„ЪҜЩҲ ШЁШ§Ш№Ш« Щ…ЫҢвҖҢШҙЩҲШҜ Ъ©ШҜ Ш§ШөЩ„ЫҢ Ш§ШІ ЩҮШҜЩҒ Ш§ШөЩ„ЫҢ Ш®ЩҲШҜШҢ ЫҢШ№ЩҶЫҢ Ш§Ш¬ШұШ§ЫҢ Щ…ЩҶШ·ЩӮ Ъ©ШіШЁвҖҢЩҲЪ©Ш§ШұШҢ ШҜЩҲШұ ШҙЩҲШҜ ЩҲ ШЁЩҮ Ш§ШЁШІШ§ШұЫҢ ШЁШұШ§ЫҢ ЩҫШҙШӘЫҢШЁШ§ЩҶЫҢ Ш§ШІ ШӘШіШӘвҖҢЩҮШ§ ШӘШЁШҜЫҢЩ„ ШҙЩҲШҜ.
ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ Щ…Ш«Ш§Щ„ШҢ ЩҒШұШ¶ Ъ©ЩҶЫҢШҜ ЫҢЪ© Ъ©Щ„Ш§Ші ШіШ§ШҜЩҮ ШЁШұШ§ЫҢ Щ„Ш§ЪҜ Ъ©ШұШҜЩҶ ЩҫЫҢШ§Щ…вҖҢЩҮШ§ ШҜШұ NET. ШҜШ§ШұЫҢЩ…:
public class CustomLogger
{
    private readonly bool _isTestingMode;
    public CustomLogger(bool isTestingMode)
    {
        _isTestingMode = isTestingMode;
    }

    public void Log(string message)
    {
        if (_isTestingMode) return;
        // Щ…ЩҶШ·ЩӮ Щ„Ш§ЪҜ Ъ©ШұШҜЩҶ ЩҫЫҢШ§Щ… ШҜШұ Ш§ЫҢЩҶШ¬Ш§
    }
}

public class ServiceController
{
    public void Execute(CustomLogger logger)
    {
        logger.Log("Execute method invoked");
    }
}
ШҜШұ Ш§ЫҢЩҶ Ъ©ШҜШҢ Щ…ШӘШәЫҢШұ _isTestingMode ЩҲ ШҙШұШ· Щ…ШұШӘШЁШ· ШЁШ§ ШўЩҶ ЩҒЩӮШ· ШЁШұШ§ЫҢ ШәЫҢШұЩҒШ№Ш§Щ„ Ъ©ШұШҜЩҶ Щ„Ш§ЪҜ ШҜШұ ШӘШіШӘвҖҢЩҮШ§ Ш§Ш¶Ш§ЩҒЩҮ ШҙШҜЩҮ Ш§ШіШӘ. Ш§ЫҢЩҶ ЫҢЪ© ЩҶЩ…ЩҲЩҶЩҮ Ъ©Щ„Ш§ШіЫҢЪ© Ш§ШІ ШўЩ„ЩҲШҜЪҜЫҢ Ъ©ШҜ Ш§ШіШӘ. Щ…ШҙЪ©Щ„ Ш§ЫҢЩҶШ¬Ш§ШіШӘ Ъ©ЩҮ Ш§ЫҢЩҶ Щ…ЩҶШ·ЩӮ ЩҮЫҢЪҶ Ш§ШұШІШҙЫҢ ШЁЩҮ Ш№Щ…Щ„Ъ©ШұШҜ Ш§ШөЩ„ЫҢ ШЁШұЩҶШ§Щ…ЩҮ Ш§Ш¶Ш§ЩҒЩҮ ЩҶЩ…ЫҢвҖҢЪ©ЩҶШҜ ЩҲ ШӘЩҶЩҮШ§ ШЁШұШ§ЫҢ ШӘШіШӘвҖҢЩҮШ§ Ш·ШұШ§ШӯЫҢ ШҙШҜЩҮ Ш§ШіШӘ. ЩҶШӘЫҢШ¬ЩҮШҹ Ъ©ШҜЫҢ Ъ©ЩҮ ЩҫЫҢЪҶЫҢШҜЩҮвҖҢШӘШұ Ш§ШІ ШӯШҜ Щ„Ш§ШІЩ… Ш§ШіШӘШҢ ЩҲШ§ШЁШіШӘЪҜЫҢ ЩҶШ§Ш®ЩҲШ§ШіШӘЩҮ ШЁЩҮ Щ…ШӯЫҢШ· ШӘШіШӘ ШҜШ§ШұШҜ ЩҲ ШҜШұ ШөЩҲШұШӘ ШӘШәЫҢЫҢШұ ЩҶЫҢШ§ШІЩҮШ§ЫҢ ШӘШіШӘШҢ ШЁШ§ЫҢШҜ ШЁШ§ШұЩҮШ§ ШЁШ§ШІЩҶЩҲЫҢШіЫҢ ШҙЩҲШҜ.

ШұЫҢШҙЩҮвҖҢЩҮШ§ЫҢ ШўЩ„ЩҲШҜЪҜЫҢ Ъ©ШҜ
ШўЩ„ЩҲШҜЪҜЫҢ Ъ©ШҜ Щ…Ш№Щ…ЩҲЩ„Ш§ЩӢ Ш§ШІ Ш№ШҜЩ… Ш§ШіШӘЩҒШ§ШҜЩҮ ШөШӯЫҢШӯ Ш§ШІ Ш§ШөЩҲЩ„ Ш·ШұШ§ШӯЫҢ ЩҶШұЩ…вҖҢШ§ЩҒШІШ§Шұ ЩҶШ§ШҙЫҢ Щ…ЫҢвҖҢШҙЩҲШҜ. ЫҢЪ©ЫҢ Ш§ШІ ШҙШ§ЫҢШ№вҖҢШӘШұЫҢЩҶ ШҜЩ„Ш§ЫҢЩ„ШҢ ЩҶШ§ШҜЫҢШҜЩҮ ЪҜШұЩҒШӘЩҶ Ш§ШөЩ„ ШӘШІШұЫҢЩӮ ЩҲШ§ШЁШіШӘЪҜЫҢ (Dependency Injection) Ш§ШіШӘ. ЩҲЩӮШӘЫҢ ЫҢЪ© Ъ©Щ„Ш§Ші ШЁЩҮвҖҢШ¬Ш§ЫҢ ЩҲШ§ШЁШіШӘЪҜЫҢ ШЁЩҮ ЫҢЪ© abstraction (Щ…Ш§ЩҶЩҶШҜ ШұШ§ШЁШ· ЫҢШ§ Interface)ШҢ Щ…ШіШӘЩӮЫҢЩ…Ш§ЩӢ ШЁЩҮ ЫҢЪ© ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ Ш®Ш§Шө ЩҲШ§ШЁШіШӘЩҮ ШЁШ§ШҙШҜШҢ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ Щ…Щ…Ъ©ЩҶ Ш§ШіШӘ ШЁШұШ§ЫҢ ШӘШіШӘ ШўЩҶШҢ Ъ©ШҜ Ш§ШөЩ„ЫҢ ШұШ§ ШӘШәЫҢЫҢШұ ШҜЩҮЩҶШҜ. ШҜЩ„ЫҢЩ„ ШҜЫҢЪҜШұШҢ Ш№ШҜЩ… ШӘЩ…Ш§ЫҢЩ„ ШЁЩҮ ШЁШ§ШІШ·ШұШ§ШӯЫҢ Ъ©ШҜ ШЁШұШ§ЫҢ Ш¬ШҜШ§ШіШ§ШІЫҢ ЩҶЪҜШұШ§ЩҶЫҢвҖҢЩҮШ§ (Separation of Concerns) Ш§ШіШӘ. ШҜШұ ЩҶЩҮШ§ЫҢШӘШҢ ЩҒШҙШ§Шұ ШІЩ…Ш§ЩҶЫҢ ЫҢШ§ Ш№ШҜЩ… ШўЪҜШ§ЩҮЫҢ Ш§ШІ ШұЩҲШҙвҖҢЩҮШ§ЫҢ ШЁЩҮШӘШұ ЩҶЫҢШІ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ ШЁЩҮ Ш§ЫҢЩҶ Щ…ШҙЪ©Щ„ ШҜШ§Щ…ЩҶ ШЁШІЩҶШҜ.

ШұШ§ЩҮЪ©Ш§ШұЩҮШ§ЫҢЫҢ ШЁШұШ§ЫҢ Ш§Ш¬ШӘЩҶШ§ШЁ Ш§ШІ ШўЩ„ЩҲШҜЪҜЫҢ Ъ©ШҜ
ШЁШұШ§ЫҢ ШұЩҒШ№ Ш§ЫҢЩҶ Щ…ШҙЪ©Щ„ШҢ ШЁШ§ЫҢШҜ Ш§ШІ ШӘЪ©ЩҶЫҢЪ©вҖҢЩҮШ§ЫҢЫҢ Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ЩҶЫҢЩ… Ъ©ЩҮ Ъ©ШҜ Ш§ШөЩ„ЫҢ ШұШ§ Ш§ШІ Щ…ЩҶШ·ЩӮ ШӘШіШӘ Ш¬ШҜШ§ ЩҶЪҜЩҮ ШҜШ§ШұЩҶШҜ. ШҜШұ Ш§ШҜШ§Щ…ЩҮШҢ ЪҶЩҶШҜ ШұШ§ЩҮЪ©Ш§Шұ Ъ©Щ„ЫҢШҜЫҢ ШұШ§ ШЁШұШұШіЫҢ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ…:

1. Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ШұШ§ШЁШ·вҖҢЩҮШ§ (Interfaces) ЩҲ ШӘШІШұЫҢЩӮ ЩҲШ§ШЁШіШӘЪҜЫҢ
ШЁЩҮ Ш¬Ш§ЫҢ Ш§ЩҒШІЩҲШҜЩҶ ШіЩҲШҰЫҢЪҶвҖҢЩҮШ§ЫҢ ШӘШіШӘ ШҜШұ Ъ©ШҜ Ш§ШөЩ„ЫҢШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЫҢЩ… Ш§ШІ ШұШ§ШЁШ·вҖҢЩҮШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ЩҶЫҢЩ… ШӘШ§ ЩҲШ§ШЁШіШӘЪҜЫҢвҖҢЩҮШ§ ШұШ§ Ш§ЩҶШӘШІШ§Ш№ЫҢ Ъ©ЩҶЫҢЩ…. ШЁШұШ§ЫҢ Щ…Ш«Ш§Щ„ШҢ Ъ©Щ„Ш§Ші CustomLogger ШұШ§ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ ШЁЩҮ Ш§ЫҢЩҶ ШҙЪ©Щ„ ШЁШ§ШІШ·ШұШ§ШӯЫҢ Ъ©ШұШҜ:
public interface ILogger
{
    void Log(string message);
}

public class RealLogger : ILogger
{
    public void Log(string message)
    {
        // Щ…ЩҶШ·ЩӮ ЩҲШ§ЩӮШ№ЫҢ Щ„Ш§ЪҜ Ъ©ШұШҜЩҶ
    }
}

public class FakeLogger : ILogger
{
    public void Log(string message)
    {
        // ШЁШҜЩҲЩҶ Ш№Щ…Щ„ЫҢШ§ШӘШҢ ШЁШұШ§ЫҢ ШӘШіШӘ
    }
}

public class ServiceController
{
    private readonly ILogger _logger;
    public ServiceController(ILogger logger)
    {
        _logger = logger;
    }

    public void Execute()
    {
        _logger.Log("Execute method invoked");
    }
}
ШҜШұ Ш§ЫҢЩҶ ШӯШ§Щ„ШӘШҢ ServiceController ШЁЩҮ ЫҢЪ© ШұШ§ШЁШ· (ILogger) ЩҲШ§ШЁШіШӘЩҮ Ш§ШіШӘШҢ ЩҶЩҮ ЫҢЪ© ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ Ш®Ш§Шө. ШҜШұ Щ…ШӯЫҢШ· ЩҲШ§ЩӮШ№ЫҢШҢ RealLogger ШӘШІШұЫҢЩӮ Щ…ЫҢвҖҢШҙЩҲШҜ ЩҲ ШҜШұ ШӘШіШӘвҖҢЩҮШ§ШҢ FakeLogger Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ. Ш§ЫҢЩҶ ШұЩҲШҙ ЩҶЩҮвҖҢШӘЩҶЩҮШ§ ШўЩ„ЩҲШҜЪҜЫҢ Ъ©ШҜ ШұШ§ ШӯШ°ЩҒ Щ…ЫҢвҖҢЪ©ЩҶШҜШҢ ШЁЩ„Ъ©ЩҮ Ш§ЩҶШ№Ш·Ш§ЩҒвҖҢЩҫШ°ЫҢШұЫҢ ЩҲ ЩӮШ§ШЁЩ„ЫҢШӘ ШӘШіШӘвҖҢЩҫШ°ЫҢШұЫҢ (Testability) ШұШ§ Ш§ЩҒШІШ§ЫҢШҙ Щ…ЫҢвҖҢШҜЩҮШҜ.

2. Ш¬ШҜШ§ШіШ§ШІЫҢ Щ…ЩҶШ·ЩӮ ШӘШіШӘ Ш§ШІ Ъ©ШҜ Ш§ШөЩ„ЫҢ
ЩҮШұЪҜШІ ЩҶШЁШ§ЫҢШҜ Ъ©ШҜ Ш§ШөЩ„ЫҢ ШұШ§ ШЁШұШ§ЫҢ ШӘШіШӘ ШӘШәЫҢЫҢШұ ШҜШ§ШҜ. Ш§ЪҜШұ ЩҶЫҢШ§ШІ ШЁЩҮ ШұЩҒШӘШ§Шұ Ш®Ш§ШөЫҢ ШҜШұ ШӘШіШӘ ШҜШ§ШұЫҢШҜШҢ Ш§ШІ Ш§ШЁШІШ§ШұЩҮШ§ЫҢЫҢ Щ…Ш«Щ„ mocking frameworks (Щ…Ш§ЩҶЩҶШҜ Moq ШҜШұ NET.) Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ЩҶЫҢШҜ. Ш§ЫҢЩҶ Ш§ШЁШІШ§ШұЩҮШ§ ШЁЩҮ ШҙЩ…Ш§ Ш§Ш¬Ш§ШІЩҮ Щ…ЫҢвҖҢШҜЩҮЩҶШҜ ШұЩҒШӘШ§ШұЩҮШ§ЫҢ ШҜЩ„Ш®ЩҲШ§ЩҮ ШұШ§ ШЁШҜЩҲЩҶ ШҜШіШӘЪ©Ш§ШұЫҢ Ъ©ШҜ Ш§ШөЩ„ЫҢ ШҙШЁЫҢЩҮвҖҢШіШ§ШІЫҢ Ъ©ЩҶЫҢШҜ.

3. ШұШ№Ш§ЫҢШӘ Ш§ШөЩҲЩ„ SOLID
Ш§ШөЩ„ ШЁШ§ШІ/ШЁШіШӘЩҮ (Open/Closed Principle) Ш§ШІ Щ…Ш¬Щ…ЩҲШ№ЩҮ Ш§ШөЩҲЩ„ SOLID ШЁЩҮ Щ…Ш§ Щ…ЫҢвҖҢЪҜЩҲЫҢШҜ Ъ©ЩҮ ШЁШ§ЫҢШҜ ШЁШӘЩҲШ§ЩҶЫҢЩ… ШұЩҒШӘШ§Шұ ЫҢЪ© Ъ©Щ„Ш§Ші ШұШ§ ШЁШҜЩҲЩҶ ШӘШәЫҢЫҢШұ Ъ©ШҜ ШўЩҶ ЪҜШіШӘШұШҙ ШҜЩҮЫҢЩ…. Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ШұШ§ШЁШ·вҖҢЩҮШ§ ЩҲ ШӘШІШұЫҢЩӮ ЩҲШ§ШЁШіШӘЪҜЫҢ Ш§ЫҢЩҶ Ш§ШөЩ„ ШұШ§ Щ…ШӯЩӮЩӮ Щ…ЫҢвҖҢЪ©ЩҶШҜ ЩҲ ЩҶЫҢШ§ШІ ШЁЩҮ ШўЩ„ЩҲШҜЪҜЫҢ Ъ©ШҜ ШұШ§ Ш§ШІ ШЁЫҢЩҶ Щ…ЫҢвҖҢШЁШұШҜ.

Щ…ШІШ§ЫҢШ§ЫҢ Ш§Ш¬ШӘЩҶШ§ШЁ Ш§ШІ ШўЩ„ЩҲШҜЪҜЫҢ Ъ©ШҜ
ЩҲЩӮШӘЫҢ Ш§ШІ ШўЩ„ЩҲШҜЪҜЫҢ Ъ©ШҜ Ш§Ш¬ШӘЩҶШ§ШЁ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ…ШҢ Ъ©ШҜ Ш§ШөЩ„ЫҢ ШӘЩ…ЫҢШІШӘШұ ЩҲ Щ…ШіШӘЩӮЩ„вҖҢШӘШұ Щ…ЫҢвҖҢШҙЩҲШҜ. Ш§ЫҢЩҶ Ъ©Ш§Шұ ЩҶЪҜЩҮШҜШ§ШұЫҢ ШұШ§ ШўШіШ§ЩҶвҖҢШӘШұ Щ…ЫҢвҖҢЪ©ЩҶШҜШҢ ШІЫҢШұШ§ ШӘШәЫҢЫҢШұШ§ШӘ ШҜШұ ШӘШіШӘвҖҢЩҮШ§ ЩҶЫҢШ§ШІЫҢ ШЁЩҮ ШЁШ§ШІЩҶЩҲЫҢШіЫҢ Ъ©ШҜ Ш§ШөЩ„ЫҢ ЩҶШҜШ§ШұЩҶШҜ. ЩҮЩ…ЪҶЩҶЫҢЩҶШҢ ЩӮШ§ШЁЩ„ЫҢШӘ ШӘШіШӘвҖҢЩҫШ°ЫҢШұЫҢ Ш§ЩҒШІШ§ЫҢШҙ Щ…ЫҢвҖҢЫҢШ§ШЁШҜ ЩҲ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЩҶШҜ ШЁШ§ Ш§Ш·Щ…ЫҢЩҶШ§ЩҶ ШЁЫҢШҙШӘШұЫҢ refactoring Ш§ЩҶШ¬Ш§Щ… ШҜЩҮЩҶШҜ. ШҜШұ ЩҶЩҮШ§ЫҢШӘШҢ Ш§ЫҢЩҶ ШұЩҲЫҢЪ©ШұШҜ ШЁЩҮ Ъ©Ш§ЩҮШҙ ШЁШ§ЪҜвҖҢЩҮШ§ЫҢ ЩҶШ§Ш®ЩҲШ§ШіШӘЩҮ ЩҶШ§ШҙЫҢ Ш§ШІ Щ…ЩҶШ·ЩӮ ШӘШіШӘ ШҜШұ Щ…ШӯЫҢШ· ЩҲШ§ЩӮШ№ЫҢ Ъ©Щ…Ъ© Щ…ЫҢвҖҢЪ©ЩҶШҜ.

ЩҶШӘЫҢШ¬ЩҮвҖҢЪҜЫҢШұЫҢ
ШўЩ„ЩҲШҜЪҜЫҢ Ъ©ШҜ ЫҢЪ© Ш§Щ„ЪҜЩҲЫҢ Ш¶ШҜ ШӘШіШӘ ЩҲШ§ШӯШҜ Ш§ШіШӘ Ъ©ЩҮ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ Ъ©ЫҢЩҒЫҢШӘ ЩҲ ЩҶЪҜЩҮШҜШ§ШұЫҢвҖҢЩҫШ°ЫҢШұЫҢ ЩҫШұЩҲЪҳЩҮвҖҢЩҮШ§ЫҢ NET. ШұШ§ ШЁЩҮ Ш®Ш·Шұ ШЁЫҢЩҶШҜШ§ШІШҜ. Ш§ЫҢЩҶ Щ…ШҙЪ©Щ„ ШІЩ…Ш§ЩҶЫҢ ШұШ® Щ…ЫҢвҖҢШҜЩҮШҜ Ъ©ЩҮ Щ…ЩҶШ·ЩӮ ШӘШіШӘШҢ ШЁЩҮ Ъ©ШҜ Ш§ШөЩ„ЫҢ ЩҶЩҒЩҲШ° Ъ©ЩҶШҜШҢ Щ…Ш§ЩҶЩҶШҜ Ш§ЩҒШІЩҲШҜЩҶ ШіЩҲШҰЫҢЪҶвҖҢЩҮШ§ ЫҢШ§ ШҙШұШ·вҖҢЩҮШ§ЫҢ ШәЫҢШұШ¶ШұЩҲШұЫҢ. ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ШұШ§ШЁШ·вҖҢЩҮШ§ШҢ ШӘШІШұЫҢЩӮ ЩҲШ§ШЁШіШӘЪҜЫҢШҢ ЩҲ Ш§ШЁШІШ§ШұЩҮШ§ЫҢ Щ…ЩҶШ§ШіШЁШҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ Ш§ШІ Ш§ЫҢЩҶ Ш§Щ„ЪҜЩҲ Ш§Ш¬ШӘЩҶШ§ШЁ Ъ©ШұШҜ ЩҲ Ъ©ШҜЫҢ ШӘЩ…ЫҢШІ ЩҲ ЩӮШ§ШЁЩ„ Ш§Ш№ШӘЩ…Ш§ШҜ ШӘЩҲЩ„ЫҢШҜ ЩҶЩ…ЩҲШҜ. ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ NET. ШЁШ§ЫҢШҜ ШЁЩҮ Ш§ЫҢЩҶ ЩҶЪ©ШӘЩҮ ШӘЩҲШ¬ЩҮ Ъ©ЩҶЩҶШҜ Ъ©ЩҮ ЩҮШҜЩҒ ШӘШіШӘ ЩҲШ§ШӯШҜШҢ Ш§Ш№ШӘШЁШ§ШұШіЩҶШ¬ЫҢ ШұЩҒШӘШ§Шұ Ъ©ШҜ Ш§ШіШӘШҢ ЩҶЩҮ ШӘШәЫҢЫҢШұ ШўЩҶ. ШЁШ§ ШұШ№Ш§ЫҢШӘ Ш§ЫҢЩҶ Ш§ШөЩҲЩ„ШҢ ЩҶЩҮвҖҢШӘЩҶЩҮШ§ Ъ©ЫҢЩҒЫҢШӘ Ъ©ШҜ ШЁЩҮШЁЩҲШҜ Щ…ЫҢвҖҢЫҢШ§ШЁШҜШҢ ШЁЩ„Ъ©ЩҮ ШӘШ¬ШұШЁЩҮ ШӘЩҲШіШ№ЩҮ ЩҶЫҢШІ ШӯШұЩҒЩҮвҖҢШ§ЫҢвҖҢШӘШұ ЩҲ Щ„Ш°ШӘвҖҢШЁШ®ШҙвҖҢШӘШұ Ш®ЩҲШ§ЩҮШҜ ШҙШҜ.

Щ…Ш·Ш§Щ„ШЁ Щ…ШҙШ§ШЁЩҮ
    Ш§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§Ш§Ш¬ШӘЩҶШ§ШЁ Ш§ШІ Ш§Щ„ЪҜЩҲЩҮШ§ЫҢ ЩҶШ§ШҜШұШіШӘ ШҜШұ ШӘШіШӘ ЩҲШ§ШӯШҜ ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ILogger ШҜШұ ШӘЩҲШіШ№ЩҮ .NETШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§Ъ©Ш§ЩҮШҙ ЩҲШ§ШЁШіШӘЪҜЫҢ ШЁЩҮ Ъ©ШӘШ§ШЁШ®Ш§ЩҶЩҮвҖҢЩҮШ§ЫҢ Ш«Ш§Щ„Ш« ШЁЩҮ Ъ©Щ…Ъ© ШӘШІШұЫҢЩӮ ЩҲШ§ШЁШіШӘЪҜЫҢвҖҢЩҮШ§Ш§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§Ш¬Щ„ЩҲЪҜЫҢШұЫҢ Ш§ШІ Ш§ЩҒШҙШ§ЫҢ ШӯШ§Щ„ШӘ Ш®ШөЩҲШөЫҢ ШҜШұ ШӘШіШӘвҖҢЩҮШ§ЫҢ ЩҲШ§ШӯШҜШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§ЪҶЩҮ ШІЩ…Ш§ЩҶвҖҢЩҮШ§ЫҢЫҢ ШЁЩҮШӘШұ Ш§ШіШӘ Ш§ШІ new ЩҲ ЫҢШ§ ШӘШІШұЫҢЩӮ ЩҲШ§ШЁШіШӘЪҜЫҢвҖҢЩҮШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ЩҶЫҢЩ…ШҹШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§ЪҶШұШ§ ЩҶШЁШ§ЫҢШҜ Щ…ШӘШҜЩҮШ§ЫҢ Ш®ШөЩҲШөЫҢ ШұШ§ ШҜШұ ШўШІЩ…ЩҲЩҶ ЩҲШ§ШӯШҜ ШӘШіШӘ Ъ©ЩҶЫҢЩ…ШҹШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§ШіШ§ШҜЩҮвҖҢШіШ§ШІЫҢ ШӘШіШӘ ЩҲШ§ШӯШҜ ШҜШұ ШіЫҢвҖҢШҙШ§ШұЩҫ ШЁШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ ЩҒШұЫҢЩ…вҖҢЩҲШұЪ© Щ…Ш§Ъ©ЫҢЩҶЪҜШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§ЩҫШұШ§Щ…ШӘвҖҢЩҮШ§ЫҢЫҢ ШЁШұШ§ЫҢ ЩҶЩҲШҙШӘЩҶ ШўШІЩ…ЩҲЩҶ ЩҲШ§ШӯШҜ ШЁЩҮ Ъ©Щ…Ъ© ЩҮЩҲШҙ Щ…ШөЩҶЩҲШ№ЫҢШ§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§ШЁЩҮШЁЩҲШҜ Ъ©ЫҢЩҒЫҢШӘ ШӘШіШӘвҖҢЩҶЩҲЫҢШіЫҢ ШЁШ§ Ш§Ш¬ШӘЩҶШ§ШЁ Ш§ШІ ШҙШЁЫҢЩҮвҖҢШіШ§ШІЫҢ (Mocking) Ъ©Щ„Ш§ШівҖҢЩҮШ§ЫҢ Щ…ШҙШ®Шө (Concrete Classes)Ш§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§Ш¬Щ„ЩҲЪҜЫҢШұЫҢ Ш§ШІ ЩҶШҙШӘ ШҜШ§ЩҶШҙ ШҜШ§Щ…ЩҶЩҮ ШҜШұ ШӘШіШӘвҖҢЩҮШ§ЫҢ ЩҲШ§ШӯШҜ ШҜШұ ШӘЩҲШіШ№ЩҮ .NETЩ…ШіЫҢШұШұШ§ЩҮвҖҢЩҮШ§ШұШ§ЩҮЪ©Ш§ШұЩҮШ§ЫҢЫҢ ШЁШұШ§ЫҢ ШЁЩҮШЁЩҲШҜ Ъ©ЫҢЩҒЫҢШӘ ШӘШіШӘвҖҢЩҮШ§ЫҢ ЩҲШ§ШӯШҜ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҮШ§ЫҢ ШҜШ§ШӘвҖҢЩҶШӘ
