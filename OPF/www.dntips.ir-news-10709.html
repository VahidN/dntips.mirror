var options = {
    onInit: function() {},
    onSpaceChange: function() {},
    spaceLimits: [0, 245, '#animate', '.spaces'],
    scrollbar: '.scrollbar',
    smoothEffect: true,
    initialSpace: 1,
    wheelStep: 150,
};
$('.scroll-container').doScroll(options);
