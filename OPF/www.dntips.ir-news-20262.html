Щ…Ш·Ш§Щ„ШЁ ЫіЫ¶Ы№Ы№
ЩҶЩҲЫҢШіЩҶШҜЪҜШ§ЩҶ ЫІЫ·Ы¶
ЪҜШұЩҲЩҮвҖҢЩҮШ§ЫҢ Щ…Ш·Ш§Щ„ШЁ ЫұЫ°ЫІЫө
ЩҶЩӮШҙЩҮвҖҢЩҮШ§ЫҢ ШұШ§ЩҮ ЫұЫұЫ№
ШҜЩҲШұЩҮвҖҢЩҮШ§ ЫұЫҙ
Ш§ШҙШӘШұШ§Ъ©вҖҢЩҮШ§ ЫұЫ·Ы№ЫұЫ·
→ вҖ«FOSSEDШӣ ЩҲШЁвҖҢШіШ§ЫҢШӘЫҢ ШЁШұШ§ЫҢ Ш§ШұШ§ШҰЩҮ ШұШ§ЩҮЩҶЩ…Ш§ЫҢЫҢ ШҜШұ Щ…ЩҲШұШҜ ЩҶШӯЩҲЩҮвҖҢЫҢ Щ…ЩҮШ§Ш¬ШұШӘ ШЁЩҮ Ъ©ШӘШ§ШЁШ®Ш§ЩҶЩҮвҖҢЩҮШ§ЫҢ ШұШ§ЫҢЪҜШ§ЩҶ ЩҲ Щ…ШӘЩҶвҖҢШЁШ§ШІ
вҖ«ЫҢЪ©ЩҫШ§ШұЪҶЩҮвҖҢШіШ§ШІЫҢ ШіШұЩҲШұ MCP GitHub ШҜШұ VS Code ШЁШұШ§ЫҢ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ .NET ←
Ш№ЩҶЩҲШ§ЩҶ:

вҖ«ШўШҙЩҶШ§ЫҢЫҢ ШЁШ§ Ш§Щ„ЪҜЩҲЫҢ Щ…Ш®ШІЩҶ ЩҶШҙШӘЫҢвҖҢШҜШ§Шұ ШҜШұ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҶЩҲЫҢШіЫҢ #C ЩҲ ШұЩҲШҙвҖҢЩҮШ§ЫҢ ШұЩҒШ№ ШўЩҶ


ЩҶЩҲЫҢШіЩҶШҜЩҮ: ЩҲШӯЫҢШҜ ЩҶШөЫҢШұЫҢ
ШӘШ§ШұЫҢШ®: ЫұЫҙЫ°Ыҙ/Ы°Ыұ/ЫІЫұ Ы°Ы·:ЫұЫё
ШўШҜШұШі: www.dntips.ir
Щ…ЩӮШҜЩ…ЩҮ
Ш§Щ„ЪҜЩҲЫҢ Щ…Ш®ШІЩҶ (Repository Pattern) ЫҢЪ©ЫҢ Ш§ШІ Ш§Щ„ЪҜЩҲЩҮШ§ЫҢ ЩҫШұЪ©Ш§ШұШЁШұШҜ ШҜШұ ШӘЩҲШіШ№ЩҮ ЩҶШұЩ…вҖҢШ§ЩҒШІШ§ШұЩҮШ§ЫҢ Щ…ШЁШӘЩҶЫҢ ШЁШұ #C ЩҲ ЩҒШұЫҢЩ…вҖҢЩҲШұЪ© .NET Ш§ШіШӘ Ъ©ЩҮ ШЁЩҮ Щ…ЩҶШёЩҲШұ Ш¬ШҜШ§ШіШ§ШІЫҢ Щ…ЩҶШ·ЩӮ Ъ©ШіШЁвҖҢЩҲЪ©Ш§Шұ Ш§ШІ Щ„Ш§ЫҢЩҮ ШҜШіШӘШұШіЫҢ ШЁЩҮ ШҜШ§ШҜЩҮвҖҢЩҮШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢШҙЩҲШҜ. Ш§ЫҢЩҶ Ш§Щ„ЪҜЩҲ ШЁШ§ ЩҮШҜЩҒ Ш§ЩҒШІШ§ЫҢШҙ ЩӮШ§ШЁЩ„ЫҢШӘ ШӘШіШӘвҖҢЩҫШ°ЫҢШұЫҢШҢ ЩҶЪҜЩҮШҜШ§ШұЫҢ ШўШіШ§ЩҶвҖҢШӘШұ Ъ©ШҜ ЩҲ Ш§ЫҢШ¬Ш§ШҜ Ш§ЩҶШ№Ш·Ш§ЩҒвҖҢЩҫШ°ЫҢШұЫҢ ШҜШұ Щ…Ш№Щ…Ш§ШұЫҢ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҮШ§ Ш·ШұШ§ШӯЫҢ ШҙШҜЩҮ Ш§ШіШӘ. ШЁШ§ Ш§ЫҢЩҶ ШӯШ§Щ„ШҢ ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ ЩҶШ§ШҜШұШіШӘ Ш§ЫҢЩҶ Ш§Щ„ЪҜЩҲ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ ШЁЩҮ Щ…ШҙЪ©Щ„Ш§ШӘ Щ…ШӘШ№ШҜШҜЫҢ Ш§ШІ Ш¬Щ…Щ„ЩҮ ЩҶШҙШӘ Ш¬ШІШҰЫҢШ§ШӘ ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ (Leaky Abstraction) Щ…ЩҶШ¬Шұ ШҙЩҲШҜ Ъ©ЩҮ ЩҲШ№ШҜЩҮвҖҢЩҮШ§ЫҢ Ш§ЩҲЩ„ЫҢЩҮ ШўЩҶ ШұШ§ ШІЫҢШұ ШіШӨШ§Щ„ Щ…ЫҢвҖҢШЁШұШҜ. ШҜШұ Ш§ЫҢЩҶ Щ…ЩӮШ§Щ„ЩҮ ШўЩ…ЩҲШІШҙЫҢШҢ Ш§ШЁШӘШҜШ§ ШЁЩҮ ШЁШұШұШіЫҢ Щ…ШҙЪ©Щ„Ш§ШӘ ШұШ§ЫҢШ¬ ШҜШұ ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ Ш§Щ„ЪҜЩҲЫҢ Щ…Ш®ШІЩҶ Щ…ЫҢвҖҢЩҫШұШҜШ§ШІЫҢЩ… ЩҲ ШіЩҫШі ШұШ§ЩҮЪ©Ш§ШұЩҮШ§ЫҢЫҢ Ш№Щ…Щ„ЫҢ ЩҲ ШӯШұЩҒЩҮвҖҢШ§ЫҢ ШұШ§ ШЁШұШ§ЫҢ ШЁЩҮЫҢЩҶЩҮвҖҢШіШ§ШІЫҢ ШўЩҶ Ш§ШұШ§ШҰЩҮ Щ…ЫҢвҖҢШҜЩҮЫҢЩ…. Ш§ЫҢЩҶ Щ…Ш·Щ„ШЁ ШЁШұШ§ЫҢ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶЫҢ Ъ©ЩҮ ШЁШ§ ЩҒШұЫҢЩ…вҖҢЩҲШұЪ© .NET Ъ©Ш§Шұ Щ…ЫҢвҖҢЪ©ЩҶЩҶШҜ ЩҲ ШЁЩҮ ШҜЩҶШЁШ§Щ„ Ш·ШұШ§ШӯЫҢ Ъ©ШҜ ШӘЩ…ЫҢШІ ЩҲ ЩӮШ§ШЁЩ„ Ш§Ш·Щ…ЫҢЩҶШ§ЩҶ ЩҮШіШӘЩҶШҜШҢ ШӘЩҮЫҢЩҮ ШҙШҜЩҮ Ш§ШіШӘ.

Щ…ШҙЪ©Щ„Ш§ШӘ ШұШ§ЫҢШ¬ ШҜШұ ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ Ш§Щ„ЪҜЩҲЫҢ Щ…Ш®ШІЩҶ

ЩҶШҙШӘ IQueryable ЩҲ Ш§ЩҶШ№Ш·Ш§ЩҒвҖҢЩҫШ°ЫҢШұЫҢ ШЁЫҢШҙ Ш§ШІ ШӯШҜ
ЫҢЪ©ЫҢ Ш§ШІ ШҙШ§ЫҢШ№вҖҢШӘШұЫҢЩҶ Ш§ШҙШӘШЁШ§ЩҮШ§ШӘ ШҜШұ Ш·ШұШ§ШӯЫҢ Щ…Ш®ШІЩҶШҢ ШЁШ§ШІЪҜШұШҜШ§ЩҶШҜЩҶ ЩҶЩҲШ№ IQueryable<T> Ш§ШІ Щ…ШӘШҜЩҮШ§ЫҢ Щ…Ш®ШІЩҶ Ш§ШіШӘ. Ш§ЫҢЩҶ Ъ©Ш§Шұ ШЁЩҮ ШёШ§ЩҮШұ Ш§ЩҶШ№Ш·Ш§ЩҒвҖҢЩҫШ°ЫҢШұЫҢ ШЁШ§Щ„Ш§ЫҢЫҢ ШЁЩҮ Щ…ШөШұЩҒвҖҢЪ©ЩҶЩҶШҜЪҜШ§ЩҶ Щ…ЫҢвҖҢШҜЩҮШҜШҢ Ш§Щ…Ш§ ШҜШұ Ш№Щ…Щ„ Ш¬ШІШҰЫҢШ§ШӘ ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ ШұШ§ Ш§ЩҒШҙШ§ Щ…ЫҢвҖҢЪ©ЩҶШҜ. ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ Щ…Ш«Ш§Щ„:
public interface IProductRepository
{
    IQueryable<Product> GetProducts();
}

public class ProductRepository : IProductRepository
{
    private readonly DbContext _context;
    
    public ProductRepository(DbContext context)
    {
        _context = context;
    }
    
    public IQueryable<Product> GetProducts()
    {
        return _context.Products;
    }
}
ШҜШұ Ш§ЫҢЩҶ Ъ©ШҜШҢ Щ…ШөШұЩҒвҖҢЪ©ЩҶЩҶШҜЩҮ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ Ъ©ЩҲШҰШұЫҢвҖҢЩҮШ§ЫҢЫҢ Щ…Ш§ЩҶЩҶШҜ ШІЫҢШұ ШЁЩҶЩҲЫҢШіШҜ:
var discountedProducts = repository.GetProducts()
    .Where(p => p.Category == "EBooks")
    .Where(p => p.Price > 100)
    .OrderBy(p => p.Name)
    .Skip(100)
    .Take(10)
    .ToList();
Щ…ШҙЪ©Щ„ Ш§ЫҢЩҶШ¬Ш§ШіШӘ Ъ©ЩҮ Щ…Ш®ШІЩҶ ЩҮЫҢЪҶ Ъ©ЩҶШӘШұЩ„ЫҢ ШЁШұ ЩҶШӯЩҲЩҮ Ш§Ш¬ШұШ§ЫҢ Ъ©ЩҲШҰШұЫҢ ЩҶШҜШ§ШұШҜ ЩҲ Щ…Щ…Ъ©ЩҶ Ш§ШіШӘ Ъ©ЩҲШҰШұЫҢвҖҢЩҮШ§ЫҢ ЩҶШ§Ъ©Ш§ШұШўЩ…ШҜ ЫҢШ§ ЩҲШ§ШЁШіШӘЩҮ ШЁЩҮ Ш§ШұШ§ШҰЩҮвҖҢШҜЩҮЩҶШҜЩҮ ШҜШ§ШҜЩҮ (Щ…Ш§ЩҶЩҶШҜ Entity Framework) Ш§ЫҢШ¬Ш§ШҜ ШҙЩҲШҜ. Ш§ЫҢЩҶ ЩҶШҙШӘШҢ Ш¬ШҜШ§ШіШ§ШІЫҢ Щ„Ш§ЫҢЩҮвҖҢЩҮШ§ ШұШ§ ЩҶЩӮШ¶ Щ…ЫҢвҖҢЪ©ЩҶШҜ.

ЩҲШ§ШЁШіШӘЪҜЫҢ ШЁЩҮ Щ…ЩҒШ§ЩҮЫҢЩ… Ш®Ш§Шө ЩҒШұЫҢЩ…вҖҢЩҲШұЪ©
ЪҜШ§ЩҮЫҢ Ш§ЩҲЩӮШ§ШӘ ШұШ§ШЁШ·вҖҢЩҮШ§ЫҢ Щ…Ш®ШІЩҶ ШЁЩҮ Щ…ЩҒШ§ЩҮЫҢЩ… Ш®Ш§Шө ЫҢЪ© ORM (Щ…Ш§ЩҶЩҶШҜ Entity Framework) ЩҲШ§ШЁШіШӘЩҮ Щ…ЫҢвҖҢШҙЩҲЩҶШҜ. ШЁШұШ§ЫҢ Щ…Ш«Ш§Щ„:
public interface IRepository<T> where T : class
{
    DbSet<T> Entities { get; }
    void Attach(T entity);
    void DetachEntity(T entity);
    EntityEntry<T> GetEntityEntry(T entity);
}
Ш§ЫҢЩҶ ШұШ§ШЁШ· ШЁЩҮ ЩҲШ¶ЩҲШӯ ШЁЩҮ Entity Framework ЩҲШ§ШЁШіШӘЩҮ Ш§ШіШӘ ЩҲ ШӘШәЫҢЫҢШұ Ш§ШұШ§ШҰЩҮвҖҢШҜЩҮЩҶШҜЩҮ ШҜШ§ШҜЩҮ ШЁШҜЩҲЩҶ ШЁШ§ШІЩҶЩҲЫҢШіЫҢ Ъ©ШҜ Щ…ШөШұЩҒвҖҢЪ©ЩҶЩҶШҜЩҮ ШәЫҢШұЩ…Щ…Ъ©ЩҶ Щ…ЫҢвҖҢШҙЩҲШҜ.

ЩҶШҙШӘ Ш§ШіШӘШ«ЩҶШ§ЩҮШ§
Ш№ШҜЩ… Щ…ШҜЫҢШұЫҢШӘ ШөШӯЫҢШӯ Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ЫҢ Щ„Ш§ЫҢЩҮ ШҜШ§ШҜЩҮ ЩҶЫҢШІ ЫҢЪ©ЫҢ ШҜЫҢЪҜШұ Ш§ШІ Щ…ШҙЪ©Щ„Ш§ШӘ ШұШ§ЫҢШ¬ Ш§ШіШӘ. ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ Щ…Ш«Ш§Щ„:
public Customer GetById(int id)
{
    try
    {
        return _context.Customers.Find(id);
    }
    catch (SqlException ex)
    {
        throw;
    }
}
ШҜШұ Ш§ЫҢЩҶШ¬Ш§ШҢ Ш§ШіШӘШ«ЩҶШ§ЫҢ SqlException Ъ©ЩҮ Ш®Ш§Шө SQL Server Ш§ШіШӘШҢ ШЁЩҮ Щ„Ш§ЫҢЩҮ ШЁШ§Щ„Ш§ШӘШұ ЩҶШҙШӘ Щ…ЫҢвҖҢЪ©ЩҶШҜ ЩҲ Щ…ШөШұЩҒвҖҢЪ©ЩҶЩҶШҜЩҮ ШұШ§ Щ…Ш¬ШЁЩҲШұ ШЁЩҮ ЩҲШ§ШЁШіШӘЪҜЫҢ ШЁЩҮ Ш¬ШІШҰЫҢШ§ШӘ ЩҫШ§ЫҢЪҜШ§ЩҮ ШҜШ§ШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶШҜ.

ШұЩҒШӘШ§ШұЩҮШ§ЫҢ ЩҶШ§Ш®ЩҲШ§ШіШӘЩҮ Щ…Ш§ЩҶЩҶШҜ ШЁШ§ШұЪҜШ°Ш§ШұЫҢ ШӘЩҶШЁЩ„ (Lazy Loading)
ЩҲЩӮШӘЫҢ Щ…Ш®ШІЩҶ Ш§ШҙЫҢШ§ШЎ ШұШ§ ШЁШ§ ЩҲЫҢЪҳЪҜЫҢвҖҢЩҮШ§ЫҢ ЩҶШ§ЩҲШЁШұЫҢ (Navigation Properties) ШЁШ§ШІЩ…ЫҢвҖҢЪҜШұШҜШ§ЩҶШҜШҢ Щ…ШөШұЩҒвҖҢЪ©ЩҶЩҶШҜЩҮ Щ…Щ…Ъ©ЩҶ Ш§ШіШӘ ЩҶШ§Ш®ЩҲШ§ШіШӘЩҮ ШЁШ§ Щ…ШҙЪ©Щ„Ш§ШӘЫҢ Щ…Ш§ЩҶЩҶШҜ "N+1 Query Problem" Щ…ЩҲШ§Ш¬ЩҮ ШҙЩҲШҜ:
public class OrderRepository : IOrderRepository
{
    private readonly DbContext _context;
    
    public Order GetById(int id)
    {
        return _context.Orders.Find(id);
    }
}

var order = orderRepository.GetById(1111);
foreach (var item in order.OrderItems)
{
    // Щ…Щ…Ъ©ЩҶ Ш§ШіШӘ Ъ©ЩҲШҰШұЫҢвҖҢЩҮШ§ЫҢ Ш§Ш¶Ш§ЩҒЫҢ Ш§Ш¬ШұШ§ ШҙЩҲШҜ
}
Ш§ЫҢЩҶ ШұЩҒШӘШ§Шұ ШЁЩҮ ШҜЩ„ЫҢЩ„ ШЁШ§ШұЪҜШ°Ш§ШұЫҢ ШӘЩҶШЁЩ„ ШҜШұ Entity Framework ШұШ® Щ…ЫҢвҖҢШҜЩҮШҜ ЩҲ Ш№Щ…Щ„Ъ©ШұШҜ ШЁШұЩҶШ§Щ…ЩҮ ШұШ§ ШӘШӯШӘ ШӘШЈШ«ЫҢШұ ЩӮШұШ§Шұ Щ…ЫҢвҖҢШҜЩҮШҜ.

ШұШ§ЩҮЪ©Ш§ШұЩҮШ§ЫҢ ШЁЩҮЫҢЩҶЩҮвҖҢШіШ§ШІЫҢ Ш§Щ„ЪҜЩҲЫҢ Щ…Ш®ШІЩҶ

ШЁШ§ШІЪҜШұШҜШ§ЩҶШҜЩҶ Щ…Ш¬Щ…ЩҲШ№ЩҮвҖҢЩҮШ§ЫҢ Щ…ШҙШ®Шө ШЁЩҮ Ш¬Ш§ЫҢ IQueryable
ШЁШұШ§ЫҢ Ш¬Щ„ЩҲЪҜЫҢШұЫҢ Ш§ШІ ЩҶШҙШӘШҢ ШЁЩҮШӘШұ Ш§ШіШӘ ШЁЩҮ Ш¬Ш§ЫҢ IQueryable Ш§ШІ Щ…Ш¬Щ…ЩҲШ№ЩҮвҖҢЩҮШ§ЫҢ Щ…ШҙШ®Шө Щ…Ш§ЩҶЩҶШҜ IReadOnlyCollection Ш§ШіШӘЩҒШ§ШҜЩҮ ШҙЩҲШҜ:
public interface IProductRepository
{
    IReadOnlyCollection<Product> GetProductsByCategory(string category);
    IReadOnlyCollection<Product> GetDiscountedProducts(decimal minPrice);
    PagedResult<Product> GetProductsPaged(int page, int pageSize, string sortField);
}

Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Ш§Щ„ЪҜЩҲЫҢ Щ…ШҙШ®ШөШ§ШӘ (Specification Pattern)
ШЁШұШ§ЫҢ Ъ©ЩҲШҰШұЫҢвҖҢЩҮШ§ЫҢ ЩҫЫҢЪҶЫҢШҜЩҮвҖҢШӘШұШҢ Ш§Щ„ЪҜЩҲЫҢ Щ…ШҙШ®ШөШ§ШӘ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ Ш¬Ш§ЫҢЪҜШІЫҢЩҶ Щ…ЩҶШ§ШіШЁЫҢ ШЁШ§ШҙШҜ:
public interface ISpecification<T>
{
    Expression<Func<T, bool>> Criteria { get; }
    List<Expression<Func<T, object>>> Includes { get; }
    Expression<Func<T, object>> OrderBy { get; }
    int Take { get; }
    int Skip { get; }
}

public interface IRepository<T>
{
    IReadOnlyList<T> FindAll(ISpecification<T> spec);
}

public class DiscountedEbooksSpecification : BaseSpecification<Product>
{
    public DiscountedEbooksSpecification(decimal minPrice)
        : base(p => p.Category == "Ebooks" && p.Price > minPrice)
    {
        AddOrderBy(p => p.Name);
        ApplyPaging(10, 10);
    }
}

var spec = new DiscountedEbooksSpecification(69);
var products = repository.FindAll(spec);

    }

public Order GetOrderWithDetails(int id)
{
    return _context.Orders
        .Include(o => o.OrderItems)
        .ThenInclude(i => i.Product)
        .FirstOrDefault(o => o.Id == id);
}

public OrderDetailsDto GetOrderDetails(int id)
{
    return _context.Orders
        .Where(o => o.Id == id)
        .Select(o => new OrderDetailsDto
        {
            OrderId = o.Id,
            CustomerName = o.Customer.Name,
            Items = o.OrderItems.Select(i => new OrderItemDto
            {
                ProductName = i.Product.Name,
                Quantity = i.Quantity
            }).ToList()
        })
        .FirstOrDefault();
}
