Щ…ЩӮШҜЩ…ЩҮ
ШҜШұ ШҜЩҶЫҢШ§ЫҢ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҶЩҲЫҢШіЫҢШҢ Щ…ШҜЫҢШұЫҢШӘ Ш®Ш·Ш§ЩҮШ§ ЫҢЪ©ЫҢ Ш§ШІ Ш¬ЩҶШЁЩҮвҖҢЩҮШ§ЫҢ ШӯЫҢШ§ШӘЫҢ ШӘЩҲШіШ№ЩҮ ЩҶШұЩ…вҖҢШ§ЩҒШІШ§ШұЩҮШ§ЫҢ ЩҫШ§ЫҢШҜШ§Шұ ЩҲ ЩӮШ§ШЁЩ„ Ш§Ш№ШӘЩ…Ш§ШҜ Ш§ШіШӘ. ШҜШұ ШІШЁШ§ЩҶ #CШҢ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ Щ…Ш№Щ…ЩҲЩ„Ш§ЩӢ Ш§ШІ Щ…Ъ©Ш§ЩҶЫҢШІЩ… Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ (Exceptions) ШЁШұШ§ЫҢ ШұШіЫҢШҜЪҜЫҢ ШЁЩҮ Ш®Ш·Ш§ЩҮШ§ Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢЪ©ЩҶЩҶШҜ. ШЁШ§ Ш§ЫҢЩҶ ШӯШ§Щ„ШҢ Ш§ШіШӘЩҒШ§ШҜЩҮ ШЁЫҢШҙ Ш§ШІ ШӯШҜ Ш§ШІ Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ ШЁШұШ§ЫҢ Ш®Ш·Ш§ЩҮШ§ЫҢ ЩӮШ§ШЁЩ„ ЩҫЫҢШҙвҖҢШЁЫҢЩҶЫҢ ЫҢШ§ ЩӮШ§ШЁЩ„ ШЁШ§ШІЫҢШ§ШЁЫҢ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ ШЁЩҮ ЩҫЫҢЪҶЫҢШҜЪҜЫҢ Ъ©ШҜШҢ Ъ©Ш§ЩҮШҙ Ш®ЩҲШ§ЩҶШ§ЫҢЫҢ ЩҲ Ш§ЩҒШӘ Ъ©Ш§ШұШ§ЫҢЫҢ Щ…ЩҶШ¬Шұ ШҙЩҲШҜ. Ш§Щ„ЪҜЩҲЫҢ Result ШұШ§ЩҮвҖҢШӯЩ„ЫҢ Ш¬Ш§ЫҢЪҜШІЫҢЩҶ ЩҲ ЩӮШҜШұШӘЩ…ЩҶШҜ Ш§ШұШ§ШҰЩҮ Щ…ЫҢвҖҢШҜЩҮШҜ Ъ©ЩҮ ШЁШ§ ШЁШ§ШІЪҜШұШҜШ§ЩҶШҜЩҶ ЫҢЪ© Щ…ЩӮШҜШ§Шұ ШіШ§Ш®ШӘШ§ШұЫҢШ§ЩҒШӘЩҮШҢ Ш§Щ…Ъ©Ш§ЩҶ Щ…ШҜЫҢШұЫҢШӘ ШөШұЫҢШӯ Щ…ЩҲЩҒЩӮЫҢШӘ ЫҢШ§ ШҙЪ©ШіШӘ Ш№Щ…Щ„ЫҢШ§ШӘ ШұШ§ ЩҒШұШ§ЩҮЩ… Щ…ЫҢвҖҢЪ©ЩҶШҜ. Ш§ЫҢЩҶ Ш§Щ„ЪҜЩҲ ШЁЩҮвҖҢЩҲЫҢЪҳЩҮ ШЁШұШ§ЫҢ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ .NET Ъ©ЩҮ ШЁЩҮ ШҜЩҶШЁШ§Щ„ Ъ©ШҜЫҢ ШӘЩ…ЫҢШІШӘШұШҢ ЩӮШ§ШЁЩ„ ЩҶЪҜЩҮШҜШ§ШұЫҢвҖҢШӘШұ ЩҲ ШЁШ§ Ъ©Ш§ШұШ§ЫҢЫҢ ШЁШ§Щ„Ш§ШӘШұ ЩҮШіШӘЩҶШҜШҢ Ш¬Ш°Ш§ШЁЫҢШӘ ШІЫҢШ§ШҜЫҢ ШҜШ§ШұШҜ.
ШҜШұ Ш§ЫҢЩҶ Щ…ЩӮШ§Щ„ЩҮШҢ ШЁЩҮ ШЁШұШұШіЫҢ Щ…ЩҒЩҮЩҲЩ… Ш§Щ„ЪҜЩҲЫҢ ResultШҢ Щ…ШІШ§ЫҢШ§ЫҢ ШўЩҶ ЩҶШіШЁШӘ ШЁЩҮ Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ШҢ ЩҲ ЩҶШӯЩҲЩҮ ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ ШўЩҶ ШҜШұ #C Щ…ЫҢвҖҢЩҫШұШҜШ§ШІЫҢЩ…. ЩҮЩ…ЪҶЩҶЫҢЩҶ ШЁШ§ Ш§ШұШ§ШҰЩҮ Щ…Ш«Ш§Щ„вҖҢЩҮШ§ЫҢ Ш№Щ…Щ„ЫҢШҢ ЩҶШҙШ§ЩҶ Щ…ЫҢвҖҢШҜЩҮЫҢЩ… Ъ©ЩҮ ЪҶЪҜЩҲЩҶЩҮ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ Ш§ШІ Ш§ЫҢЩҶ Ш§Щ„ЪҜЩҲ ШҜШұ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҮШ§ЫҢ ASP.NET Core ШЁЩҮШұЩҮ ШЁШұШҜ. ЩҮШҜЩҒ Ш§ЫҢЩҶ Ш§ШіШӘ Ъ©ЩҮ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ ШЁШӘЩҲШ§ЩҶЩҶШҜ Ш§ЫҢЩҶ Ш§Щ„ЪҜЩҲ ШұШ§ ШЁЩҮ ШҙЪ©Щ„ЫҢ Щ…ЩҶШіШ¬Щ… ШҜШұ ЩҫШұЩҲЪҳЩҮвҖҢЩҮШ§ЫҢ Ш®ЩҲШҜ ШЁЩҮ Ъ©Ш§Шұ ЪҜЫҢШұЩҶШҜ.

Ш§Щ„ЪҜЩҲЫҢ Result ЪҶЫҢШіШӘШҹ
Ш§Щ„ЪҜЩҲЫҢ Result ЫҢЪ© ШұЩҲШҙ ШіШ§Ш®ШӘШ§ШұЫҢШ§ЩҒШӘЩҮ ШЁШұШ§ЫҢ ЩҶШҙШ§ЩҶ ШҜШ§ШҜЩҶ ЩҶШӘЫҢШ¬ЩҮ ЫҢЪ© Ш№Щ…Щ„ЫҢШ§ШӘ Ш§ШіШӘ Ъ©ЩҮ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ Щ…ЩҲЩҒЩӮЫҢШӘвҖҢШўЩ…ЫҢШІ (Success) ЫҢШ§ ЩҶШ§Щ…ЩҲЩҒЩӮ (Failure) ШЁШ§ШҙШҜ. ШЁЩҮ Ш¬Ш§ЫҢ ЩҫШұШӘШ§ШЁ Ш§ШіШӘШ«ЩҶШ§ШҢ Ш§ЫҢЩҶ Ш§Щ„ЪҜЩҲ ЫҢЪ© ШҙЫҢШЎ ШұШ§ ШЁШ§ШІЩ…ЫҢвҖҢЪҜШұШҜШ§ЩҶШҜ Ъ©ЩҮ ЩҲШ¶Ш№ЫҢШӘ Ш№Щ…Щ„ЫҢШ§ШӘ ЩҲ Ш§Ш·Щ„Ш§Ш№Ш§ШӘ Щ…ШұШӘШЁШ· ШЁШ§ ШўЩҶ (Щ…Ш§ЩҶЩҶШҜ Щ…ЩӮШҜШ§Шұ ШЁШ§ШІЪҜШҙШӘЫҢ ЫҢШ§ ЩҫЫҢШ§Щ… Ш®Ш·Ш§) ШұШ§ Щ…ШҙШ®Шө Щ…ЫҢвҖҢЪ©ЩҶШҜ. Ш§ЫҢЩҶ ШұЩҲЫҢЪ©ШұШҜ Ш§ШІ Щ…ЩҒШ§ЩҮЫҢЩ… ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҶЩҲЫҢШіЫҢ ШӘШ§ШЁШ№ЫҢ (Functional Programming) Ш§Щ„ЩҮШ§Щ… ЪҜШұЩҒШӘЩҮ ШҙШҜЩҮ ЩҲ ШЁЩҮ ШӘЩҲШіШ№ЩҮвҖҢШҜЩҮЩҶШҜЪҜШ§ЩҶ Ш§Ш¬Ш§ШІЩҮ Щ…ЫҢвҖҢШҜЩҮШҜ Ш¬ШұЫҢШ§ЩҶ Ъ©ШҜ ШұШ§ ШЁЩҮ ШҙЪ©Щ„ЫҢ ЩҲШ§Ш¶ШӯвҖҢШӘШұ Щ…ШҜЫҢШұЫҢШӘ Ъ©ЩҶЩҶШҜ.

ЪҶШұШ§ Ш§ШІ Ш§Щ„ЪҜЩҲЫҢ Result Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ЩҶЫҢЩ…Шҹ
Ш§ШіШӘЩҒШ§ШҜЩҮ Ш§ШІ Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ ШҜШұ #C ШЁШұШ§ЫҢ Ш®Ш·Ш§ЩҮШ§ЫҢ ЩӮШ§ШЁЩ„ Ш§ЩҶШӘШёШ§ШұШҢ Щ…ШҙЪ©Щ„Ш§ШӘЫҢ ШЁЩҮ ЩҮЩ…ШұШ§ЩҮ ШҜШ§ШұШҜ:
  • ЩҮШІЫҢЩҶЩҮ Ш№Щ…Щ„Ъ©ШұШҜЫҢ: ЩҫШұШӘШ§ШЁ ЩҲ ЪҜШұЩҒШӘЩҶ Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ (Throwing and Catching Exceptions) Ш№Щ…Щ„ЫҢШ§ШӘЫҢ ЩҫШұЩҮШІЫҢЩҶЩҮ Ш§ШіШӘ ЩҲ ШҜШұ ШЁШұЩҶШ§Щ…ЩҮвҖҢЩҮШ§ЫҢ ШЁШ§ ШӯШ¬Щ… ШЁШ§Щ„Ш§ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶШҜ Ъ©Ш§ШұШ§ЫҢЫҢ ШұШ§ Ъ©Ш§ЩҮШҙ ШҜЩҮШҜ.
  • ЩҫЫҢЪҶЫҢШҜЪҜЫҢ Ш¬ШұЫҢШ§ЩҶ Ъ©ШҜ: Ъ©ШҜ ЩҫШұ Ш§ШІ ШЁЩ„ЩҲЪ©вҖҢЩҮШ§ЫҢ try-catch Ш®ЩҲШ§ЩҶШҜЩҶ ЩҲ ЩҶЪҜЩҮШҜШ§ШұЫҢ ШўЩҶ ШұШ§ ШҜШҙЩҲШ§Шұ Щ…ЫҢвҖҢЪ©ЩҶШҜ.
  • Ш№ШҜЩ… ШҙЩҒШ§ЩҒЫҢШӘ: Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ ШЁЩҮ ЩҲШ¶ЩҲШӯ ЩҶШҙШ§ЩҶ ЩҶЩ…ЫҢвҖҢШҜЩҮЩҶШҜ Ъ©ЩҮ ЪҶЩҮ ЩҶШӘШ§ЫҢШ¬ЫҢ Ш§ШІ ЫҢЪ© Ш№Щ…Щ„ЫҢШ§ШӘ Щ…Щ…Ъ©ЩҶ Ш§ШіШӘ ШұШ® ШҜЩҮШҜ.

Ш§Щ„ЪҜЩҲЫҢ Result Ш§ЫҢЩҶ Щ…ШҙЪ©Щ„Ш§ШӘ ШұШ§ ШЁШ§ Щ…ЩҲШ§ШұШҜ ШІЫҢШұ ШЁШұШ·ШұЩҒ Щ…ЫҢвҖҢЪ©ЩҶШҜ:
  • ШҙЩҒШ§ЩҒЫҢШӘ: ЩҶШӘЫҢШ¬ЩҮ Щ…ЩҲЩҒЩӮЫҢШӘ ЫҢШ§ ШҙЪ©ШіШӘ ШЁЩҮ ШөЩҲШұШӘ ШөШұЫҢШӯ ШҜШұ ЩҶЩҲШ№ ШЁШ§ШІЪҜШҙШӘЫҢ Щ…ШҙШ®Шө Щ…ЫҢвҖҢШҙЩҲШҜ.
  • Ъ©Ш§ШұШ§ЫҢЫҢ: ШЁШ§ ШӯШ°ЩҒ Ш§ШіШӘШ«ЩҶШ§ЩҮШ§ ШЁШұШ§ЫҢ Ш®Ш·Ш§ЩҮШ§ЫҢ ЩҫЫҢШҙвҖҢШЁЫҢЩҶЫҢвҖҢШҙШҜЩҮШҢ ШіШұШЁШ§Шұ Ш№Щ…Щ„Ъ©ШұШҜЫҢ Ъ©Ш§ЩҮШҙ Щ…ЫҢвҖҢЫҢШ§ШЁШҜ.
  • Ш®ЩҲШ§ЩҶШ§ЫҢЫҢ: Ш¬ШұЫҢШ§ЩҶ Ъ©ШҜ Ш®Ш·ЫҢвҖҢШӘШұ ЩҲ ЩӮШ§ШЁЩ„ ЩҒЩҮЩ…вҖҢШӘШұ Щ…ЫҢвҖҢШҙЩҲШҜ.

ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ ЩҫШ§ЫҢЩҮ Ш§Щ„ЪҜЩҲЫҢ Result ШҜШұ #C
ШЁШұШ§ЫҢ ШҙШұЩҲШ№ШҢ ЫҢЪ© Ъ©Щ„Ш§Ші ЩҫШ§ЫҢЩҮ ШЁШұШ§ЫҢ Ш§Щ„ЪҜЩҲЫҢ Result ШӘШ№ШұЫҢЩҒ Щ…ЫҢвҖҢЪ©ЩҶЫҢЩ… Ъ©ЩҮ ШЁШӘЩҲШ§ЩҶШҜ Щ…ЩҲЩҒЩӮЫҢШӘ ЫҢШ§ ШҙЪ©ШіШӘ ШұШ§ ЩҶШҙШ§ЩҶ ШҜЩҮШҜ. ШҜШұ Щ…Ш«Ш§Щ„ ШІЫҢШұШҢ Ш§ШІ ЫҢЪ© ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ ШіШ§ШҜЩҮ Ш§ШіШӘЩҒШ§ШҜЩҮ ШҙШҜЩҮ Ш§ШіШӘ:
public class Result<T>
{
    public bool IsSuccess { get; }
    public T Value { get; }
    public string Error { get; }

    protected Result(bool isSuccess, T value, string error)
    {
        IsSuccess = isSuccess;
        Value = value;
        Error = error;
    }

    public static Result<T> Success(T value) => new Result<T>(true, value, null);
    public static Result<T> Failure(string error) => new Result<T>(false, default, error);
}
ШҜШұ Ш§ЫҢЩҶ Ъ©ШҜ:
  • IsSuccess ЩҲШ¶Ш№ЫҢШӘ Ш№Щ…Щ„ЫҢШ§ШӘ ШұШ§ ЩҶШҙШ§ЩҶ Щ…ЫҢвҖҢШҜЩҮШҜ.
  • Value Щ…ЩӮШҜШ§Шұ ШЁШ§ШІЪҜШҙШӘЫҢ ШҜШұ ШөЩҲШұШӘ Щ…ЩҲЩҒЩӮЫҢШӘ Ш§ШіШӘ.
  • Error ЩҫЫҢШ§Щ… Ш®Ш·Ш§ ШҜШұ ШөЩҲШұШӘ ШҙЪ©ШіШӘ ШұШ§ ЩҶЪҜЩҮ Щ…ЫҢвҖҢШҜШ§ШұШҜ.
  • Щ…ШӘШҜЩҮШ§ЫҢ Ш§ШіШӘШ§ШӘЫҢЪ© Success ЩҲ Failure ШЁШұШ§ЫҢ ШіШ§Ш®ШӘ ШіШұЫҢШ№ Ш§ШҙЫҢШ§ШЎ Result Ш§ШіШӘЩҒШ§ШҜЩҮ Щ…ЫҢвҖҢШҙЩҲЩҶШҜ.
Щ…Ш«Ш§Щ„ Ш§ШіШӘЩҒШ§ШҜЩҮ: 
public Result<int> Divide(int a, int b)
{
    if (b == 0)
        return Result<int>.Failure("ШӘЩӮШіЫҢЩ… ШЁШұ ШөЩҒШұ Щ…Щ…Ъ©ЩҶ ЩҶЫҢШіШӘ");
    return Result<int>.Success(a / b);
}

var result = Divide(10, 2);
if (result.IsSuccess)
    Console.WriteLine($"ЩҶШӘЫҢШ¬ЩҮ: {result.Value}");
else
    Console.WriteLine($"Ш®Ш·Ш§: {result.Error}");

public class Result<T>
{
    public bool IsSuccess { get; }
    public T Value { get; }
    public Problem Problem { get; }

    private Result(bool isSuccess, T value, Problem problem)
    {
        IsSuccess = isSuccess;
        Value = value;
        Problem = problem;
    }

    public static Result<T> Success(T value) => new Result<T>(true, value, null);
    public static Result<T> Failure(Problem problem) => new Result<T>(false, default, problem);
}

Ш§ЫҢЩҶ ЩҲЫҢЪҳЪҜЫҢ ШЁЩҮ ШҙЩ…Ш§ Ш§Щ…Ъ©Ш§ЩҶ Щ…ЫҢвҖҢШҜЩҮШҜ Ш§ЩҶЩҲШ§Ш№ Щ…Ш®ШӘЩ„ЩҒЫҢ Ш§ШІ ЩҶШӘШ§ЫҢШ¬ ШұШ§ ШЁЩҮ ШҙЪ©Щ„ЫҢ Ш§ЫҢЩ…ЩҶ ЩҲ Щ…Ш®ШӘШөШұ ШӘШ№ШұЫҢЩҒ Ъ©ЩҶЫҢШҜШҢ Ъ©ЩҮ Ш¬Ш§ЫҢЪҜШІЫҢЩҶЫҢ ШЁЩҲЩ…ЫҢ ШЁШұШ§ЫҢ ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢвҖҢЩҮШ§ЫҢ ШҜШіШӘЫҢ Ъ©ЩҶЩҲЩҶЫҢ Ш§ШіШӘ.

ЩҫЫҢШ§ШҜЩҮвҖҢШіШ§ШІЫҢ ШўЩҶ ШіШ§ШҜЩҮ Ш§ШіШӘ ЩҲ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶ ШўЩҶ ШұШ§ ШЁШ§ ЩҶЫҢШ§ШІЩҮШ§ЫҢ Ш®Ш§Шө ЩҫШұЩҲЪҳЩҮ ШӘШ·ШЁЫҢЩӮ ШҜШ§ШҜ.
